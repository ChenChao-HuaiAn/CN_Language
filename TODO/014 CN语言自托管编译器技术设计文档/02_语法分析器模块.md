## 3. 详细任务描述

### 3.1 AST节点设计

```cn
// src_cn/语法分析器/AST节点.cn
模块 语法分析器 {
公开:
    // AST节点基类
    类 AST节点 {
    保护:
        源位置 位置;
        
    公开:
        函数 获取位置() -> 源位置 {
            返回 位置;
        }
        
        虚拟 函数 接受(AST访问者* 访问者) -> 空类型;
    }
    
    // 程序节点
    类 程序节点 : AST节点 {
    私有:
        声明列表 声明集合;
        
    公开:
        函数 添加声明(声明节点* 声明) -> 空类型 {
            声明集合.添加(声明);
        }
        
        函数 获取声明数量() -> 整数 {
            返回 声明集合.数量;
        }
        
        函数 接受(AST访问者* 访问者) -> 空类型 重写 {
            访问者.访问程序(自身);
        }
    }
    
    // 表达式节点基类
    类 表达式节点 : AST节点 {
    保护:
        类型* 节点类型;
        
    公开:
        函数 获取类型() -> 类型* {
            返回 节点类型;
        }
    }
    
    // 二元表达式节点
    类 二元表达式 : 表达式节点 {
    私有:
        表达式节点* 左操作数;
        二元运算符 运算符;
        表达式节点* 右操作数;
        
    公开:
        函数 创建(表达式节点* 左, 二元运算符 运算, 表达式节点* 右) -> 二元表达式* {
            变量 节点 = 新建 二元表达式();
            节点.左操作数 = 左;
            节点.运算符 = 运算;
            节点.右操作数 = 右;
            返回 节点;
        }
        
        函数 接受(AST访问者* 访问者) -> 空类型 重写 {
            访问者.访问二元表达式(自身);
        }
    }
    
    // 函数声明节点
    类 函数声明 : 声明节点 {
    私有:
        字符串 名称;
        参数列表 参数;
        类型* 返回类型;
        语句块* 函数体;
        
    公开:
        函数 获取名称() -> 字符串 { 返回 名称; }
        函数 获取参数() -> 参数列表* { 返回 参数; }
        函数 获取返回类型() -> 类型* { 返回 返回类型; }
        函数 获取函数体() -> 语句块* { 返回 函数体; }
    }
}
```

### 3.3 核心解析逻辑

```cn
// 解析函数声明示例
函数 解析函数声明() -> 函数声明* {
    变量 起始位置 = 当前词元.位置;
    
    期望(词元类型.关键字_函数);
    
    // 解析函数名
    变量 名称 = 期望标识符();
    
    // 解析参数列表
    期望(词元类型.左括号);
    变量 参数 = 解析参数列表();
    期望(词元类型.右括号);
    
    // 解析返回类型
    变量 返回类型 = 无;
    如果 (当前词元.类型 == 词元类型.箭头 或
          当前词元.类型 == 词元类型.关键字_返回) {
        如果 (当前词元.类型 == 词元类型.箭头) {
            前进();
        } 否则 {
            前进();  // 跳过 '返回'
        }
        返回类型 = 解析类型();
    }
    
    // 解析函数体
    变量 函数体 = 解析语句块();
    
    返回 创建函数声明(名称, 参数, 返回类型, 函数体, 起始位置);
}

// 解析表达式示例（运算符优先级）
函数 解析表达式() -> 表达式节点* {
    返回 解析赋值表达式();
}

函数 解析赋值表达式() -> 表达式节点* {
    变量 表达式 = 解析或表达式();
    
    如果 (匹配(词元类型.赋值)) {
        变量 运算符 = 当前词元;
        前进();
        变量 右值 = 解析赋值表达式();
        表达式 = 创建赋值表达式(表达式, 运算符, 右值);
    }
    
    返回 表达式;
}
```
