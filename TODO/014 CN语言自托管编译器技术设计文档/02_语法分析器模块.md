## 1. 任务概述

### 1.1 任务目标

将现有C语言实现的语法分析器（`src/frontend/parser/`）使用CN语言重写，实现自托管编译器的语法分析功能。

### 1.2 功能要求

| 功能项 | 说明 | 优先级 |
|--------|------|--------|
| 程序解析 | 解析顶层声明（函数、结构体、枚举、模块等） | ⭐⭐⭐ 高 |
| 函数解析 | 解析函数声明、参数列表、返回类型 | ⭐⭐⭐ 高 |
| 语句解析 | 解析控制流语句、变量声明、表达式语句 | ⭐⭐⭐ 高 |
| 表达式解析 | 解析运算符优先级、函数调用、成员访问 | ⭐⭐⭐ 高 |
| 类型解析 | 解析基础类型、指针类型、数组类型 | ⭐⭐⭐ 高 |
| 错误恢复 | 语法错误时的恢复机制 | ⭐⭐ 中 |

- [x] 语法分析器能正确解析所有CN语言语法结构

### 3.2 解析器类设计

```cn
// src_cn/语法分析器/解析器.cn
模块 语法分析器 {
公开:
    类 解析器 {
    私有:
        词法分析器* 词法;
        词元 当前词元;
        词元 下一个词元;
        诊断管理器* 诊断;
        整数 错误计数;
        
    公开:
        函数 创建(词法分析器* 词法分析, 诊断管理器* 诊断器) -> 解析器* {
            变量 实例 = 分配内存(解析器大小) 作为 解析器*;
            实例.词法 = 词法分析;
            实例.诊断 = 诊断器;
            实例.错误计数 = 0;
            
            // 预读两个词元
            实例.当前词元 = 词法分析.下一个词元();
            实例.下一个词元 = 词法分析.下一个词元();
            
            返回 实例;
        }
        
        函数 销毁(解析器* 实例) -> 空类型 {
            释放内存(实例);
        }
        
        函数 解析程序() -> 程序节点* {
            变量 程序 = 创建程序节点();
            
            当 (当前词元.类型 != 词元类型.结束) {
                变量 声明 = 解析顶层声明();
                如果 (声明 != 无) {
                    程序.添加声明(声明);
                }
            }
            
            返回 程序;
        }
        
    私有:
        函数 解析顶层声明() -> 声明节点* {
            选择 当前词元.类型 {
                情况 词元类型.关键字_函数:
                    返回 解析函数声明();
                    
                情况 词元类型.关键字_结构体:
                    返回 解析结构体声明();
                    
                情况 词元类型.关键字_枚举:
                    返回 解析枚举声明();
                    
                情况 词元类型.关键字_类:
                    返回 解析类声明();
                    
                情况 词元类型.关键字_接口:
                    返回 解析接口声明();
                    
                情况 词元类型.关键字_模块:
                    返回 解析模块声明();
                    
                情况 词元类型.关键字_导入:
                    返回 解析导入声明();
                    
                情况 词元类型.关键字_公开:
                情况 词元类型.关键字_私有:
                    // 全局变量声明
                    返回 解析变量声明();
                    
                默认:
                    // 可能是类型关键字开头的变量声明
                    如果 (是类型关键字(当前词元.类型)) {
                        返回 解析变量声明();
                    }
                    
                    报告错误("意外的词元: " + 当前词元.值);
                    前进();  // 跳过错误词元
                    返回 无;
            }
        }
        
        函数 解析函数声明() -> 函数声明* {
            变量 声明 = 创建函数声明();
            声明.位置 = 当前词元.位置;
            
            // 解析修饰符
            当 (当前词元.类型 == 词元类型.关键字_重写 ||
                  当前词元.类型 == 词元类型.关键字_虚拟 ||
                  当前词元.类型 == 词元类型.关键字_静态 ||
                  当前词元.类型 == 词元类型.关键字_抽象) {
                
                选择 当前词元.类型 {
                    情况 词元类型.关键字_重写:
                        声明.是重写 = 真;
                        中断;
                    情况 词元类型.关键字_虚拟:
                        声明.是虚拟 = 真;
                        中断;
                    情况 词元类型.关键字_静态:
                        声明.是静态 = 真;
                        中断;
                    情况 词元类型.关键字_抽象:
                        声明.是抽象 = 真;
                        中断;
                }
                前进();
            }
            
            // 期望 '函数' 关键字
            期望(词元类型.关键字_函数);
            
            // 函数名
            声明.名称 = 当前词元.值;
            期望(词元类型.标识符);
            
            // 参数列表
            期望(词元类型.左括号);
            声明.参数 = 解析参数列表();
            期望(词元类型.右括号);
            
            // 返回类型（可选）
            如果 (当前词元.类型 == 词元类型.箭头) {
                前进();  // 跳过 ->
                声明.返回类型 = 解析类型();
            }
            
            // 函数体
            如果 (当前词元.类型 == 词元类型.左大括号) {
                声明.函数体 = 解析块语句();
            } 否则 {
                // 抽象方法没有函数体
                声明.函数体 = 无;
            }
            
            返回 声明;
        }
        
        函数 解析参数列表() -> 参数列表* {
            变量 列表 = 创建参数列表();
            
            当 (当前词元.类型 != 词元类型.右括号) {
                变量 参数 = 解析参数();
                列表.添加(参数);
                
                如果 (当前词元.类型 == 词元类型.逗号) {
                    前进();
                } 否则 {
                    中断;
                }
            }
            
            返回 列表;
        }
        
        函数 解析参数() -> 参数* {
            变量 参数 = 创建参数();
            
            // 类型
            参数.类型 = 解析类型();
            
            // 参数名
            参数.名称 = 当前词元.值;
            期望(词元类型.标识符);
            
            // 数组维度
            当 (当前词元.类型 == 词元类型.左方括号) {
                参数.是数组 = 真;
                参数.数组维度 = 参数.数组维度 + 1;
                前进();  // 跳过 [
                
                如果 (当前词元.类型 != 词元类型.右方括号) {
                    // 有大小的数组参数
                    变量 大小表达式 = 解析表达式();
                    // 处理大小...
                }
                期望(词元类型.右方括号);
            }
            
            返回 参数;
        }
        
        函数 解析类型() -> 类型节点* {
            变量 类型 = 创建类型节点();
            
            // 基础类型
            选择 当前词元.类型 {
                情况 词元类型.关键字_整数:
                    类型.类型 = 节点类型.基础类型;
                    类型.名称 = "整数";
                    前进();
                    中断;
                    
                情况 词元类型.关键字_小数:
                    类型.类型 = 节点类型.基础类型;
                    类型.名称 = "小数";
                    前进();
                    中断;
                    
                情况 词元类型.关键字_字符串:
                    类型.类型 = 节点类型.基础类型;
                    类型.名称 = "字符串";
                    前进();
                    中断;
                    
                情况 词元类型.关键字_布尔:
                    类型.类型 = 节点类型.基础类型;
                    类型.名称 = "布尔";
                    前进();
                    中断;
                    
                情况 词元类型.关键字_空类型:
                    类型.类型 = 节点类型.基础类型;
                    类型.名称 = "空类型";
                    前进();
                    中断;
                    
                情况 词元类型.标识符:
                    // 自定义类型（结构体、枚举、类）
                    类型.类型 = 节点类型.结构体类型;
                    类型.名称 = 当前词元.值;
                    前进();
                    中断;
                    
                默认:
                    报告错误("期望类型");
                    返回 类型;
            }
            
            // 指针层级
            当 (当前词元.类型 == 词元类型.星号) {
                类型.指针层级 = 类型.指针层级 + 1;
                前进();
            }
            
            返回 类型;
        }
        
        函数 解析块语句() -> 块语句* {
            变量 块 = 创建块语句();
            块.位置 = 当前词元.位置;
            
            期望(词元类型.左大括号);
            
            当 (当前词元.类型 != 词元类型.右大括号 && 
                  当前词元.类型 != 词元类型.结束) {
                变量 语句 = 解析语句();
                如果 (语句 != 无) {
                    块.添加语句(语句);
                }
            }
            
            期望(词元类型.右大括号);
            
            返回 块;
        }
        
        函数 解析语句() -> 语句节点* {
            选择 当前词元.类型 {
                情况 词元类型.左大括号:
                    返回 解析块语句();
                    
                情况 词元类型.关键字_如果:
                    返回 解析如果语句();
                    
                情况 词元类型.关键字_当:
                    返回 解析当语句();
                    
                情况 词元类型.关键字_循环:
                    返回 解析循环语句();
                    
                情况 词元类型.关键字_返回:
                    返回 解析返回语句();
                    
                情况 词元类型.关键字_中断:
                    前进();
                    期望(词元类型.分号);
                    返回 创建中断语句();
                    
                情况 词元类型.关键字_继续:
                    前进();
                    期望(词元类型.分号);
                    返回 创建继续语句();
                    
                情况 词元类型.关键字_选择:
                    返回 解析选择语句();
                    
                情况 词元类型.关键字_尝试:
                    返回 解析尝试语句();
                    
                情况 词元类型.关键字_抛出:
                    返回 解析抛出语句();
                    
                情况 词元类型.关键字_变量:
                情况 词元类型.关键字_常量:
                情况 词元类型.关键字_静态:
                    返回 解析变量声明();
                    
                默认:
                    // 检查是否是类型关键字开头的变量声明
                    如果 (是类型关键字(当前词元.类型)) {
                        返回 解析变量声明();
                    }
                    
                    // 表达式语句
                    返回 解析表达式语句();
            }
        }
        
        函数 解析如果语句() -> 如果语句* {
            变量 语句 = 创建如果语句();
            语句.位置 = 当前词元.位置;
            
            期望(词元类型.关键字_如果);
            期望(词元类型.左括号);
            语句.条件 = 解析表达式();
            期望(词元类型.右括号);
            语句.真分支 = 解析块语句();
            
            // 否则分支
            如果 (当前词元.类型 == 词元类型.关键字_否则) {
                前进();  // 跳过 否则
                
                如果 (当前词元.类型 == 词元类型.关键字_如果) {
                    // 否则 如果
                    语句.假分支 = 解析如果语句();
                } 否则 {
                    // 否则
                    语句.假分支 = 解析块语句();
                }
            }
            
            返回 语句;
        }
        
        函数 解析当语句() -> 当语句* {
            变量 语句 = 创建当语句();
            语句.位置 = 当前词元.位置;
            
            期望(词元类型.关键字_当);
            期望(词元类型.左括号);
            语句.条件 = 解析表达式();
            期望(词元类型.右括号);
            语句.循环体 = 解析块语句();
            
            返回 语句;
        }
        
        函数 解析循环语句() -> 循环语句* {
            变量 语句 = 创建循环语句();
            语句.位置 = 当前词元.位置;
            
            期望(词元类型.关键字_循环);
            期望(词元类型.左括号);
            
            // 初始化
            如果 (当前词元.类型 != 词元类型.分号) {
                如果 (是类型关键字(当前词元.类型) || 
                      当前词元.类型 == 词元类型.关键字_变量) {
                    语句.初始化 = 解析变量声明();
                } 否则 {
                    语句.初始化 = 解析表达式语句();
                }
            } 否则 {
                前进();  // 跳过 ;
            }
            
            // 条件
            如果 (当前词元.类型 != 词元类型.分号) {
                语句.条件 = 解析表达式();
            }
            期望(词元类型.分号);
            
            // 更新
            如果 (当前词元.类型 != 词元类型.右括号) {
                语句.更新 = 解析表达式();
            }
            期望(词元类型.右括号);
            
            语句.循环体 = 解析块语句();
            
            返回 语句;
        }
        
        函数 解析表达式() -> 表达式节点* {
            返回 解析赋值表达式();
        }
        
        函数 解析赋值表达式() -> 表达式节点* {
            变量 表达式 = 解析三元表达式();
            
            如果 (是赋值运算符(当前词元.类型)) {
                变量 赋值 = 创建赋值表达式();
                赋值.位置 = 表达式.位置;
                赋值.目标 = 表达式;
                赋值.运算符 = 获取赋值运算符(当前词元.类型);
                前进();
                赋值.值 = 解析赋值表达式();
                返回 赋值;
            }
            
            返回 表达式;
        }
        
        函数 解析三元表达式() -> 表达式节点* {
            变量 条件 = 解析逻辑或表达式();
            
            如果 (当前词元.类型 == 词元类型.问号) {
                前进();  // 跳过 ?
                变量 三元 = 创建三元表达式();
                三元.条件 = 条件;
                三元.真值 = 解析表达式();
                期望(词元类型.冒号);
                三元.假值 = 解析三元表达式();
                返回 三元;
            }
            
            返回 条件;
        }
        
        函数 解析逻辑或表达式() -> 表达式节点* {

                前进();
                变量 右 = 解析逻辑与表达式();
                左 = 创建二元表达式(左, 运算符, 右);
            }
            
            返回 左;
        }
        
        函数 解析逻辑与表达式() -> 表达式节点* {

            当 (当前词元.类型 == 词元类型.逻辑与) {
                变量 运算符 = 当前词元;
                前进();
                变量 右 = 解析按位或表达式();
                左 = 创建二元表达式(左, 运算符, 右);

        函数 解析按位与表达式() -> 表达式节点* {
            变量 左 = 解析相等表达式();
            
            当 (当前词元.类型 == 词元类型.按位与) {
                变量 运算符 = 当前词元;
                前进();
                变量 右 = 解析相等表达式();
                左 = 创建二元表达式(左, 运算符, 右);
            }
            
            返回 左;

        函数 解析相等表达式() -> 表达式节点* {

        函数 解析乘除表达式() -> 表达式节点* {
            变量 左 = 解析一元表达式();
            
            当 (当前词元.类型 == 词元类型.星号 || 
                  当前词元.类型 == 词元类型.斜杠 ||
                  当前词元.类型 == 词元类型.百分号) {
                变量 运算符 = 当前词元;
                前进();
                变量 右 = 解析一元表达式();

        函数 解析一元表达式() -> 表达式节点* {
            如果 (当前词元.类型 == 词元类型.逻辑非 ||
                  当前词元.类型 == 词元类型.减号 ||
                  当前词元.类型 == 词元类型.按位取反 ||
                  当前词元.类型 == 词元类型.自增 ||
                  当前词元.类型 == 词元类型.自减) {
                变量 运算符 = 当前词元;
                前进();
                变量 操作数 = 解析一元表达式();
                返回 创建一元表达式(运算符, 操作数, 真);
            }
            
            返回 解析后缀表达式();
        }
        
        函数 解析后缀表达式() -> 表达式节点* {
            变量 表达式 = 解析基本表达式();
            
            当 (真) {
                选择 当前词元.类型 {
                    情况 词元类型.左括号:
                        // 函数调用
                        表达式 = 解析函数调用(表达式);
                        中断;
                        
                    情况 词元类型.点:
                        // 成员访问
                        前进();
                        变量 成员名 = 当前词元.值;
                        期望(词元类型.标识符);
                        表达式 = 创建成员访问表达式(表达式, 成员名, 假);
                        中断;
                        
                    情况 词元类型.箭头:
                        // 指针成员访问
                        前进();
                        变量 成员名 = 当前词元.值;
                        期望(词元类型.标识符);
                        表达式 = 创建成员访问表达式(表达式, 成员名, 真);
                        中断;
                        
                    情况 词元类型.左方括号:
                        // 数组访问
                        前进();
                        变量 索引 = 解析表达式();
                        期望(词元类型.右方括号);
                        表达式 = 创建数组访问表达式(表达式, 索引);
                        中断;
                        
                    情况 词元类型.自增:
                    情况 词元类型.自减:
                        // 后缀自增/自减
                        变量 运算符 = 当前词元;
                        前进();
                        表达式 = 创建一元表达式(运算符, 表达式, 假);
                        中断;
                        
                    默认:
                        返回 表达式;
                }
            }
        }
        
        函数 解析基本表达式() -> 表达式节点* {
            选择 当前词元.类型 {
                情况 词元类型.标识符:
                    变量 名称 = 当前词元.值;
                    前进();
                    返回 创建标识符表达式(名称);
                    
                情况 词元类型.整数字面量:
                    变量 值 = 当前词元.值;
                    前进();
                    返回 创建字面量表达式(字面量类型.整数, 值);
                    
                情况 词元类型.浮点字面量:
                    变量 值 = 当前词元.值;
                    前进();
                    返回 创建字面量表达式(字面量类型.小数, 值);
                    
                情况 词元类型.字符串字面量:
                    变量 值 = 当前词元.值;
                    前进();
                    返回 创建字面量表达式(字面量类型.字符串, 值);
                    
                情况 词元类型.关键字_真:
                    前进();
                    返回 创建字面量表达式(字面量类型.布尔, "真");
                    
                情况 词元类型.关键字_假:
                    前进();
                    返回 创建字面量表达式(字面量类型.布尔, "假");
                    
                情况 词元类型.关键字_无:
                    前进();
                    返回 创建字面量表达式(字面量类型.空, "无");
                    
                情况 词元类型.左括号:
                    前进();
                    变量 表达式 = 解析表达式();
                    期望(词元类型.右括号);
                    返回 表达式;
                    
                默认:
                    报告错误("期望表达式");
                    返回 创建错误表达式();
            }
        }
        
        函数 解析函数调用(表达式节点* 被调函数) -> 函数调用表达式* {
            期望(词元类型.左括号);
            
            变量 调用 = 创建函数调用表达式();
            调用.被调函数 = 被调函数;
            调用.参数 = 创建参数列表();
            
            当 (当前词元.类型 != 词元类型.右括号) {
                变量 参数 = 解析表达式();
                调用.参数.添加(参数);
                
                如果 (当前词元.类型 == 词元类型.逗号) {
                    前进();
                } 否则 {
                    中断;
                }
            }
            
            期望(词元类型.右括号);
            
            返回 调用;
        }
        
        // 辅助函数
        函数 前进() -> 空类型 {
            当前词元 = 下一个词元;
            下一个词元 = 词法.下一个词元();
        }
        
        函数 期望(词元类型 类型) -> 布尔 {
            如果 (当前词元.类型 == 类型) {
                前进();
                返回 真;
            }
            
            报告错误("期望 " + 词元类型名称(类型) + "，实际为 " + 当前词元.值);
            返回 假;
        }
        
        函数 是类型关键字(词元类型 类型) -> 布尔 {
            返回 类型 == 词元类型.关键字_整数 ||
                   类型 == 词元类型.关键字_小数 ||
                   类型 == 词元类型.关键字_字符串 ||
                   类型 == 词元类型.关键字_布尔 ||
                   类型 == 词元类型.关键字_空类型 ||
                   类型 == 词元类型.关键字_结构体 ||
                   类型 == 词元类型.关键字_枚举;
        }
        
        函数 是赋值运算符(词元类型 类型) -> 布尔 {
            返回 类型 == 词元类型.赋值;
        }
        
        函数 报告错误(字符串 消息) -> 空类型 {
            错误计数 = 错误计数 + 1;
            诊断.报告错误(当前词元.位置, 消息, 错误恢复策略.跳过词元);
        }
    }
}
```
