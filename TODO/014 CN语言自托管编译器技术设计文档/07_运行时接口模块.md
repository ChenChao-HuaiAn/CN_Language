### 3.2 CN语言接口使用示例

```cn
// src_cn/运行时/接口使用示例.cn
模块 运行时示例 {
公开:
    函数 主程序() -> 整数 {
        // 内存管理
        变量 缓冲区 = 分配内存(1024);
        
        // 字符串操作
        变量 消息 = "你好，CN语言！";
        变量 长度 = 获取字符串长度(消息);
        打印字符串("消息长度: ");
        打印行(整数转字符串(长度));
        
        // 文件操作
        变量 文件 = 打开文件("test.cn", "r");
        如果 (文件 != 无) {
            读取文件(文件, 缓冲区, 1024);
            关闭文件(文件);
        }
        
        // 动态数组
        变量 数组 = 创建数组(10);
        数组添加(数组, "元素1");
        数组添加(数组, "元素2");
        
        循环 (整数 i = 0; i < 数组长度(数组); i = i + 1) {
            打印行(数组获取(数组, i) 作为 字符串);
        }
        
        // 命令行参数
        变量 参数个数 = 获取参数个数();
        打印字符串("参数个数: ");
        打印行(整数转字符串(参数个数));
        
        // 释放资源
        销毁数组(数组);
        释放内存(缓冲区);
        
        返回 0;
    }
}
```

    // 查找字符
    函数 查找字符(字符串 s, 整数 c) -> 字符串;

    // ==================== 文件操作 ====================

    // 读取文件

    // 写入文件

    // 打印一行
    函数 打印行(字符串 str) -> 整数;
    
    // 读取一行
    函数 读取行() -> 字符串;
    
    // 读取整数
    函数 读取整数(整数* 输出值) -> 整数;
    
    // 读取小数
    函数 读取小数(小数* 输出值) -> 整数;
    
    // ==================== 动态数组 ====================
    
    // 创建数组
    函数 创建数组(整数 初始容量) -> 数组类型*;
    
    // 销毁数组
    函数 销毁数组(数组类型* 数组) -> 空类型;
    
    // 数组添加
    函数 数组添加(数组类型* 数组, 空类型* 元素) -> 整数;

    // 获取参数
    函数 获取参数(整数 索引) -> 字符串;
    
    // 获取程序名称
    函数 获取程序名称() -> 字符串;
    
    // 查找选项
    函数 查找选项(字符串 选项名) -> 字符串;
    
    // 选项存在
