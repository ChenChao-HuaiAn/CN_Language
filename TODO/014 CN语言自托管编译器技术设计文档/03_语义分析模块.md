```cn
// src_cn/语义分析器/符号表.cn
模块 语义分析器 {
公开:
    // 符号类型
    枚举 符号类型 {
        变量符号,
        函数符号,
        参数符号,
        结构体符号,
        枚举符号,
        类符号,
        接口符号,
        模块符号,
        枚举成员符号,
        类成员符号
    }
    
    // 符号
    结构体 符号 {
        符号类型 类型;
        字符串 名称;
        类型节点* 符号类型;
        源位置 定义位置;
        布尔 是常量;
        布尔 是静态;
        布尔 是公开;
        布尔 是私有;
        布尔 是保护;
        布尔 已初始化;
        
        // 函数特有
        参数列表* 参数;
        类型节点* 返回类型;
        
        // 结构体/类特有
        成员列表* 成员;
        
        // 枚举特有
        整数 枚举值;
    }
    
    // 作用域
    结构体 作用域 {
        作用域类型 类型;
        字符串 名称;
        作用域* 父作用域;
        符号表 符号;
        子作用域列表 子作用域;
    }
    
    // 作用域类型
    枚举 作用域类型 {
        全局作用域,
        函数作用域,
        块作用域,
        类作用域,
        结构体作用域,
        模块作用域
    }
    
    // 符号表管理器
    类 符号表管理器 {
    私有:
        作用域* 当前作用域;
        作用域* 全局作用域;
        诊断管理器* 诊断;
        
    公开:
        函数 创建(诊断管理器* 诊断器) -> 符号表管理器* {
            变量 管理器 = 分配内存(符号表管理器大小) 作为 符号表管理器*;
            管理器.诊断 = 诊断器;
            
            // 创建全局作用域
            管理器.全局作用域 = 创建作用域(作用域类型.全局作用域, "全局", 无);
            管理器.当前作用域 = 管理器.全局作用域;
            
            返回 管理器;
        }
        
        函数 进入作用域(作用域类型 类型, 字符串 名称) -> 空类型 {
            变量 新作用域 = 创建作用域(类型, 名称, 当前作用域);
            当前作用域.子作用域.添加(新作用域);
            当前作用域 = 新作用域;
        }
        
        函数 离开作用域() -> 空类型 {
            如果 (当前作用域.父作用域 != 无) {
                当前作用域 = 当前作用域.父作用域;
            }
        }
        
        函数 声明符号(符号* 符号) -> 布尔 {
            // 检查当前作用域是否已存在同名符号
            如果 (当前作用域.符号.包含(符号.名称)) {
                诊断.报告错误(符号.定义位置, 
                    "重复定义的标识符: " + 符号.名称, 
                    错误恢复策略.无恢复);
                返回 假;
            }
            
            当前作用域.符号.添加(符号.名称, 符号);
            返回 真;
        }
        
        函数 查找符号(字符串 名称) -> 符号* {
            变量 作用域 = 当前作用域;
            
            当 (作用域 != 无) {
                如果 (作用域.符号.包含(名称)) {
                    返回 作用域.符号.获取(名称);
                }
                作用域 = 作用域.父作用域;
            }
            
            返回 无;  // 未找到
        }
        
        函数 查找当前作用域符号(字符串 名称) -> 符号* {
            如果 (当前作用域.符号.包含(名称)) {
                返回 当前作用域.符号.获取(名称);
            }
            返回 无;
        }
    }
}
```

```cn
// src_cn/语义分析器/类型系统.cn
模块 语义分析器 {
公开:
    // 类型种类
    枚举 类型种类 {
        空类型,
        整数类型,
        小数类型,
        字符串类型,
        布尔类型,
        指针类型,
        数组类型,
        函数类型,
        结构体类型,
        枚举类型,
        类类型,
        接口类型,
        类型参数
    }
    
    // 类型结构
    结构体 类型 {
        类型种类 种类;
        字符串 名称;
        类型* 元素类型;    // 指针/数组的元素类型
        整数 指针层级;
        整数 数组维度;
        整数[] 数组大小;
        
        // 函数类型
        类型列表* 参数类型;
        类型* 返回类型;
        
        // 结构体/类/接口
        成员列表* 成员;
        类型* 基类;
        接口列表* 实现接口;
    }
    
    // 类型检查器
    类 类型检查器 {
    私有:
        诊断管理器* 诊断;
        
    公开:
        函数 创建(诊断管理器* 诊断器) -> 类型检查器* {
            变量 检查器 = 分配内存(类型检查器大小) 作为 类型检查器*;
            检查器.诊断 = 诊断器;
            返回 检查器;
        }
        
        函数 类型相等(类型* 左, 类型* 右) -> 布尔 {
            如果 (左 == 无 || 右 == 无) {
                返回 左 == 右;
            }
            
            如果 (左.种类 != 右.种类) {
                返回 假;
            }
            
            选择 左.种类 {
                情况 类型种类.空类型:
                情况 类型种类.整数类型:
                情况 类型种类.小数类型:
                情况 类型种类.字符串类型:
                情况 类型种类.布尔类型:
                    返回 真;
                    
                情况 类型种类.指针类型:
                    返回 类型相等(左.元素类型, 右.元素类型);
                    
                情况 类型种类.数组类型:
                    返回 类型相等(左.元素类型, 右.元素类型) &&
                           左.数组维度 == 右.数组维度;
                    
                情况 类型种类.函数类型:
                    返回 函数类型相等(左, 右);
                    
                情况 类型种类.结构体类型:
                情况 类型种类.类类型:
                情况 类型种类.枚举类型:
                情况 类型种类.接口类型:
                    返回 左.名称 == 右.名称;
                    
                默认:
                    返回 假;
            }
        }
        
        函数 类型兼容(类型* 目标, 类型* 源) -> 布尔 {
            // 完全相等
            如果 (类型相等(目标, 源)) {
                返回 真;
            }
            
            // 整数类型兼容
            如果 (是整数类型(目标) && 是整数类型(源)) {
                返回 真;
            }
            
            // 小数类型兼容
            如果 (是数值类型(目标) && 是数值类型(源)) {
                返回 真;
            }
            
            // 指针与空类型兼容
            如果 (目标.种类 == 类型种类.指针类型 && 源.种类 == 类型种类.空类型) {
                返回 真;
            }
            
            // 子类可以赋值给父类
            如果 (目标.种类 == 类型种类.类类型 && 源.种类 == 类型种类.类类型) {
                返回 是子类(源, 目标);
            }
            
            返回 假;
        }
        
        函数 是数值类型(类型* 类型) -> 布尔 {
            返回 类型.种类 == 类型种类.整数类型 ||
                   类型.种类 == 类型种类.小数类型;
        }
        
        函数 是整数类型(类型* 类型) -> 布尔 {
            返回 类型.种类 == 类型种类.整数类型 ||
                   类型.种类 == 类型种类.枚举类型;
        }
        
        函数 是布尔类型(类型* 类型) -> 布尔 {
            返回 类型.种类 == 类型种类.布尔类型;
        }
        
        函数 是指针类型(类型* 类型) -> 布尔 {
            返回 类型.种类 == 类型种类.指针类型;
        }
        
        函数 是函数类型(类型* 类型) -> 布尔 {
            返回 类型.种类 == 类型种类.函数类型;
        }
        
        函数 是类类型(类型* 类型) -> 布尔 {
            返回 类型.种类 == 类型种类.类类型;
        }
        
        函数 获取表达式类型(表达式节点* 表达式) -> 类型* {
            选择 表达式.类型 {
                情况 节点类型.字面量表达式:
                    变量 字面量 = 表达式 作为 字面量表达式*;
                    选择 字面量.字面量类型 {
                        情况 字面量类型.整数:
                            返回 创建基础类型(类型种类.整数类型);
                        情况 字面量类型.小数:
                            返回 创建基础类型(类型种类.小数类型);
                        情况 字面量类型.字符串:
                            返回 创建基础类型(类型种类.字符串类型);
                        情况 字面量类型.布尔:
                            返回 创建基础类型(类型种类.布尔类型);
                        情况 字面量类型.空:
                            返回 创建基础类型(类型种类.空类型);
                    }
                    
                情况 节点类型.标识符表达式:
                    变量 标识符 = 表达式 作为 标识符表达式*;
                    变量 符号 = 查找符号(标识符.名称);
                    如果 (符号 != 无) {
                        返回 符号.符号类型;
                    }
                    返回 无;
                    
                情况 节点类型.二元表达式:
                    返回 获取二元表达式类型(表达式 作为 二元表达式*);
                    
                情况 节点类型.一元表达式:
                    返回 获取一元表达式类型(表达式 作为 一元表达式*);
                    
                情况 节点类型.函数调用表达式:
                    变量 调用 = 表达式 作为 函数调用表达式*;
                    变量 函数类型 = 获取表达式类型(调用.被调函数);
                    如果 (函数类型 != 无 && 函数类型.种类 == 类型种类.函数类型) {
                        返回 函数类型.返回类型;
                    }
                    返回 无;
                    
                情况 节点类型.成员访问表达式:
                    返回 获取成员访问类型(表达式 作为 成员访问表达式*);
                    
                情况 节点类型.数组访问表达式:
                    变量 数组访问 = 表达式 作为 数组访问表达式*;
                    变量 数组类型 = 获取表达式类型(数组访问.数组);
                    如果 (数组类型 != 无 && 数组类型.种类 == 类型种类.数组类型) {
                        返回 数组类型.元素类型;
                    }
                    返回 无;
                    
                默认:
                    返回 无;
            }
        }
        
    私有:
        函数 获取二元表达式类型(二元表达式* 表达式) -> 类型* {
            变量 左类型 = 获取表达式类型(表达式.左操作数);
            变量 右类型 = 获取表达式类型(表达式.右操作数);
            
            选择 表达式.运算符 {
                情况 二元运算符.加:
                情况 二元运算符.减:
                情况 二元运算符.乘:
                情况 二元运算符.除:
                情况 二元运算符.取模:
                    // 算术运算
                    如果 (左类型.种类 == 类型种类.小数类型 || 
                          右类型.种类 == 类型种类.小数类型) {
                        返回 创建基础类型(类型种类.小数类型);
                    }
                    返回 创建基础类型(类型种类.整数类型);
                    
                情况 二元运算符.等于:
                情况 二元运算符.不等于:
                情况 二元运算符.小于:
                情况 二元运算符.小于等于:
                情况 二元运算符.大于:
                情况 二元运算符.大于等于:
                情况 二元运算符.逻辑与:
                情况 二元运算符.逻辑或:
                    // 比较和逻辑运算返回布尔
                    返回 创建基础类型(类型种类.布尔类型);
                    
                情况 二元运算符.按位与:
                情况 二元运算符.按位或:
                情况 二元运算符.按位异或:
                情况 二元运算符.左移:
                情况 二元运算符.右移:
                    // 位运算返回整数
                    返回 创建基础类型(类型种类.整数类型);
                    
                默认:
                    返回 无;
            }
        }
```

### 3.3 语义分析器类设计

```cn
// src_cn/语义分析器/语义分析器.cn
模块 语义分析器 {
公开:
    类 语义分析器 {
    私有:
        符号表管理器* 符号表;
        类型检查器* 类型检查;
        诊断管理器* 诊断;
        整数 错误计数;
        
        // 当前函数信息（用于返回值检查）
        函数声明* 当前函数;
        布尔 有返回语句;
        
    公开:
        函数 创建(诊断管理器* 诊断器) -> 语义分析器* {
            变量 分析器 = 分配内存(语义分析器大小) 作为 语义分析器*;
            分析器.诊断 = 诊断器;
            分析器.符号表 = 符号表管理器.创建(诊断器);
            分析器.类型检查 = 类型检查器.创建(诊断器);
            分析器.错误计数 = 0;
            
            返回 分析器;
        }
        
        函数 分析程序(程序节点* 程序) -> 布尔 {
            // 第一遍：收集所有顶层声明
            收集顶层声明(程序);
            
            // 第二遍：分析所有声明
            循环 (整数 i = 0; i < 程序.声明.长度; i = i + 1) {
                分析声明(程序.声明.获取(i));
            }
            
            返回 错误计数 == 0;
        }
        
    私有:
        函数 收集顶层声明(程序节点* 程序) -> 空类型 {
            循环 (整数 i = 0; i < 程序.声明.长度; i = i + 1) {
                变量 声明 = 程序.声明.获取(i);
                
                选择 声明.类型 {
                    情况 节点类型.函数声明:
                        收集函数声明(声明 作为 函数声明*);
                        中断;
                        
                    情况 节点类型.结构体声明:
                        收集结构体声明(声明 作为 结构体声明*);
                        中断;
                        
                    情况 节点类型.枚举声明:
                        收集枚举声明(声明 作为 枚举声明*);
                        中断;
                        
                    情况 节点类型.类声明:
                        收集类声明(声明 作为 类声明*);
                        中断;
                        
                    情况 节点类型.接口声明:
                        收集接口声明(声明 作为 接口声明*);
                        中断;
                        
                    情况 节点类型.变量声明:
                        收集变量声明(声明 作为 变量声明*);
                        中断;
                }
            }
        }
        
        函数 收集函数声明(函数声明* 声明) -> 空类型 {
            变量 符号 = 创建函数符号(声明);
            
            如果 (!符号表.声明符号(符号)) {
                错误计数 = 错误计数 + 1;
            }
        }
        
        函数 分析声明(声明节点* 声明) -> 空类型 {
            选择 声明.类型 {
                情况 节点类型.函数声明:
                    分析函数声明(声明 作为 函数声明*);
                    中断;
                    
                情况 节点类型.变量声明:
                    分析变量声明(声明 作为 变量声明*);
                    中断;
                    
                情况 节点类型.结构体声明:
                    分析结构体声明(声明 作为 结构体声明*);
                    中断;
                    
                情况 节点类型.类声明:
                    分析类声明(声明 作为 类声明*);
                    中断;
            }
        }
        
        函数 分析函数声明(函数声明* 声明) -> 空类型 {
            当前函数 = 声明;
            有返回语句 = 假;
            
            // 进入函数作用域
            符号表.进入作用域(作用域类型.函数作用域, 声明.名称);
            
            // 添加参数到符号表
            循环 (整数 i = 0; i < 声明.参数.长度; i = i + 1) {
                变量 参数 = 声明.参数.获取(i);
                变量 符号 = 创建参数符号(参数);
                符号表.声明符号(符号);
            }
            
            // 分析函数体
            如果 (声明.函数体 != 无) {
                分析块语句(声明.函数体);
            }
            
            // 检查返回值
            如果 (声明.返回类型 != 无 && 
                  声明.返回类型.种类 != 类型种类.空类型 &&
                  !有返回语句) {
                诊断.报告错误(声明.位置, 
                    "函数 '" + 声明.名称 + "' 缺少返回语句", 
                    错误恢复策略.无恢复);
                错误计数 = 错误计数 + 1;
            }
            
            // 离开函数作用域
            符号表.离开作用域();
            当前函数 = 无;
        }
        
        函数 分析块语句(块语句* 块) -> 空类型 {
            循环 (整数 i = 0; i < 块.语句.长度; i = i + 1) {
                分析语句(块.语句.获取(i));
            }
        }
        
        函数 分析语句(语句节点* 语句) -> 空类型 {
            选择 语句.类型 {
                情况 节点类型.变量声明:
                    分析变量声明(语句 作为 变量声明*);
                    中断;
                    
                情况 节点类型.表达式语句:
                    分析表达式语句(语句 作为 表达式语句*);
                    中断;
                    
                情况 节点类型.块语句:
                    符号表.进入作用域(作用域类型.块作用域, "");
                    分析块语句(语句 作为 块语句*);
                    符号表.离开作用域();
                    中断;
                    
                情况 节点类型.如果语句:
                    分析如果语句(语句 作为 如果语句*);
                    中断;
                    
                情况 节点类型.当语句:
                    分析当语句(语句 作为 当语句*);
                    中断;
                    
                情况 节点类型.循环语句:
                    分析循环语句(语句 作为 循环语句*);
                    中断;
                    
                情况 节点类型.返回语句:
                    分析返回语句(语句 作为 返回语句*);
                    中断;
                    
                情况 节点类型.选择语句:
                    分析选择语句(语句 作为 选择语句*);
                    中断;
            }
        }
        
        函数 分析变量声明(变量声明* 声明) -> 空类型 {
            // 检查静态变量初始化
            如果 (声明.是静态 && 声明.初始值 != 无) {
                如果 (!是编译时常量(声明.初始值)) {
                    诊断.报告错误(声明.位置, 
                        "静态变量 '" + 声明.名称 + "' 的初始化必须是编译时常量", 
                        错误恢复策略.无恢复);
                    错误计数 = 错误计数 + 1;
                }
            }
            
            // 检查常量初始化
            如果 (声明.是常量) {
                如果 (声明.初始值 == 无) {
                    诊断.报告错误(声明.位置, 
                        "常量 '" + 声明.名称 + "' 必须有初始值", 
                        错误恢复策略.无恢复);
                    错误计数 = 错误计数 + 1;
                } 否则 如果 (!是编译时常量(声明.初始值)) {
                    诊断.报告错误(声明.位置, 
                        "常量 '" + 声明.名称 + "' 的初始值必须是编译时常量", 
                        错误恢复策略.无恢复);
                    错误计数 = 错误计数 + 1;
                }
            }
            
            // 分析初始值
            如果 (声明.初始值 != 无) {
                分析表达式(声明.初始值);
                
                // 类型检查
                变量 初始值类型 = 类型检查.获取表达式类型(声明.初始值);
                如果 (声明.类型 != 无 && 初始值类型 != 无) {
                    如果 (!类型检查.类型兼容(声明.类型, 初始值类型)) {
                        诊断.报告错误(声明.位置, 
                            "类型不匹配: 期望 '" + 声明.类型.名称 + 
                            "', 实际 '" + 初始值类型.名称 + "'", 
                            错误恢复策略.无恢复);
                        错误计数 = 错误计数 + 1;
                    }
                }
            }
            
            // 添加到符号表
            变量 符号 = 创建变量符号(声明);
            如果 (!符号表.声明符号(符号)) {
                错误计数 = 错误计数 + 1;
            }
        }
        
        函数 分析表达式(表达式节点* 表达式) -> 类型* {
            选择 表达式.类型 {
                情况 节点类型.标识符表达式:
                    返回 分析标识符表达式(表达式 作为 标识符表达式*);
                    
                情况 节点类型.二元表达式:
                    返回 分析二元表达式(表达式 作为 二元表达式*);
                    
                情况 节点类型.一元表达式:
                    返回 分析一元表达式(表达式 作为 一元表达式*);
                    
                情况 节点类型.函数调用表达式:
                    返回 分析函数调用表达式(表达式 作为 函数调用表达式*);
                    
                情况 节点类型.成员访问表达式:
                    返回 分析成员访问表达式(表达式 作为 成员访问表达式*);
                    
                情况 节点类型.赋值表达式:
                    返回 分析赋值表达式(表达式 作为 赋值表达式*);
                    
                情况 节点类型.字面量表达式:
                    返回 类型检查.获取表达式类型(表达式);
                    
                默认:
                    返回 无;
            }
        }
        
        函数 分析标识符表达式(标识符表达式* 表达式) -> 类型* {
            变量 符号 = 符号表.查找符号(表达式.名称);
            
            如果 (符号 == 无) {
                诊断.报告错误(表达式.位置, 
                    "未定义的标识符: " + 表达式.名称, 
                    错误恢复策略.无恢复);
                错误计数 = 错误计数 + 1;
                返回 无;
            }
            
            返回 符号.符号类型;
        }
        
        函数 分析二元表达式(二元表达式* 表达式) -> 类型* {
            变量 左类型 = 分析表达式(表达式.左操作数);
            变量 右类型 = 分析表达式(表达式.右操作数);
            
            // 类型检查
            选择 表达式.运算符 {
                情况 二元运算符.逻辑与:
                情况 二元运算符.逻辑或:
                    // 逻辑运算符要求布尔操作数
                    如果 (!类型检查.是布尔类型(左类型) || 
                          !类型检查.是布尔类型(右类型)) {
                        诊断.报告错误(表达式.位置, 
                            "逻辑运算符需要布尔操作数", 
                            错误恢复策略.无恢复);
                        错误计数 = 错误计数 + 1;
                    }
                    中断;
                    
                情况 二元运算符.等于:
                情况 二元运算符.不等于:
                    // 相等比较需要兼容类型
                    如果 (!类型检查.类型兼容(左类型, 右类型)) {
                        诊断.报告错误(表达式.位置, 
                            "无法比较不兼容的类型", 
                            错误恢复策略.无恢复);
                        错误计数 = 错误计数 + 1;
                    }
                    中断;
                    
                默认:
                    // 算术运算符需要数值类型
                    如果 (!类型检查.是数值类型(左类型) || 
                          !类型检查.是数值类型(右类型)) {
                        诊断.报告错误(表达式.位置, 
                            "算术运算符需要数值操作数", 
                            错误恢复策略.无恢复);
                        错误计数 = 错误计数 + 1;
                    }
            }
            
            返回 类型检查.获取表达式类型(表达式);
        }
        
        函数 分析函数调用表达式(函数调用表达式* 表达式) -> 类型* {
            变量 函数类型 = 分析表达式(表达式.被调函数);
            
            如果 (函数类型 == 无) {
                返回 无;
            }
            
            如果 (!类型检查.是函数类型(函数类型)) {
                诊断.报告错误(表达式.位置, 
                    "调用的不是函数", 
                    错误恢复策略.无恢复);
                错误计数 = 错误计数 + 1;
                返回 无;
            }
            
            // 检查参数数量
            变量 期望数量 = 函数类型.参数类型.长度;
            变量 实际数量 = 表达式.参数.长度;
            
            如果 (期望数量 != 实际数量) {
                诊断.报告错误(表达式.位置, 
                    "参数数量不匹配: 期望 " + 期望数量 + 
                    ", 实际 " + 实际数量, 
                    错误恢复策略.无恢复);
                错误计数 = 错误计数 + 1;
            }
            
            // 检查参数类型
            变量 检查数量 = 最小值(期望数量, 实际数量);
            循环 (整数 i = 0; i < 检查数量; i = i + 1) {
                变量 参数类型 = 分析表达式(表达式.参数.获取(i));
                变量 期望类型 = 函数类型.参数类型.获取(i);
                
                如果 (!类型检查.类型兼容(期望类型, 参数类型)) {
                    诊断.报告错误(表达式.参数.获取(i).位置, 
                        "参数类型不匹配", 
                        错误恢复策略.无恢复);
                    错误计数 = 错误计数 + 1;
                }
            }
            
            返回 函数类型.返回类型;
        }
        
        函数 分析返回语句(返回语句* 语句) -> 空类型 {
            有返回语句 = 真;
            
            如果 (当前函数 == 无) {
                诊断.报告错误(语句.位置, 
                    "返回语句不在函数中", 
                    错误恢复策略.无恢复);
                错误计数 = 错误计数 + 1;
                返回;
            }
            
            变量 返回类型 = 无;
            
            如果 (语句.返回值 != 无) {
                返回类型 = 分析表达式(语句.返回值);
            }
            
            // 检查返回类型
            变量 期望类型 = 当前函数.返回类型;
            
            如果 (期望类型 == 无 || 期望类型.种类 == 类型种类.空类型) {
                // 函数返回空类型
                如果 (语句.返回值 != 无) {
                    诊断.报告错误(语句.位置, 
                        "空类型函数不应返回值", 
                        错误恢复策略.无恢复);
                    错误计数 = 错误计数 + 1;
                }
            } 否则 {
                // 函数有返回类型
                如果 (语句.返回值 == 无) {
                    诊断.报告错误(语句.位置, 
                        "函数应返回 " + 期望类型.名称 + " 类型的值", 
                        错误恢复策略.无恢复);
                    错误计数 = 错误计数 + 1;
                } 否则 如果 (!类型检查.类型兼容(期望类型, 返回类型)) {
                    诊断.报告错误(语句.位置, 
                        "返回类型不匹配: 期望 " + 期望类型.名称 + 
                        ", 实际 " + 返回类型.名称, 
                        错误恢复策略.无恢复);
                    错误计数 = 错误计数 + 1;
                }
            }
        }
        
        函数 是编译时常量(表达式节点* 表达式) -> 布尔 {
            选择 表达式.类型 {
                情况 节点类型.字面量表达式:
                    返回 真;
                    
                情况 节点类型.标识符表达式:
                    变量 标识符 = 表达式 作为 标识符表达式*;
                    变量 符号 = 符号表.查找符号(标识符.名称);
                    返回 符号 != 无 && 符号.是常量;
                    
                情况 节点类型.二元表达式:
                    变量 二元 = 表达式 作为 二元表达式*;
                    返回 是编译时常量(二元.左操作数) && 
                           是编译时常量(二元.右操作数);
                    
                情况 节点类型.一元表达式:
                    变量 一元 = 表达式 作为 一元表达式*;
                    返回 是编译时常量(一元.操作数);
                    
                默认:
                    返回 假;
            }
        }
    }
}
```
