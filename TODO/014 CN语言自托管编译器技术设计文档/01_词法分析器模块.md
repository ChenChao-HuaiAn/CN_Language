## 1. 任务概述

### 1.1 任务目标

将现有C语言实现的词法分析器（`src/frontend/lexer/`）使用CN语言重写，实现自托管编译器的词法分析功能。

### 1.2 功能要求

| 功能项 | 说明 | 优先级 |
|--------|------|--------|
| 词元扫描 | 识别标识符、关键字、数字、字符串、运算符 | ⭐⭐⭐ 高 |
| 关键字识别 | 识别CN语言所有关键字（函数、变量、如果等） | ⭐⭐⭐ 高 |
| 错误恢复 | 遇到非法字符时的错误恢复机制 | ⭐⭐ 中 |
| 源位置追踪 | 记录每个词元的行号、列号 | ⭐⭐⭐ 高 |
| 注释处理 | 支持单行注释和多行注释 | ⭐⭐ 中 |

### 1.3 验收标准

- [ ] 词法分析器能正确扫描所有CN语言词元
- [ ] 关键字表完整覆盖所有CN语言关键字
- [ ] 错误恢复机制正常工作
- [ ] 词法分析测试用例100%通过

### 3.2 扫描器类设计

```cn
// src_cn/词法分析器/扫描器.cn
模块 词法分析器 {
公开:
    类 扫描器 {
    私有:
        字符串 源码;
        整数 位置;
        整数 行号;
        整数 列号;
        整数 源码长度;
        
    公开:
        函数 创建(字符串 源码内容, 整数 长度) -> 扫描器* {
            变量 实例 = 分配内存(扫描器大小) 作为 扫描器*;
            实例.源码 = 源码内容;
            实例.位置 = 0;
            实例.行号 = 1;
            实例.列号 = 1;
            实例.源码长度 = 长度;
            返回 实例;
        }
        
        函数 销毁(扫描描器* 实例) -> 空类型 {
            释放内存(实例);
        }
        
        函数 下一个词元() -> 词元 {
            跳过空白字符();
            
            如果 (位置 >= 源码长度) {
                返回 创建词元(词元类型.结束, "", 行号, 列号);
            }
            
            变量 当前字符 = 源码[位置];
            
            // 标识符或关键字
            如果 (是标识符开头(当前字符)) {
                返回 扫描标识符();
            }
            
            // 数字
            如果 (是数字(当前字符)) {
                返回 扫描数字();
            }
            
            // 字符串
            如果 (当前字符 == '"') {
                返回 扫描字符串();
            }
            
            // 字符
            如果 (当前字符 == '\'') {
                返回 扫描字符();
            }
            
            // 运算符和分隔符
            返回 扫描运算符();
        }
        
        函数 预览词元() -> 词元 {
            变量 保存位置 = 位置;
            变量 保存行号 = 行号;
            变量 保存列号 = 列号;
            
            变量 词元 = 下一个词元();
            
            位置 = 保存位置;
            行号 = 保存行号;
            列号 = 保存列号;
            
            返回 词元;
        }
        
    私有:
        函数 跳过空白字符() -> 空类型 {
            当 (位置 < 源码长度) {
                变量 字符 = 源码[位置];
                
                // 空白字符
                如果 (字符 == ' ' || 字符 == '\t' || 字符 == '\r') {
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                }
                // 换行
                否则 如果 (字符 == '\n') {
                    位置 = 位置 + 1;
                    行号 = 行号 + 1;
                    列号 = 1;
                }
                // 单行注释
                否则 如果 (字符 == '/' && 位置 + 1 < 源码长度 && 源码[位置 + 1] == '/') {
                    跳过单行注释();
                }
                // 块注释
                否则 如果 (字符 == '/' && 位置 + 1 < 源码长度 && 源码[位置 + 1] == '*') {
                    跳过块注释();
                }
                否则 {
                    中断;
                }
            }
        }
        
        函数 跳过单行注释() -> 空类型 {
            位置 = 位置 + 2;  // 跳过 //
            当 (位置 < 源码长度 && 源码[位置] != '\n') {
                位置 = 位置 + 1;
            }
        }
        
        函数 跳过块注释() -> 布尔 {
            位置 = 位置 + 2;  // 跳过 /*
            变量 起始行号 = 行号;
            
            当 (位置 + 1 < 源码长度) {
                如果 (源码[位置] == '\n') {
                    行号 = 行号 + 1;
                    列号 = 1;
                }
                
                如果 (源码[位置] == '*' && 源码[位置 + 1] == '/') {
                    位置 = 位置 + 2;  // 跳过 */
                    返回 真;
                }
                
                位置 = 位置 + 1;
            }
            
            // 未闭合的块注释
            返回 假;
        }
        
        函数 扫描标识符() -> 词元 {
            变量 起始位置 = 位置;
            变量 起始列号 = 列号;
            
            当 (位置 < 源码长度 && 是标识符字符(源码[位置])) {
                位置 = 位置 + 1;
                列号 = 列号 + 1;
            }
            
            变量 值 = 源码.子串(起始位置, 位置 - 起始位置);
            变量 类型 = 查找关键字(值);
            
            返回 创建词元(类型, 值, 行号, 起始列号);
        }
        
        函数 扫描数字() -> 词元 {
            变量 起始位置 = 位置;
            变量 起始列号 = 列号;
            变量 是浮点数 = 假;
            
            // 处理十六进制、二进制、八进制前缀
            如果 (源码[位置] == '0' && 位置 + 1 < 源码长度) {
                变量 下一个 = 源码[位置 + 1];
                如果 (下一个 == 'x' || 下一个 == 'X') {
                    返回 扫描十六进制(起始列号);
                }
                如果 (下一个 == 'b' || 下一个 == 'B') {
                    返回 扫描二进制(起始列号);
                }
                如果 (下一个 == 'o' || 下一个 == 'O') {
                    返回 扫描八进制(起始列号);
                }
            }
            
            // 十进制数字
            当 (位置 < 源码长度 && 是数字(源码[位置])) {
                位置 = 位置 + 1;
                列号 = 列号 + 1;
            }
            
            // 浮点数小数部分
            如果 (位置 < 源码长度 && 源码[位置] == '.') {
                是浮点数 = 真;
                位置 = 位置 + 1;
                列号 = 列号 + 1;
                
                当 (位置 < 源码长度 && 是数字(源码[位置])) {
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                }
            }
            
            // 科学计数法
            如果 (位置 < 源码长度 && (源码[位置] == 'e' || 源码[位置] == 'E')) {
                是浮点数 = 真;
                位置 = 位置 + 1;
                列号 = 列号 + 1;
                
                如果 (位置 < 源码长度 && (源码[位置] == '+' || 源码[位置] == '-')) {
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                }
                
                当 (位置 < 源码长度 && 是数字(源码[位置])) {
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                }
            }
            
            // 后缀
            如果 (位置 < 源码长度) {
                变量 字符 = 源码[位置];
                如果 (字符 == 'f' || 字符 == 'F') {
                    是浮点数 = 真;
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                }
            }
            
            变量 值 = 源码.子串(起始位置, 位置 - 起始位置);
            变量 类型 = 是浮点数 ? 词元类型.浮点字面量 : 词元类型.整数字面量;
            
            返回 创建词元(类型, 值, 行号, 起始列号);
        }
        
        函数 扫描字符串() -> 词元 {
            变量 起始列号 = 列号;
            位置 = 位置 + 1;  // 跳过开始引号
            列号 = 列号 + 1;
            
            变量 起始位置 = 位置;
            
            当 (位置 < 源码长度 && 源码[位置] != '"') {
                如果 (源码[位置] == '\\' && 位置 + 1 < 源码长度) {
                    位置 = 位置 + 2;  // 跳过转义字符
                    列号 = 列号 + 2;
                } 否则 {
                    如果 (源码[位置] == '\n') {
                        行号 = 行号 + 1;
                        列号 = 1;
                    } 否则 {
                        列号 = 列号 + 1;
                    }
                    位置 = 位置 + 1;
                }
            }
            
            变量 值 = 源码.子串(起始位置, 位置 - 起始位置);
            
            如果 (位置 < 源码长度) {
                位置 = 位置 + 1;  // 跳过结束引号
                列号 = 列号 + 1;
            }
            
            返回 创建词元(词元类型.字符串字面量, 值, 行号, 起始列号);
        }
        
        函数 扫描运算符() -> 词元 {
            变量 起始列号 = 列号;
            变量 字符 = 源码[位置];
            
            选择 字符 {
                情况 '+':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    如果 (位置 < 源码长度 && 源码[位置] == '+') {
                        位置 = 位置 + 1;
                        列号 = 列号 + 1;
                        返回 创建词元(词元类型.自增, "++", 行号, 起始列号);
                    }
                    返回 创建词元(词元类型.加号, "+", 行号, 起始列号);
                    
                情况 '-':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    如果 (位置 < 源码长度 && 源码[位置] == '-') {
                        位置 = 位置 + 1;
                        列号 = 列号 + 1;
                        返回 创建词元(词元类型.自减, "--", 行号, 起始列号);
                    }
                    如果 (位置 < 源码长度 && 源码[位置] == '>') {
                        位置 = 位置 + 1;
                        列号 = 列号 + 1;
                        返回 创建词元(词元类型.箭头, "->", 行号, 起始列号);
                    }
                    返回 创建词元(词元类型.减号, "-", 行号, 起始列号);
                    
                情况 '*':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    返回 创建词元(词元类型.星号, "*", 行号, 起始列号);
                    
                情况 '/':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    返回 创建词元(词元类型.斜杠, "/", 行号, 起始列号);
                    
                情况 '%':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    返回 创建词元(词元类型.百分号, "%", 行号, 起始列号);
                    
                情况 '=':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    如果 (位置 < 源码长度 && 源码[位置] == '=') {
                        位置 = 位置 + 1;

                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    返回 创建词元(词元类型.点, ".", 行号, 起始列号);
                    
                情况 ':':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    返回 创建词元(词元类型.冒号, ":", 行号, 起始列号);
                    
                情况 '?':
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    返回 创建词元(词元类型.问号, "?", 行号, 起始列号);
                    
                默认:
                    位置 = 位置 + 1;
                    列号 = 列号 + 1;
                    返回 创建词元(词元类型.错误, 字符串.从字符(字符), 行号, 起始列号);
            }
        }
    }
}
```

// src_cn/词法分析器/字符工具.cn
模块 词法分析器 {
公开:
    函数 是空白字符(字符 c) -> 布尔 {
        返回 c == ' ' || c == '\t' || c == '\n' || c == '\r';
    }
    
    函数 是数字(字符 c) -> 布尔 {
        返回 c >= '0' && c <= '9';
    }
    
    函数 是十六进制数字(字符 c) -> 布尔 {
        返回 (c >= '0' && c <= '9') || 
               (c >= 'a' && c <= 'f') || 
               (c >= 'A' && c <= 'F');
    }
    
    函数 是二进制数字(字符 c) -> 布尔 {
        返回 c == '0' || c == '1';
    }
    
    函数 是八进制数字(字符 c) -> 布尔 {
        返回 c >= '0' && c <= '7';
    }
    
    函数 是字母(字符 c) -> 布尔 {
        返回 (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z');
    }
    
    函数 是标识符开头(字符 c) -> 布尔 {
        // 下划线、字母、或UTF-8多字节字符（中文等）
        返回 c == '_' || 是字母(c) || (c & 0x80) != 0;
    }
    
    函数 是标识符字符(字符 c) -> 布尔 {
        // 下划线、字母、数字、或UTF-8多字节字符
        返回 c == '_' || 是字母(c) || 是数字(c) || (c & 0x80) != 0;
    }
    
    函数 是运算符字符(字符 c) -> 布尔 {
        返回 c == '+' || c == '-' || c == '*' || c == '/' ||
               c == '%' || c == '=' || c == '!' || c == '<' ||
               c == '>' || c == '&' || c == '|' || c == '^' ||
               c == '~';
    }
    
    函数 是分隔符字符(字符 c) -> 布尔 {
        返回 c == '(' || c == ')' || c == '{' || c == '}' ||
               c == '[' || c == ']' || c == ';' || c == ',' ||
