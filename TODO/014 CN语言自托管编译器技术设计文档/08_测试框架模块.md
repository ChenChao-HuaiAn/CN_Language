// src_cn/测试框架/测试框架.cn
模块 测试框架 {
公开:

        通过,
        失败,
        跳过,

    }
    
    // 测试套件
    类 测试套件 {
    私有:

        整数 通过数;
        整数 失败数;
        整数 跳过数;
        
    公开:
        函数 创建(字符串 名称) -> 测试套件*;
        函数 添加用例(测试用例* 用例) -> 空类型;
        函数 执行所有() -> 空类型;
        函数 打印报告() -> 空类型;
        函数 获取通过数() -> 整数 { 返回 通过数; }

    // 断言函数

### 3.2 自举测试设计

```cn
// tests/自举测试/自举测试.cn
模块 自举测试 {
公开:
    类 自举测试器 {
    私有:
        字符串 源码目录;
        字符串 构建目录;
        字符串 引导编译器;
        
    公开:
        函数 创建(字符串 源码目录, 字符串 构建目录) -> 自举测试器*;
        函数 执行测试() -> 布尔 {
            打印("=== CN语言自举测试 ===")
            
            // 阶段1: 使用C编译器编译CN源码
            打印("[阶段1] 使用C编译器编译CN源码...")
            变量 v1 = 编译("cnc.exe", 源码目录, "cn_v1.exe")
            如果 (非 v1.成功) {
                打印("❌ 阶段1编译失败")
                返回 假
            }
            
            // 阶段2: 使用v1编译器编译CN源码
            打印("[阶段2] 使用v1编译器编译CN源码...")
            变量 v2 = 编译("cn_v1.exe", 源码目录, "cn_v2.exe")
            如果 (非 v2.成功) {
                打印("❌ 阶段2编译失败")
                返回 假
            }
            
            // 阶段3: 使用v2编译器编译CN源码
            打印("[阶段3] 使用v2编译器编译CN源码...")
            变量 v3 = 编译("cn_v2.exe", 源码目录, "cn_v3.exe")
            如果 (非 v3.成功) {
                打印("❌ 阶段3编译失败")
                返回 假
            }
            
            // 验证: 比较v2和v3
            打印("[验证] 比较编译结果...")
            变量 哈希v2 = 计算文件哈希("cn_v2.exe")
            变量 哈希v3 = 计算文件哈希("cn_v3.exe")
            
            如果 (哈希v2 == 哈希v3) {
                打印("✅ 自举验证通过！")
                返回 真
            } 否则 {
                打印("❌ 自举验证失败！v2和v3不一致")
                返回 假
            }
        }
        
    私有:
        函数 编译(字符串 编译器, 字符串 源码, 字符串 输出) -> 编译结果;
        函数 计算文件哈希(字符串 文件路径) -> 字符串;
    }
}
```

### 3.3 对比测试设计

```cn
// tests/对比测试/对比测试.cn
模块 对比测试 {
公开:
    类 对比测试器 {
    私有:
        字符串 C编译器路径;
        字符串 CN编译器路径;
        测试文件列表 测试文件;
        
    公开:
        函数 创建(字符串 C编译器, 字符串 CN编译器) -> 对比测试器*;
        函数 执行测试() -> 测试结果 {
            变量 结果 = 创建测试结果()
            
            循环 变量 文件 在 测试文件 {
                变量 对比结果 = 对比单个文件(文件)
                结果.添加(对比结果)
            }
            
            返回 结果
        }
        
    私有:
        函数 对比单个文件(字符串 文件路径) -> 对比结果 {
            // 使用C编译器编译
            变量 C结果 = 使用C编译器编译(文件路径)
            
            // 使用CN编译器编译
            变量 CN结果 = 使用CN编译器编译(文件路径)
            
            // 比较结果
            变量 相等 = 比较编译结果(C结果, CN结果)
            
            返回 创建对比结果(文件路径, 相等, C结果, CN结果)
        }
        
        函数 使用C编译器编译(字符串 文件路径) -> 编译结果;
        函数 使用CN编译器编译(字符串 文件路径) -> 编译结果;
        函数 比较编译结果(编译结果 左, 编译结果 右) -> 布尔;
    }
    
    结构体 对比结果 {
        字符串 文件名;
        布尔 相等;
        编译结果 C结果;
        编译结果 CN结果;
        字符串 差异描述;
    }
}
```

### 3.4 回归测试设计

```cn
// tests/回归测试/回归测试.cn
模块 回归测试 {
私有:
    常量 字符串[] 测试文件列表 = [
        "基础语法/变量声明.cn",
        "基础语法/函数定义.cn",
        "基础语法/控制流.cn",
        "类型系统/基础类型.cn",
        "类型系统/数组类型.cn",
        "类型系统/结构体.cn",
        "模块系统/导入导出.cn",
        "模块系统/可见性.cn",
        "面向对象/类定义.cn",
        "面向对象/继承.cn",
        "异常处理/try_catch.cn"
    ]
    
公开:
    类 回归测试器 {
    私有:
        字符串 测试目录;
        字符串 预期目录;
        整数 通过数;
        整数 失败数;
        
    公开:
        函数 创建(字符串 测试目录, 字符串 预期目录) -> 回归测试器*;
        
        函数 执行测试() -> 整数 {
            打印("=== CN语言回归测试 ===")
            
            通过数 = 0
            失败数 = 0
            
            循环 变量 文件名 在 测试文件列表 {
                变量 结果 = 执行单个测试(文件名)
                
                如果 (结果.通过) {
                    通过数 = 通过数 + 1
                    打印("✅ " + 文件名)
                } 否则 {
                    失败数 = 失败数 + 1
                    打印("❌ " + 文件名 + ": " + 结果.错误信息)
                }
            }
            
            打印("")
            打印("回归测试结果: " + 通过数 + "/" + (通过数 + 失败数) + " 通过")
            
            返回 失败数
        }
        
    私有:
        函数 执行单个测试(字符串 文件名) -> 测试结果 {
            变量 源码 = 读取文件(测试目录 + "/" + 文件名)
            
            // 编译
            变量 编译结果 = 编译(源码)
            如果 (非 编译结果.成功) {
                返回 创建失败结果("编译失败: " + 编译结果.错误)
            }
            
            // 执行
            变量 执行结果 = 执行(编译结果.输出文件)
            如果 (非 执行结果.成功) {
                返回 创建失败结果("执行失败: " + 执行结果.错误)
            }
            
            // 验证输出
            变量 预期输出 = 读取文件(预期目录 + "/" + 文件名 + ".expected")
            如果 (执行结果.输出 != 预期输出) {
                返回 创建失败结果("输出不匹配")
            }
            
            返回 创建成功结果()
        }
    }
}
```

### 3.5 测试报告生成

```cn
// src_cn/测试框架/测试报告.cn
模块 测试框架 {
公开:
    类 测试报告生成器 {
    公开:
        函数 生成报告(测试套件 套件, 字符串 输出路径) -> 空类型 {
            变量 报告 = 创建报告内容(套件);
            写入文件(输出路径, 报告);
        }
        
    私有:
        函数 创建报告内容(测试套件 套件) -> 字符串 {
            变量 内容 = "# 测试报告\n\n"
            内容 = 内容 + "## 概要\n\n"
            内容 = 内容 + "- 通过: " + 套件.获取通过数() + "\n"
            内容 = 内容 + "- 失败: " + 套件.获取失败数() + "\n"
            内容 = 内容 + "- 跳过: " + 套件.获取跳过数() + "\n"
            内容 = 内容 + "\n## 详细结果\n\n"
            
            循环 变量 用例 在 套件.获取用例列表() {
                变量 状态 = 用例.获取结果() == 测试结果.通过 ? "✅" : "❌"
                内容 = 内容 + "- " + 状态 + " " + 用例.名称 + "\n"
                
                如果 (用例.获取结果() == 测试结果.失败) {
                    内容 = 内容 + "  - 错误: " + 用例.获取错误信息() + "\n"
                }
            }
            
            返回 内容
        }
    }
}
```
