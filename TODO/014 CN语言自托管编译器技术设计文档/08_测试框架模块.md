```cn
// src_cn/测试框架/测试框架.cn
模块 测试框架 {
public:
    // 测试结果
    枚举 测试结果 {
        通过,
        失败,
        跳过
    }
    
    // 测试用例
    结构体 测试用例 {
        字符串 名称;
        字符串 文件名;
        整数 行号;
        函数指针 执行函数;  // () -> 空类型
        测试结果 结果;
        字符串 错误信息;
    }
    
    // 测试套件
    结构体 测试套件 {
        字符串 名称;
        测试用例列表 用例;
        整数 通过数;
        整数 失败数;
        整数 跳过数;
    }
    
    // 测试运行器
    类 测试运行器 {
    私有:
        测试套件列表 套件列表;
        整数 总通过数;
        整数 总失败数;
        整数 总跳过数;
        
    公开:
        函数 创建() -> 测试运行器* {
            变量 运行器 = 分配内存(测试运行器大小) 作为 测试运行器*;
            运行器.总通过数 = 0;
            运行器.总失败数 = 0;
            运行器.总跳过数 = 0;
            返回 运行器;
        }
        
        函数 添加套件(测试套件* 套件) -> 空类型 {
            套件列表.添加(套件);
        }
        
        函数 运行所有测试() -> 整数 {
            打印行("========================================");
            打印行("CN语言自托管编译器测试");
            打印行("========================================");
            打印行("");
            
            循环 (整数 i = 0; i < 套件列表.长度; i = i + 1) {
                运行套件(套件列表.获取(i));
            }
            
            打印行("");
            打印行("========================================");
            打印行("测试结果汇总");
            打印行("========================================");
            打印行("通过: " + 整数转字符串(总通过数));
            打印行("失败: " + 整数转字符串(总失败数));
            打印行("跳过: " + 整数转字符串(总跳过数));
            打印行("========================================");
            
            返回 总失败数;
        }
        
    私有:
        函数 运行套件(测试套件* 套件) -> 空类型 {
            打印行("");
            打印行("[" + 套件.名称 + "]");
            
            循环 (整数 i = 0; i < 套件.用例.长度; i = i + 1) {
                运行用例(套件, 套件.用例.获取(i));
            }
            
            打印行("  结果: " + 整数转字符串(套件.通过数) + " 通过, " +
                      整数转字符串(套件.失败数) + " 失败");
        }
        
        函数 运行用例(测试套件* 套件, 测试用例* 用例) -> 空类型 {
            // 执行测试
            变量 有异常 = 假;
            
            尝试 {
                用例.执行函数();
                用例.结果 = 测试结果.通过;
            } 捕获 (异常 e) {
                用例.结果 = 测试结果.失败;
                用例.错误信息 = e.消息;
                有异常 = 真;
            }
            
            // 更新计数
            选择 用例.结果 {
                情况 测试结果.通过:
                    套件.通过数 = 套件.通过数 + 1;
                    总通过数 = 总通过数 + 1;
                    打印行("  ✅ " + 用例.名称);
                    中断;
                    
                情况 测试结果.失败:
                    套件.失败数 = 套件.失败数 + 1;
                    总失败数 = 总失败数 + 1;
                    打印行("  ❌ " + 用例.名称);
                    如果 (用例.错误信息 != "") {
                        打印行("     错误: " + 用例.错误信息);
                    }
                    中断;
                    
                情况 测试结果.跳过:
                    套件.跳过数 = 套件.跳过数 + 1;
                    总跳过数 = 总跳过数 + 1;
                    打印行("  ⏭️ " + 用例.名称 + " (跳过)");
                    中断;
            }
        }
    }
}
```

### 3.2 断言功能设计

```cn
// src_cn/测试框架/断言.cn
模块 测试框架 {
public:
    // 断言管理器
    类 断言 {
    公开:
        // 相等断言
        函数 相等(字符串 消息, 整数 期望值, 整数 实际值) -> 空类型 {
            如果 (期望值 != 实际值) {
                抛出 "断言失败: " + 消息 + 
                      " - 期望 " + 整数转字符串(期望值) + 
                      ", 实际 " + 整数转字符串(实际值);
            }
        }
        
        函数 相等字符串(字符串 消息, 字符串 期望值, 字符串 实际值) -> 空类型 {
            如果 (比较字符串(期望值, 实际值) != 0) {
                抛出 "断言失败: " + 消息 + 
                      " - 期望 \"" + 期望值 + 
                      "\", 实际 \"" + 实际值 + "\"";
            }
        }
        
        // 布尔断言
        函数 为真(字符串 消息, 布尔 条件) -> 空类型 {
            如果 (!条件) {
                抛出 "断言失败: " + 消息 + " - 期望为真";
            }
        }
        
        函数 为假(字符串 消息, 布尔 条件) -> 空类型 {
            如果 (条件) {
                抛出 "断言失败: " + 消息 + " - 期望为假";
            }
        }
        
        // 空值断言
        函数 为空(字符串 消息, 空类型* 指针) -> 空类型 {
            如果 (指针 != 无) {
                抛出 "断言失败: " + 消息 + " - 期望为空";
            }
        }
        
        函数 不为空(字符串 消息, 空类型* 指针) -> 空类型 {
            如果 (指针 == 无) {
                抛出 "断言失败: " + 消息 + " - 期望不为空";
            }
        }
        
        // 范围断言
        函数 在范围内(字符串 消息, 整数 值, 整数 最小值, 整数 最大值) -> 空类型 {
            如果 (值 < 最小值 || 值 > 最大值) {
                抛出 "断言失败: " + 消息 + 
                      " - 值 " + 整数转字符串(值) + 
                      " 不在范围 [" + 整数转字符串(最小值) + 
                      ", " + 整数转字符串(最大值) + "] 内";
            }
        }
        
        // 类型断言
        函数 类型相等(字符串 消息, 类型节点* 期望类型, 类型节点* 实际类型) -> 空类型 {
            如果 (!类型相等(期望类型, 实际类型)) {
                抛出 "断言失败: " + 消息 + 
                      " - 类型不匹配: 期望 " + 期望类型.名称 + 
                      ", 实际 " + 实际类型.名称;
            }
        }
    }
}
```

```cn
// tests/对比测试.cn
模块 对比测试 {
public:
    函数 运行对比测试() -> 整数 {
        打印行("========================================");
        打印行("CN语言编译器对比测试");
        打印行("========================================");
        
        变量 测试文件列表 = 获取测试文件列表();
        变量 通过数 = 0;
        变量 失败数 = 0;
        
        循环 (整数 i = 0; i < 测试文件列表.长度; i = i + 1) {
            变量 文件名 = 测试文件列表.获取(i);
            变量 结果 = 对比单个文件(文件名);
            
            如果 (结果) {
                通过数 = 通过数 + 1;
                打印行("✅ " + 文件名);
            } 否则 {
                失败数 = 失败数 + 1;
                打印行("❌ " + 文件名);
            }
        }
        
        打印行("");
        打印行("对比测试结果: " + 整数转字符串(通过数) + " 通过, " +
                  整数转字符串(失败数) + " 失败");
        
        返回 失败数;
    }
    
私有:
    函数 对比单个文件(字符串 文件名) -> 布尔 {
        // 使用C版本编译
        变量 C输出 = 使用C编译器编译(文件名);
        
        // 使用CN版本编译
        变量 CN输出 = 使用CN编译器编译(文件名);
        
        // 比较输出
        返回 比较输出(C输出, CN输出);
    }
    
    函数 使用C编译器编译(字符串 文件名) -> 字符串 {
        // 调用 cnc.exe 编译
        变量 命令 = "./build/cnc.exe " + 文件名 + " --emit-c";
        返回 执行命令并获取输出(命令);
    }
    
    函数 使用CN编译器编译(字符串 文件名) -> 字符串 {
        // 调用 cn.exe 编译
        变量 命令 = "./build/cn.exe " + 文件名 + " --emit-c";
        返回 执行命令并获取输出(命令);
    }
    
    函数 比较输出(字符串 输出1, 字符串 输出2) -> 布尔 {
        // 忽略空白差异进行比较
        返回 规范化比较(输出1, 输出2);
    }
}
```

```cn
// tests/自举测试.cn
模块 自举测试 {
public:
    函数 运行自举测试() -> 整数 {
        打印行("========================================");
        打印行("CN语言编译器自举测试");
        打印行("========================================");
        
        // 阶段1: 使用C编译器编译CN源码
        打印行("[阶段1] 使用 cnc.exe 编译 CN 源码...");
        变量 v1结果 = 编译CN源码("./build/cnc.exe", "./build/cn_v1.exe");
        如果 (!v1结果.成功) {
            打印行("❌ 阶段1失败: " + v1结果.错误信息);
            返回 1;
        }
        打印行("✅ 生成 cn_v1.exe");
        
        // 阶段2: 使用v1编译器编译CN源码
        打印行("[阶段2] 使用 cn_v1.exe 编译 CN 源码...");
        变量 v2结果 = 编译CN源码("./build/cn_v1.exe", "./build/cn_v2.exe");
        如果 (!v2结果.成功) {
            打印行("❌ 阶段2失败: " + v2结果.错误信息);
            返回 1;
        }
        打印行("✅ 生成 cn_v2.exe");
        
        // 阶段3: 使用v2编译器编译CN源码
        打印行("[阶段3] 使用 cn_v2.exe 编译 CN 源码...");
        变量 v3结果 = 编译CN源码("./build/cn_v2.exe", "./build/cn_v3.exe");
        如果 (!v3结果.成功) {
            打印行("❌ 阶段3失败: " + v3结果.错误信息);
            返回 1;
        }
        打印行("✅ 生成 cn_v3.exe");
        
        // 验证: v2 和 v3 应该相同
        打印行("[验证] 比较编译结果...");
        变量 哈希v2 = 计算文件哈希("./build/cn_v2.exe");
        变量 哈希v3 = 计算文件哈希("./build/cn_v3.exe");
        
        如果 (哈希v2 == 哈希v3) {
            打印行("✅ 自举验证通过！");
            打印行("v2 和 v3 哈希一致: " + 哈希v2);
            返回 0;
        } 否则 {
            打印行("❌ 自举验证失败！");
            打印行("v2 哈希: " + 哈希v2);
```
