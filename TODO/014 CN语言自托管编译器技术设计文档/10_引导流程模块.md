### 3.1 引导脚本设计

```bash
#!/bin/bash
# bootstrap.sh - CN语言自托管引导脚本
# 用法: ./bootstrap.sh [--verify] [--clean]

set -e

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# 配置
BUILD_DIR="./build"
SRC_CN="./src_cn"
LOG_DIR="./logs"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
LOG_FILE="${LOG_DIR}/bootstrap_${TIMESTAMP}.log"

# 创建日志目录
mkdir -p ${LOG_DIR}

# 日志函数
log() {
    echo -e "$1" | tee -a ${LOG_FILE}
}

log_info() {
    log "${BLUE}[INFO]${NC} $1"
}

log_success() {
    log "${GREEN}[SUCCESS]${NC} $1"
}

log_warning() {
    log "${YELLOW}[WARNING]${NC} $1"
}

log_error() {
    log "${RED}[ERROR]${NC} $1"
}

# ========================================
# 阶段0: 准备阶段
# ========================================
stage0_prepare() {
    log ""
    log "========================================"
    log "阶段0: 准备阶段"
    log "========================================"
    
    # 检查引导编译器
    log_info "检查引导编译器..."
    if [ ! -f "${BUILD_DIR}/bin/cnc.exe" ] && [ ! -f "${BUILD_DIR}/bin/Release/cnc.exe" ]; then
        log_error "未找到引导编译器 cnc.exe"
        log_info "正在构建引导编译器..."
        
        mkdir -p ${BUILD_DIR}
        cd ${BUILD_DIR}
        cmake .. -DBUILD_BOOTSTRAP=ON -DBUILD_SELF_HOSTED=OFF
        cmake --build . --config Release
        cd ..
    fi
    
    # 确定编译器路径
    if [ -f "${BUILD_DIR}/bin/cnc.exe" ]; then
        CNC_COMPILER="${BUILD_DIR}/bin/cnc.exe"
    else
        CNC_COMPILER="${BUILD_DIR}/bin/Release/cnc.exe"
    fi
    
    log_success "引导编译器: ${CNC_COMPILER}"
    
    # 检查CN源码目录
    log_info "检查CN源码目录..."
    if [ ! -d "${SRC_CN}" ]; then
        log_error "未找到CN源码目录: ${SRC_CN}"
        exit 1
    fi
    
    # 统计源文件
    CN_FILE_COUNT=$(find ${SRC_CN} -name "*.cn" | wc -l)
    log_success "找到 ${CN_FILE_COUNT} 个CN源文件"
}

# ========================================
# 阶段1: 首次编译
# ========================================
stage1_compile() {
    log ""
    log "========================================"
    log "阶段1: 使用C编译器编译CN源码"
    log "========================================"
    
    log_info "编译中..."
    
    ${CNC_COMPILER} ${SRC_CN} -o ${BUILD_DIR}/cn_v1.exe 2>&1 | tee -a ${LOG_FILE}
    
    if [ -f "${BUILD_DIR}/cn_v1.exe" ]; then
        log_success "生成 cn_v1.exe"
        
        # 记录文件信息
        FILE_SIZE=$(stat -c%s "${BUILD_DIR}/cn_v1.exe" 2>/dev/null || stat -f%z "${BUILD_DIR}/cn_v1.exe")
        log_info "文件大小: ${FILE_SIZE} 字节"
    else
        log_error "编译失败"
        exit 1
    fi
}

# ========================================
# 阶段2: 自举编译
# ========================================
stage2_bootstrap() {
    log ""
    log "========================================"
    log "阶段2: 使用cn_v1编译CN源码"
    log "========================================"
    
    log_info "编译中..."
    
    ${BUILD_DIR}/cn_v1.exe ${SRC_CN} -o ${BUILD_DIR}/cn_v2.exe 2>&1 | tee -a ${LOG_FILE}
    
    if [ -f "${BUILD_DIR}/cn_v2.exe" ]; then
        log_success "生成 cn_v2.exe"
        
        # 记录文件信息
        FILE_SIZE=$(stat -c%s "${BUILD_DIR}/cn_v2.exe" 2>/dev/null || stat -f%z "${BUILD_DIR}/cn_v2.exe")
        log_info "文件大小: ${FILE_SIZE} 字节"
    else
        log_error "编译失败"
        exit 1
    fi
}

# ========================================
# 阶段3: 验证阶段
# ========================================
stage3_verify() {
    log ""
    log "========================================"
    log "阶段3: 使用cn_v2编译CN源码"
    log "========================================"
    
    log_info "编译中..."
    
    ${BUILD_DIR}/cn_v2.exe ${SRC_CN} -o ${BUILD_DIR}/cn_v3.exe 2>&1 | tee -a ${LOG_FILE}
    
    if [ -f "${BUILD_DIR}/cn_v3.exe" ]; then
        log_success "生成 cn_v3.exe"
        
        # 记录文件信息
        FILE_SIZE=$(stat -c%s "${BUILD_DIR}/cn_v3.exe" 2>/dev/null || stat -f%z "${BUILD_DIR}/cn_v3.exe")
        log_info "文件大小: ${FILE_SIZE} 字节"
    else
        log_error "编译失败"
        exit 1
    fi
}

# ========================================
# 验证: 比较编译结果
# ========================================
verify_results() {
    log ""
    log "========================================"
    log "验证: 比较编译结果"
    log "========================================"
    
    # 计算哈希
    log_info "计算文件哈希..."
    
    HASH_V1=$(sha256sum ${BUILD_DIR}/cn_v1.exe | cut -d' ' -f1)
    HASH_V2=$(sha256sum ${BUILD_DIR}/cn_v2.exe | cut -d' ' -f1)
    HASH_V3=$(sha256sum ${BUILD_DIR}/cn_v3.exe | cut -d' ' -f1)
    
    log_info "v1 哈希: ${HASH_V1}"
    log_info "v2 哈希: ${HASH_V2}"
    log_info "v3 哈希: ${HASH_V3}"
    
    # 比较
    log ""
    log_info "比较 v1 和 v2..."
    if [ "$HASH_V1" = "$HASH_V2" ]; then
        log_success "v1 == v2"
    else
        log_warning "v1 != v2 (这是正常的，因为编译器实现不同)"
    fi
    
    log_info "比较 v2 和 v3..."
    if [ "$HASH_V2" = "$HASH_V3" ]; then
        log ""
        log "========================================"
        log_success "✅ 自举验证通过！"
        log "========================================"
        log ""
        log_success "三个版本哈希一致，编译器实现正确。"
        log_success "CN语言编译器已成功实现自托管。"
        
        # 保存成功标记
        echo "SUCCESS" > ${BUILD_DIR}/bootstrap_success
        
        return 0
    else
        log ""
        log "========================================"
        log_error "❌ 自举验证失败！"
        log "========================================"
        log ""
        log_error "v2 和 v3 哈希不一致，需要调查差异。"
        
        # 生成差异报告
        generate_diff_report
        
        return 1
    fi
}

# ========================================
# 生成差异报告
# ========================================
generate_diff_report() {
    log ""
    log_info "生成差异报告..."
    
    DIFF_REPORT="${LOG_DIR}/diff_report_${TIMESTAMP}.txt"
    
    echo "CN语言自托管编译器差异报告" > ${DIFF_REPORT}
    echo "生成时间: $(date)" >> ${DIFF_REPORT}
    echo "" >> ${DIFF_REPORT}
    echo "文件哈希:" >> ${DIFF_REPORT}
    echo "  v1: ${HASH_V1}" >> ${DIFF_REPORT}
    echo "  v2: ${HASH_V2}" >> ${DIFF_REPORT}
    echo "  v3: ${HASH_V3}" >> ${DIFF_REPORT}
    echo "" >> ${DIFF_REPORT}
    
    # 比较生成的C代码（如果存在）
    if [ -f "${BUILD_DIR}/v2.c" ] && [ -f "${BUILD_DIR}/v3.c" ]; then
        echo "C代码差异:" >> ${DIFF_REPORT}
        diff ${BUILD_DIR}/v2.c ${BUILD_DIR}/v3.c >> ${DIFF_REPORT} 2>&1 || true
    fi
    
    log_info "差异报告已保存到: ${DIFF_REPORT}"
}

# ========================================
# 清理
# ========================================
clean() {
    log_info "清理构建目录..."
    rm -rf ${BUILD_DIR}
    rm -rf ${LOG_DIR}
    log_success "清理完成"
}

# ========================================
# 主流程
# ========================================
main() {
    log ""
    log "========================================"
    log "   CN语言自托管编译器引导流程"
    log "========================================"
    log "开始时间: $(date)"
    log ""
    
    # 解析参数
    if [ "$1" == "--clean" ]; then
        clean
        exit 0
    fi
    
    # 执行各阶段
    stage0_prepare
    stage1_compile
    stage2_bootstrap
    stage3_verify
    
    # 验证结果
    if verify_results; then
        log ""
        log "========================================"
        log "引导流程完成"
        log "结束时间: $(date)"
        log "========================================"
        exit 0
    else
        log ""
        log "========================================"
        log "引导流程失败"
        log "结束时间: $(date)"
        log "========================================"
        exit 1
    fi
}

# 运行主流程
main "$@"
```

### 3.2 Windows引导脚本

```powershell
# bootstrap.ps1 - CN语言自托管引导脚本 (Windows)
# 用法: .\bootstrap.ps1 [-Verify] [-Clean]

param(
    [switch]$Verify,
    [switch]$Clean
)

# 配置
$BuildDir = ".\build"
$SrcCN = ".\src_cn"
$LogDir = ".\logs"
$Timestamp = Get-Date -Format "yyyyMMdd_HHmmss"
$LogFile = "$LogDir\bootstrap_$Timestamp.log"

# 创建日志目录
New-Item -ItemType Directory -Force -Path $LogDir | Out-Null

# 日志函数
function Log-Info {
    param([string]$Message)
    $output = "[INFO] $Message"
    Write-Host $output -ForegroundColor Blue
    Add-Content -Path $LogFile -Value $output
}

function Log-Success {
    param([string]$Message)
    $output = "[SUCCESS] $Message"
    Write-Host $output -ForegroundColor Green
    Add-Content -Path $LogFile -Value $output
}

function Log-Warning {
    param([string]$Message)
    $output = "[WARNING] $Message"
    Write-Host $output -ForegroundColor Yellow
    Add-Content -Path $LogFile -Value $output
}

function Log-Error {
    param([string]$Message)
    $output = "[ERROR] $Message"
    Write-Host $output -ForegroundColor Red
    Add-Content -Path $LogFile -Value $output
}

# ========================================
# 阶段0: 准备阶段
# ========================================
function Stage0-Prepare {
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "阶段0: 准备阶段" -ForegroundColor Cyan
    Write-Host "========================================" -ForegroundColor Cyan
    
    # 检查引导编译器
    Log-Info "检查引导编译器..."
    
    $CncPath = Join-Path $BuildDir "bin\Release\cnc.exe"
    if (-not (Test-Path $CncPath)) {
        $CncPath = Join-Path $BuildDir "bin\cnc.exe"
    }
    
    if (-not (Test-Path $CncPath)) {
        Log-Warning "未找到引导编译器，正在构建..."
        
        New-Item -ItemType Directory -Force -Path $BuildDir | Out-Null
        Set-Location $BuildDir
        
        cmake .. -DBUILD_BOOTSTRAP=ON -DBUILD_SELF_HOSTED=OFF -G "Visual Studio 17 2022" -A x64
        cmake --build . --config Release
        
        Set-Location ..
        $CncPath = Join-Path $BuildDir "bin\Release\cnc.exe"
    }
    
    $script:CNC_COMPILER = $CncPath
    Log-Success "引导编译器: $CncPath"
    
    # 检查CN源码目录
    Log-Info "检查CN源码目录..."
    if (-not (Test-Path $SrcCN)) {
        Log-Error "未找到CN源码目录: $SrcCN"
        exit 1
    }
    
    $CNFileCount = (Get-ChildItem -Path $SrcCN -Filter "*.cn" -Recurse).Count
    Log-Success "找到 $CNFileCount 个CN源文件"
}

# ========================================
# 阶段1: 首次编译
# ========================================
function Stage1-Compile {
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "阶段1: 使用C编译器编译CN源码" -ForegroundColor Cyan
    Write-Host "========================================" -ForegroundColor Cyan
    
    Log-Info "编译中..."
    
    $OutputPath = Join-Path $BuildDir "cn_v1.exe"
    & $CNC_COMPILER $SrcCN "-o" $OutputPath 2>&1 | Add-Content -Path $LogFile
    
    if (Test-Path $OutputPath) {
        Log-Success "生成 cn_v1.exe"
        
        $FileSize = (Get-Item $OutputPath).Length
        Log-Info "文件大小: $FileSize 字节"
    } else {
        Log-Error "编译失败"
        exit 1
    }
}

# ========================================
# 阶段2: 自举编译
# ========================================
function Stage2-Bootstrap {
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "阶段2: 使用cn_v1编译CN源码" -ForegroundColor Cyan
    Write-Host "========================================" -ForegroundColor Cyan
    
    Log-Info "编译中..."
    
    $CompilerPath = Join-Path $BuildDir "cn_v1.exe"
    $OutputPath = Join-Path $BuildDir "cn_v2.exe"
    
    & $CompilerPath $SrcCN "-o" $OutputPath 2>&1 | Add-Content -Path $LogFile
    
    if (Test-Path $OutputPath) {
        Log-Success "生成 cn_v2.exe"
        
        $FileSize = (Get-Item $OutputPath).Length
        Log-Info "文件大小: $FileSize 字节"
    } else {
        Log-Error "编译失败"
        exit 1
    }
}

# ========================================
# 阶段3: 验证阶段
# ========================================
function Stage3-Verify {
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "阶段3: 使用cn_v2编译CN源码" -ForegroundColor Cyan
    Write-Host "========================================" -ForegroundColor Cyan
    
    Log-Info "编译中..."
    
    $CompilerPath = Join-Path $BuildDir "cn_v2.exe"
    $OutputPath = Join-Path $BuildDir "cn_v3.exe"
    
    & $CompilerPath $SrcCN "-o" $OutputPath 2>&1 | Add-Content -Path $LogFile
    
    if (Test-Path $OutputPath) {
        Log-Success "生成 cn_v3.exe"
        
        $FileSize = (Get-Item $OutputPath).Length
        Log-Info "文件大小: $FileSize 字节"
    } else {
        Log-Error "编译失败"
        exit 1
    }
}

# ========================================
# 验证: 比较编译结果
# ========================================
function Verify-Results {
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "验证: 比较编译结果" -ForegroundColor Cyan
    Write-Host "========================================" -ForegroundColor Cyan
    
    Log-Info "计算文件哈希..."
    
    $V1Path = Join-Path $BuildDir "cn_v1.exe"
    $V2Path = Join-Path $BuildDir "cn_v2.exe"
    $V3Path = Join-Path $BuildDir "cn_v3.exe"
    
    $HashV1 = (Get-FileHash -Path $V1Path -Algorithm SHA256).Hash
    $HashV2 = (Get-FileHash -Path $V2Path -Algorithm SHA256).Hash
    $HashV3 = (Get-FileHash -Path $V3Path -Algorithm SHA256).Hash
    
    Log-Info "v1 哈希: $HashV1"
    Log-Info "v2 哈希: $HashV2"
    Log-Info "v3 哈希: $HashV3"
    
    # 比较
    Write-Host ""
    Log-Info "比较 v2 和 v3..."
    
    if ($HashV2 -eq $HashV3) {
        Write-Host ""
        Write-Host "========================================" -ForegroundColor Green
        Log-Success "✅ 自举验证通过！"
        Write-Host "========================================" -ForegroundColor Green
        Write-Host ""
        Log-Success "三个版本哈希一致，编译器实现正确。"
        Log-Success "CN语言编译器已成功实现自托管。"
        
        # 保存成功标记
        "SUCCESS" | Out-File -FilePath (Join-Path $BuildDir "bootstrap_success")
        
        return $true
    } else {
        Write-Host ""
        Write-Host "========================================" -ForegroundColor Red
        Log-Error "❌ 自举验证失败！"
        Write-Host "========================================" -ForegroundColor Red
        Write-Host ""
        Log-Error "v2 和 v3 哈希不一致，需要调查差异。"
        
        return $false
    }
}

# ========================================
# 清理
# ========================================
function Clean-Build {
    Log-Info "清理构建目录..."
    Remove-Item -Recurse -Force $BuildDir -ErrorAction SilentlyContinue
    Remove-Item -Recurse -Force $LogDir -ErrorAction SilentlyContinue
    Log-Success "清理完成"
}

# ========================================
# 主流程
# ========================================
Write-Host ""
Write-Host "========================================" -ForegroundColor Cyan
Write-Host "   CN语言自托管编译器引导流程" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host "开始时间: $(Get-Date)" -ForegroundColor White
Write-Host ""

# 清理
if ($Clean) {
    Clean-Build
    exit 0
}

# 执行各阶段
Stage0-Prepare
Stage1-Compile
Stage2-Bootstrap
Stage3-Verify

# 验证结果
if (Verify-Results) {
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "引导流程完成" -ForegroundColor Cyan
    Write-Host "结束时间: $(Get-Date)" -ForegroundColor White
    Write-Host "========================================" -ForegroundColor Cyan
    exit 0
} else {
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "引导流程失败" -ForegroundColor Cyan
    Write-Host "结束时间: $(Get-Date)" -ForegroundColor White
    Write-Host "========================================" -ForegroundColor Cyan
    exit 1
}
```

```bash
    echo "✅ C版本编译器可用"
else
    echo "❌ C版本编译器不可用"
    exit 1
fi

# 4. 更新文档
echo "[4/4] 更新文档..."
echo "回退时间: $(date)" >> docs/回退日志.md
echo "回退原因: $1" >> docs/回退日志.md

echo ""
echo "✅ 回退完成，已恢复到C实现版本"
```
