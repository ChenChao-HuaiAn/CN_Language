### 3.1 诊断系统设计

```cn
// src_cn/诊断/诊断器.cn
模块 诊断系统 {
public:
    // 诊断严重级别
    枚举 诊断级别 {
        错误,
        警告,
        信息,
        注释
    }
    
    // 错误代码
    枚举 错误代码 {
        // 词法错误 (1-99)
        未终止的字符串,
        未终止的字符,
        无效的进制,
        数字溢出,
        未终止的块注释,
        非法字符,
        无效的转义序列,
        
        // 语法错误 (100-199)
        期望的词元,
        期望的函数名,
        期望的参数列表,
        期望的变量名,
        期望的表达式,
        缺少分号,
        缺少右括号,
        缺少右大括号,
        
        // 语义错误 (200-299)
        未定义的标识符,
        重复定义的标识符,
        类型不匹配,
        参数数量不匹配,
        缺少返回语句,
        无效的赋值,
        未初始化的常量,
        静态变量初始化错误,
        无效的类型转换,
        未实现的抽象方法,
        
        // 常量错误 (300-399)
        常量缺少初始化,
        常量表达式错误,
        switch_case不是常量,
        常量重定义
    }
    
    // 源位置
    结构体 源位置 {
        字符串 文件名;
        整数 行号;
        整数 列号;
    }
    
    // 诊断信息
    结构体 诊断信息 {
        诊断级别 级别;
        错误代码 代码;
        源位置 位置;
        字符串 消息;
        字符串 提示;
        字符串 建议;
    }
    
    // 恢复策略
    枚举 恢复策略 {
        无恢复,
        跳过词元,
        同步到分隔符,
        插入词元
    }
    
    // 诊断管理器
    类 诊断管理器 {
    私有:
        诊断信息列表 诊断集合;
        整数 错误计数;
        整数 警告计数;
        整数 最大错误数;
        布尔 抑制级联错误;
        语言类型 当前语言;
        
    公开:
        函数 创建() -> 诊断管理器* {
            变量 管理器 = 分配内存(诊断管理器大小) 作为 诊断管理器*;
            管理器.错误计数 = 0;
            管理器.警告计数 = 0;
            管理器.最大错误数 = 100;
            管理器.抑制级联错误 = 真;
            管理器.当前语言 = 语言类型.中文;
            返回 管理器;
        }
        
        函数 报告错误(源位置 位置, 错误代码 代码, 字符串[] 参数) -> 恢复策略 {
            如果 (错误计数 >= 最大错误数) {
                返回 恢复策略.无恢复;
            }
            
            变量 诊断 = 创建诊断信息(诊断级别.错误, 代码, 位置, 参数);
            诊断集合.添加(诊断);
            错误计数 = 错误计数 + 1;
            
            返回 获取恢复策略(代码);
        }
        
        函数 报告警告(源位置 位置, 错误代码 代码, 字符串[] 参数) -> 空类型 {
            变量 诊断 = 创建诊断信息(诊断级别.警告, 代码, 位置, 参数);
            诊断集合.添加(诊断);
            警告计数 = 警告计数 + 1;
        }
        
        函数 有错误() -> 布尔 {
            返回 错误计数 > 0;
        }
        
        函数 获取错误计数() -> 整数 {
            返回 错误计数;
        }
        
        函数 获取警告计数() -> 整数 {
            返回 警告计数;
        }
        
        函数 应该继续() -> 布尔 {
            返回 错误计数 < 最大错误数;
        }
        
        函数 打印所有诊断() -> 空类型 {
            循环 (整数 i = 0; i < 诊断集合.长度; i = i + 1) {
                打印诊断(诊断集合.获取(i));
            }
        }
        
        函数 清空() -> 空类型 {
            诊断集合.清空();
            错误计数 = 0;
            警告计数 = 0;
        }
        
        函数 设置语言(语言类型 语言) -> 空类型 {
            当前语言 = 语言;
        }
        
    私有:
        函数 创建诊断信息(诊断级别 级别, 错误代码 代码, 源位置 位置, 字符串[] 参数) -> 诊断信息 {
            变量 诊断 = 创建诊断信息结构();
            诊断.级别 = 级别;
            诊断.代码 = 代码;
            诊断.位置 = 位置;
            诊断.消息 = 获取消息(代码, 当前语言, 参数);
            诊断.提示 = 获取提示(代码, 当前语言);
            诊断.建议 = 获取建议(代码, 当前语言);
            返回 诊断;
        }
        
        函数 打印诊断(诊断信息 诊断) -> 空类型 {
            // 格式: 文件名:行号:列号: 级别: 消息
            变量 级别字符串 = 级别到字符串(诊断.级别);
            打印字符串(诊断.位置.文件名 + ":" + 
                        整数转字符串(诊断.位置.行号) + ":" +
                        整数转字符串(诊断.位置.列号) + ": " +
                        级别字符串 + ": " +
                        诊断.消息);
            
            如果 (诊断.提示 != "") {
                打印字符串("  提示: " + 诊断.提示);
            }
            
            如果 (诊断.建议 != "") {
                打印字符串("  建议: " + 诊断.建议);
            }
        }
        
        函数 级别到字符串(诊断级别 级别) -> 字符串 {
            选择 级别 {
                情况 诊断级别.错误:
                    返回 "错误";
                情况 诊断级别.警告:
                    返回 "警告";
                情况 诊断级别.信息:
                    返回 "信息";
                情况 诊断级别.注释:
                    返回 "注释";
                默认:
                    返回 "未知";
            }
        }
        
        函数 获取恢复策略(错误代码 代码) -> 恢复策略 {
            选择 代码 {
                情况 错误代码.缺少分号:
                    返回 恢复策略.插入词元;
                    
                情况 错误代码.缺少右括号:
                情况 错误代码.缺少右大括号:
                    返回 恢复策略.同步到分隔符;
                    
                默认:
                    返回 恢复策略.跳过词元;
            }
        }
    }
}
```
