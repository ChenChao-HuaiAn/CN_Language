```cn
// src_cn/诊断/诊断类型.cn
模块 诊断系统 {

        备注       // 附加说明
    }
    
    // 错误代码
    枚举 错误代码 {
        // 词法错误 (1xx)

        无效的转义序列 = 103,
        无效的数字格式 = 104,

        缺少右大括号 = 205,
        
        // 语义错误 (3xx)
```

```cn
// src_cn/诊断/诊断信息.cn
模块 诊断系统 {
公开:
    // 源位置
    结构体 源位置 {
        字符串 文件名;
        整数 行号;
        整数 列号;
    }
    
    // 诊断信息
    类 诊断信息 {
    私有:
        诊断级别 级别;

        函数 设置消息(字符串 消息值) -> 空类型;
        函数 设置提示(字符串 提示值) -> 空类型;
        函数 设置建议(字符串 建议值) -> 空类型;

        函数 获取代码() -> 错误代码 { 返回 代码; }

        函数 是错误() -> 布尔 { 返回 级别 == 诊断级别.错误; }
    }
```

### 3.4 错误消息表

```cn
// src_cn/诊断/消息表.cn
模块 诊断系统 {
私有:
    // 消息模板结构
    结构体 消息模板 {
        错误代码 代码;
        字符串 中文消息;
        字符串 英文消息;
        字符串 中文提示;
        字符串 英文提示;
    }
    
    // 消息表
    常量 消息模板[] 消息表 = [
        {
            代码: 错误代码.未定义的标识符,
            中文消息: "未定义的标识符 '{0}'",
            英文消息: "Undefined identifier '{0}'",
            中文提示: "请检查标识符是否正确拼写",
            英文提示: "Check if the identifier is spelled correctly"
        },
        {
            代码: 错误代码.类型不匹配,
            中文消息: "类型不匹配: 期望 '{0}', 实际 '{1}'",
            英文消息: "Type mismatch: expected '{0}', got '{1}'",
            中文提示: "请检查表达式类型是否正确",
            英文提示: "Check if the expression type is correct"
        },
        {
            代码: 错误代码.缺少分号,
            中文消息: "缺少分号",
            英文消息: "Missing semicolon",
            中文提示: "在语句末尾添加分号 ';'",
            英文提示: "Add a semicolon ';' at the end of the statement"
        }
        // ... 更多消息模板
    ]
    
公开:
    函数 获取消息(错误代码 代码, 语言类型 语言) -> 字符串 {
        循环 变量 模板 在 消息表 {
            如果 (模板.代码 == 代码) {
                如果 (语言 == 语言类型.中文) {
                    返回 模板.中文消息;
                } 否则 {
                    返回 模板.英文消息;
                }
            }
        }
        返回 "未知错误";
    }
    
    函数 格式化消息(字符串 模板, 字符串[] 参数) -> 字符串 {
        变量 结果 = 模板;
        循环 变量 i = 0; i < 参数.长度; i = i + 1 {
            变量 占位符 = "{" + i + "}";
            结果 = 结果.替换(占位符, 参数[i]);
        }
        返回 结果;
    }
}
```

### 3.5 错误恢复机制

```cn
// src_cn/诊断/错误恢复.cn
模块 诊断系统 {
公开:
    类 错误恢复器 {
    私有:
        整数 连续错误计数;
        整数 最大连续错误;
        集合 同步标记集合;
        
    公开:
        函数 创建() -> 错误恢复器* {
            变量 实例 = 新建 错误恢复器();
            实例.最大连续错误 = 3;
            实例.初始化同步标记();
            返回 实例;
        }
        
        函数 选择恢复策略(错误代码 代码) -> 恢复策略 {
            选择 代码 {
                情况 错误代码.缺少分号:
                    返回 恢复策略.插入词元;
                情况 错误代码.意外的词元:
                    返回 恢复策略.跳过词元;
                情况 错误代码.缺少右大括号:
                    返回 恢复策略.同步到分隔符;
                默认:
                    返回 恢复策略.跳过词元;
            }
        }
        
        函数 是同步标记(词元类型 类型) -> 布尔 {
            返回 同步标记集合.包含(类型);
        }
        
        函数 应该抑制() -> 布尔 {
            返回 连续错误计数 >= 最大连续错误;
        }
        
        函数 重置() -> 空类型 {
            连续错误计数 = 0;
        }
        
    私有:
        函数 初始化同步标记() -> 空类型 {
            同步标记集合.添加(词元类型.分号);
            同步标记集合.添加(词元类型.右大括号);
            同步标记集合.添加(词元类型.右括号);
            同步标记集合.添加(词元类型.关键字_函数);
            同步标记集合.添加(词元类型.关键字_变量);
            同步标记集合.添加(词元类型.关键字_类);
        }
    }
}
```
