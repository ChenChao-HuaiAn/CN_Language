## 3. 详细任务描述

### 3.1 类型映射设计

```cn
// src_cn/代码生成器/类型映射.cn
模块 代码生成器 {
公开:
    类 类型映射器 {
    公开:
        函数 映射类型(类型* CN类型) -> 字符串 {
            选择 CN类型.类别 {
                情况 类型类别.基础类型:
                    返回 映射基础类型(CN类型)
                情况 类型类别.数组类型:
                    返回 映射数组类型(CN类型)
                情况 类型类别.结构体类型:
                    返回 映射结构体类型(CN类型)
                情况 类型类别.指针类型:
                    返回 映射指针类型(CN类型)
                情况 类型类别.函数类型:
                    返回 映射函数类型(CN类型)
                默认:
                    返回 "void"
            }
        }
        
    私有:
        函数 映射基础类型(类型* CN类型) -> 字符串 {
            如果 (CN类型 == 整数类型) 返回 "int"
            如果 (CN类型 == 小数类型) 返回 "double"
            如果 (CN类型 == 布尔类型) 返回 "int"
            如果 (CN类型 == 字符串类型) 返回 "char*"
            如果 (CN类型 == 空类型值) 返回 "void"
            返回 "int"  // 默认
        }
        
        函数 映射数组类型(类型* CN类型) -> 字符串 {
            变量 元素类型 = 映射类型(CN类型.元素类型)
            返回 元素类型 + "*"
        }
        
        函数 映射指针类型(类型* CN类型) -> 字符串 {
            变量 目标类型 = 映射类型(CN类型.目标类型)
            返回 目标类型 + "*"
        }
    }
}
```

### 3.3 指令生成示例

```cn
// 算术指令生成
函数 生成算术指令(IR指令* 指令) -> 空类型 {
    变量 目标 = 获取值名称(指令.目标);
    变量 操作数1 = 获取值名称(指令.操作数1);
    变量 操作数2 = 获取值名称(指令.操作数2);
    变量 运算符 = "";
    
    选择 指令.操作码 {
        情况 IR操作码.加法:
            运算符 = "+";
        情况 IR操作码.减法:
            运算符 = "-";
        情况 IR操作码.乘法:
            运算符 = "*";
        情况 IR操作码.除法:
            运算符 = "/";
        情况 IR操作码.取模:
            运算符 = "%";
    }
    
    发送格式化("    %s = %s %s %s;", 目标, 操作数1, 运算符, 操作数2);
}

// 函数调用生成
函数 生成调用指令(IR指令* 指令) -> 空类型 {
    变量 目标 = 获取值名称(指令.目标);
    变量 函数名 = 获取值名称(指令.操作数1);
    变量 参数列表 = 生成参数列表(指令.参数);
    
    如果 (指令.目标 != 空) {
        发送格式化("    %s = %s(%s);", 目标, 函数名, 参数列表);
    } 否则 {
        发送格式化("    %s(%s);", 函数名, 参数列表);
    }
}

// 条件跳转生成
函数 生成条件跳转指令(IR指令* 指令) -> 空类型 {
    变量 条件 = 获取值名称(指令.操作数1);
    变量 真标签 = 获取块标签(指令.真目标);
    变量 假标签 = 获取块标签(指令.假目标);
    
    发送格式化("    if (%s) goto %s;", 条件, 真标签);
    发送格式化("    goto %s;", 假标签);
}
```

### 3.4 运行时调用生成

```cn
// src_cn/代码生成器/运行时调用.cn
模块 代码生成器 {
公开:
    类 运行时调用生成器 {
    私有:
        映射 运行时函数表;
        
    公开:
        函数 创建() -> 运行时调用生成器* {
            变量 实例 = 新建 运行时调用生成器();
            实例.初始化函数表();
            返回 实例;
        }
        
        函数 获取运行时函数名(字符串 CN函数名) -> 字符串 {
            如果 (运行时函数表.包含(CN函数名)) {
                返回 运行时函数表[CN函数名];
            }
            返回 CN函数名;  // 默认使用原名
        }
        
    私有:
        函数 初始化函数表() -> 空类型 {
            // 内存管理
            运行时函数表["分配内存"] = "cn_rt_malloc";
            运行时函数表["释放内存"] = "cn_rt_free";
            运行时函数表["重新分配"] = "cn_rt_realloc";
            
            // 字符串操作

            运行时函数表["字符串长度"] = "cn_rt_string_length";
            运行时函数表["字符串比较"] = "cn_rt_string_compare";
            
            // 数组操作
            运行时函数表["创建数组"] = "cn_rt_vector_create";
            运行时函数表["数组添加"] = "cn_rt_vector_push";
            运行时函数表["数组获取"] = "cn_rt_vector_get";
            
            // 文件操作
            运行时函数表["打开文件"] = "cn_rt_file_open";
            运行时函数表["读取文件"] = "cn_rt_file_read";
            运行时函数表["写入文件"] = "cn_rt_file_write";
            运行时函数表["关闭文件"] = "cn_rt_file_close";
            
            // 命令行参数
            运行时函数表["获取参数个数"] = "cn_rt_cli_argc";
            运行时函数表["获取参数"] = "cn_rt_cli_argv";
        }
    }
}
```
