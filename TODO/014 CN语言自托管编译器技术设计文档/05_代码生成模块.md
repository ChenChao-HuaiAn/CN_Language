## 1. 任务概述

### 1.1 任务目标

将现有C语言实现的代码生成器（`src/backend/`）使用CN语言重写，实现自托管编译器的C代码生成功能。

### 1.2 功能要求

| 功能项 | 说明 | 优先级 |
|--------|------|--------|
| C代码生成 | 将IR转换为C代码 | ⭐⭐⭐ 高 |
| 函数代码生成 | 生成函数的C代码 | ⭐⭐⭐ 高 |
| 类型映射 | CN类型到C类型的映射 | ⭐⭐⭐ 高 |
| 运行时调用 | 生成运行时库调用 | ⭐⭐⭐ 高 |
| 头文件生成 | 生成必要的头文件包含 | ⭐⭐ 中 |
| 代码格式化 | 生成格式化的C代码 | ⭐⭐ 中 |

### 1.3 验收标准

- [ ] 代码生成器能正确转换所有IR指令
- [ ] 生成的C代码可编译
- [ ] 生成的C代码行为正确
- [ ] 代码生成测试用例100%通过
- [ ] 与C版本代码生成器输出一致

### 3.1 类型映射

```cn
// src_cn/代码生成器/类型映射.cn
模块 代码生成器 {
public:
    类 类型映射器 {
    公开:
        函数 映射类型(类型节点* 类型) -> 字符串 {
            选择 类型.种类 {
                情况 类型种类.空类型:
                    返回 "void";
                    
                情况 类型种类.整数类型:
                    返回 "int";
                    
                情况 类型种类.小数类型:
                    返回 "double";
                    
                情况 类型种类.字符串类型:
                    返回 "const char*";
                    
                情况 类型种类.布尔类型:
                    返回 "int";
                    
                情况 类型种类.指针类型:
                    变量 元素类型 = 映射类型(类型.元素类型);
                    返回 元素类型 + "*";
                    
                情况 类型种类.数组类型:
                    变量 元素类型 = 映射类型(类型.元素类型);
                    返回 元素类型 + "*";  // 数组作为指针传递
                    
                情况 类型种类.函数类型:
                    返回 生成函数指针类型(类型);
                    
                情况 类型种类.结构体类型:
                    返回 "struct " + 类型.名称;
                    
                情况 类型种类.枚举类型:
                    返回 "enum " + 类型.名称;
                    
                情况 类型种类.类类型:
                    返回 "struct " + 类型.名称;
                    
                情况 类型种类.接口类型:
                    返回 "struct " + 类型.名称;
                    
                默认:
                    返回 "void";
            }
        }
        
        函数 映射运算符(IR操作码 操作码) -> 字符串 {
            选择 操作码 {
                情况 IR操作码.加:
                    返回 "+";
                情况 IR操作码.减:
                    返回 "-";
                情况 IR操作码.乘:
                    返回 "*";
                情况 IR操作码.除:
                    返回 "/";
                情况 IR操作码.取模:
                    返回 "%";
                情况 IR操作码.相等:
                    返回 "==";
                情况 IR操作码.不等:
                    返回 "!=";
                情况 IR操作码.小于:
                    返回 "<";
                情况 IR操作码.小于等于:
                    返回 "<=";
                情况 IR操作码.大于:
                    返回 ">";
                情况 IR操作码.大于等于:
                    返回 ">=";
                情况 IR操作码.逻辑与:
                    返回 "&&";
                情况 IR操作码.逻辑或:
                    返回 "||";
                情况 IR操作码.按位与:
                    返回 "&";
                情况 IR操作码.按位或:
                    返回 "|";
                情况 IR操作码.按位异或:
                    返回 "^";
                情况 IR操作码.左移:
                    返回 "<<";
                情况 IR操作码.右移:
                    返回 ">>";
                默认:
                    返回 "";
            }
        }
        
    私有:
        函数 生成函数指针类型(类型节点* 类型) -> 字符串 {
            变量 返回类型 = 映射类型(类型.返回类型);
            变量 参数列表 = "";
            
            循环 (整数 i = 0; i < 类型.参数类型.长度; i = i + 1) {
                如果 (i > 0) {
                    参数列表 = 参数列表 + ", ";
                }
                参数列表 = 参数列表 + 映射类型(类型.参数类型.获取(i));
            }
            
            如果 (参数列表 == "") {
                参数列表 = "void";
            }
            
            返回 返回类型 + "(*)(" + 参数列表 + ")";
        }
    }
}
```

### 3.2 代码生成器类设计

```cn
// src_cn/代码生成器/代码生成器.cn
模块 代码生成器 {
public:
    类 代码生成器 {
    私有:
        IR模块* 当前模块;
        类型映射器* 类型映射;
        字符串缓冲 输出缓冲;
        整数 缩进级别;
        诊断管理器* 诊断;
        
    公开:
        函数 创建(诊断管理器* 诊断器) -> 代码生成器* {
            变量 生成器 = 分配内存(代码生成器大小) 作为 代码生成器*;
            生成器.诊断 = 诊断器;
            生成器.类型映射 = 类型映射器.创建();
            生成器.缩进级别 = 0;
            返回 生成器;
        }
        
        函数 生成(IR模块* 模块, 字符串 输出路径) -> 布尔 {
            当前模块 = 模块;
            输出缓冲 = 创建字符串缓冲();
            
            // 生成文件头
            生成文件头();
            
            // 生成结构体定义
            生成结构体定义();
            
            // 生成枚举定义
            生成枚举定义();
            
            // 生成函数声明
            生成函数声明();
            
            // 生成函数定义
            生成函数定义();
            
            // 写入文件
            返回 写入文件(输出路径, 输出缓冲.内容);
        }
        
    私有:
        函数 生成文件头() -> 空类型 {
            输出行("/* 由CN编译器生成的C代码 */");
            输出行("#include <stdio.h>");
            输出行("#include <stdlib.h>");
            输出行("#include <string.h>");
            输出行("#include <stdint.h>");
            输出行("");
            输出行("#include \"cnlang/runtime/stdlib.h\"");
            输出行("#include \"cnlang/runtime/runtime.h\"");
            输出行("");
        }
        
        函数 生成结构体定义() -> 空类型 {
            // 遍历模块中的结构体定义
            // ...
        }
        
        函数 生成枚举定义() -> 空类型 {
            // 遍历模块中的枚举定义
            // ...
        }
        
        函数 生成函数声明() -> 空类型 {
            循环 (整数 i = 0; i < 当前模块.函数.长度; i = i + 1) {
                变量 函数 = 当前模块.函数.获取(i);
                生成函数原型(函数);
                输出(";");
                输出新行();
            }
            输出新行();
        }
        
        函数 生成函数定义() -> 空类型 {
            循环 (整数 i = 0; i < 当前模块.函数.长度; i = i + 1) {
                变量 函数 = 当前模块.函数.获取(i);
                生成函数(函数);
            }
        }
        
        函数 生成函数原型(IR函数* 函数) -> 空类型 {
            // 返回类型
            变量 返回类型 = 类型映射.映射类型(函数.返回类型);
            输出(返回类型 + " ");
            
            // 函数名
            输出(函数.名称);
            
            // 参数列表
            输出("(");
            
            如果 (函数.参数.长度 == 0) {
                输出("void");
            } 否则 {
                循环 (整数 i = 0; i < 函数.参数.长度; i = i + 1) {
                    如果 (i > 0) {
                        输出(", ");
                    }
                    变量 参数 = 函数.参数.获取(i);
                    输出(类型映射.映射类型(参数.类型) + " " + 参数.名称);
                }
            }
            
            输出(")");
        }
        
        函数 生成函数(IR函数* 函数) -> 空类型 {
            // 函数原型
            生成函数原型(函数);
            输出新行();
            
            // 函数体
            输出("{");
            输出新行();
            增加缩进();
            
            // 局部变量声明
            生成局部变量声明(函数);
            
            // 基本块
            循环 (整数 i = 0; i < 函数.基本块.长度; i = i + 1) {
                变量 块 = 函数.基本块.获取(i);
                生成基本块(块);
            }
            
            减少缩进();
            输出("}");
            输出新行();
            输出新行();
        }
        
        函数 生成局部变量声明(IR函数* 函数) -> 空类型 {
            // 为每个虚拟寄存器生成临时变量
            循环 (整数 i = 0; i < 函数.虚拟寄存器计数; i = i + 1) {
                输出行("int _t" + i + ";");
            }
        }
        
        函数 生成基本块(基本块* 块) -> 空类型 {
            // 标签
            输出(块.标签 + ":");
            输出新行();
            
            增加缩进();
            
            // 指令
            变量 指令 = 块.第一条指令;
            当 (指令 != 无) {
                生成指令(指令);
                指令 = 指令.下一条;
            }
            
            减少缩进();
        }
        
        函数 生成指令(IR指令* 指令) -> 空类型 {
            选择 指令.操作码 {
                情况 IR操作码.加:
                情况 IR操作码.减:
                情况 IR操作码.乘:
                情况 IR操作码.除:
                情况 IR操作码.取模:
                    生成二元运算(指令);
                    中断;
                    
                情况 IR操作码.相等:
                情况 IR操作码.不等:
                情况 IR操作码.小于:
                情况 IR操作码.小于等于:
                情况 IR操作码.大于:
                情况 IR操作码.大于等于:
                    生成比较运算(指令);
                    中断;
                    
                情况 IR操作码.逻辑与:
                情况 IR操作码.逻辑或:
                    生成逻辑运算(指令);
                    中断;
                    
                情况 IR操作码.按位与:
                情况 IR操作码.按位或:
                情况 IR操作码.按位异或:
                情况 IR操作码.左移:
                情况 IR操作码.右移:
                    生成位运算(指令);
                    中断;
                    
                情况 IR操作码.加载:
                    生成加载(指令);
                    中断;
                    
                情况 IR操作码.存储:
                    生成存储(指令);
                    中断;
                    
                情况 IR操作码.分配:
                    生成分配(指令);
                    中断;
                    
                情况 IR操作码.跳转:
                    生成跳转(指令);
                    中断;
                    
                情况 IR操作码.条件跳转:
                    生成条件跳转(指令);
                    中断;
                    
                情况 IR操作码.返回:
                    生成返回(指令);
                    中断;
                    
                情况 IR操作码.调用:
                    生成调用(指令);
                    中断;
                    
                情况 IR操作码.标签:
                    // 标签已处理
                    中断;
                    
                情况 IR操作码.空操作:
                    // 无操作
                    中断;
            }
        }
        
        函数 生成二元运算(IR指令* 指令) -> 空类型 {
            变量 目标 = 格式化值(指令.目标);
            变量 左 = 格式化值(指令.源操作数1);
            变量 右 = 格式化值(指令.源操作数2);
            变量 运算符 = 类型映射.映射运算符(指令.操作码);
            
            输出行(目标 + " = " + 左 + " " + 运算符 + " " + 右 + ";");
        }
        
        函数 生成比较运算(IR指令* 指令) -> 空类型 {
            变量 目标 = 格式化值(指令.目标);
            变量 左 = 格式化值(指令.源操作数1);
            变量 右 = 格式化值(指令.源操作数2);
            变量 运算符 = 类型映射.映射运算符(指令.操作码);
            
            输出行(目标 + " = " + 左 + " " + 运算符 + " " + 右 + ";");
        }
        
        函数 生成逻辑运算(IR指令* 指令) -> 空类型 {
            变量 目标 = 格式化值(指令.目标);
            变量 左 = 格式化值(指令.源操作数1);
            变量 右 = 格式化值(指令.源操作数2);
            变量 运算符 = 类型映射.映射运算符(指令.操作码);
            
            输出行(目标 + " = " + 左 + " " + 运算符 + " " + 右 + ";");
        }
        
        函数 生成跳转(IR指令* 指令) -> 空类型 {
            变量 目标标签 = 指令.目标.名称;
            输出行("goto " + 目标标签 + ";");
        }
        
        函数 生成条件跳转(IR指令* 指令) -> 空类型 {
            变量 条件 = 格式化值(指令.源操作数1);
            变量 真标签 = 指令.目标.名称;
            变量 假标签 = 指令.源操作数2.名称;
            
            输出行("if (" + 条件 + ") goto " + 真标签 + ";");
            输出行("goto " + 假标签 + ";");
        }
        
        函数 生成返回(IR指令* 指令) -> 空类型 {
            如果 (指令.源操作数1.类型 == IR值类型.常量整数 &&
                  指令.源操作数1.整数值 == 0 &&
                  指令.目标.类型 == IR值类型.虚拟寄存器 &&
                  指令.目标.编号 == 0) {
                // 空返回
                输出行("return;");
            } 否则 {
                变量 返回值 = 格式化值(指令.源操作数1);
                输出行("return " + 返回值 + ";");
            }
        }
        
        函数 生成调用(IR指令* 指令) -> 空类型 {
            变量 目标 = 格式化值(指令.目标);
            变量 函数名 = 指令.源操作数1.名称;
            
            输出(目标 + " = " + 函数名 + "(");
            
            // 参数
            变量 参数列表 = 指令.源操作数2.作为参数列表;
            循环 (整数 i = 0; i < 参数列表.长度; i = i + 1) {
                如果 (i > 0) {
                    输出(", ");
                }
                输出(格式化值(参数列表.获取(i)));
            }
            
            输出(");");
            输出新行();
        }
        
        函数 生成加载(IR指令* 指令) -> 空类型 {
            变量 目标 = 格式化值(指令.目标);
            变量 地址 = 格式化值(指令.源操作数1);
            
            输出行(目标 + " = *" + 地址 + ";");
        }
        
        函数 生成存储(IR指令* 指令) -> 空类型 {
            变量 地址 = 格式化值(指令.目标);
            变量 值 = 格式化值(指令.源操作数1);
            
            输出行("*" + 地址 + " = " + 值 + ";");
        }
        
        函数 生成分配(IR指令* 指令) -> 空类型 {
            变量 目标 = 格式化值(指令.目标);
            变量 大小 = 指令.源操作数1.整数值;
            
            输出行(目标 + " = (int*)alloca(" + 大小 + " * sizeof(int));");
        }
        
        函数 格式化值(IR值 值) -> 字符串 {
            选择 值.类型 {
                情况 IR值类型.虚拟寄存器:
                    返回 "_t" + 值.编号;
                    
                情况 IR值类型.常量整数:
                    返回 整数转字符串(值.整数值);
                    
                情况 IR值类型.常量小数:
                    返回 小数转字符串(值.小数值);
                    
                情况 IR值类型.常量字符串:
                    返回 "\"" + 值.字符串值 + "\"";
                    
                情况 IR值类型.全局变量:
                    返回 值.名称;
                    
                情况 IR值类型.函数引用:
                    返回 值.名称;
                    
                情况 IR值类型.标签引用:
                    返回 值.名称;
                    
                情况 IR值类型.参数引用:
                    返回 值.名称;
                    
                默认:
                    返回 "0";
            }
        }
        
        // 输出辅助函数
        函数 输出(字符串 内容) -> 空类型 {
            输出缓冲.追加(内容);
        }
        
        函数 输出行(字符串 内容) -> 空类型 {
            输出(获取缩进());
            输出(内容);
            输出新行();
        }
        
        函数 输出新行() -> 空类型 {
            输出("\n");
        }
        
        函数 获取缩进() -> 字符串 {
            变量 结果 = "";
            循环 (整数 i = 0; i < 缩进级别; i = i + 1) {
                结果 = 结果 + "    ";
            }
            返回 结果;
        }
        
        函数 增加缩进() -> 空类型 {
            缩进级别 = 缩进级别 + 1;
        }
        
        函数 减少缩进() -> 空类型 {
            如果 (缩进级别 > 0) {
                缩进级别 = 缩进级别 - 1;
            }
        }
    }
}
```
