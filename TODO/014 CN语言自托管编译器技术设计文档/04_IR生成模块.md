## 1. 任务概述

### 1.1 任务目标

将现有C语言实现的IR生成器（`src/ir/`）使用CN语言重写，实现自托管编译器的中间表示生成功能。

### 1.2 功能要求

| 功能项 | 说明 | 优先级 |
|--------|------|--------|
| IR指令生成 | 将AST转换为IR指令序列 | ⭐⭐⭐ 高 |
| 基本块构建 | 构建基本块和控制流图 | ⭐⭐⭐ 高 |
| 函数IR生成 | 生成函数的IR表示 | ⭐⭐⭐ 高 |
| 表达式IR生成 | 生成表达式的IR指令 | ⭐⭐⭐ 高 |
| 控制流IR生成 | 生成控制流语句的IR | ⭐⭐⭐ 高 |
| 优化Pass | 常量折叠、死代码消除等 | ⭐⭐ 中 |

### 1.3 验收标准

- [ ] IR生成器能正确转换所有AST节点
- [ ] 基本块正确构建
- [ ] 控制流图正确生成
- [ ] IR生成测试用例100%通过
- [ ] 与C版本IR生成器输出一致

### 3.2 IR生成器类设计

```cn
// src_cn/IR生成器/IR生成器.cn
模块 IR生成器 {
public:
    类 IR生成器 {
    私有:
        IR模块* 当前模块;
        IR函数* 当前函数;
        基本块* 当前基本块;
        整数 寄存器计数;
        整数 标签计数;
        诊断管理器* 诊断;
        
    公开:
        函数 创建(诊断管理器* 诊断器) -> IR生成器* {
            变量 生成器 = 分配内存(IR生成器大小) 作为 IR生成器*;
            生成器.诊断 = 诊断器;
            生成器.寄存器计数 = 0;
            生成器.标签计数 = 0;
            返回 生成器;
        }
        
        函数 生成(程序节点* 程序) -> IR模块* {
            当前模块 = 创建IR模块();
            
            // 第一遍：收集所有函数声明
            循环 (整数 i = 0; i < 程序.声明.长度; i = i + 1) {
                变量 声明 = 程序.声明.获取(i);
                如果 (声明.类型 == 节点类型.函数声明) {
                    收集函数声明(声明 作为 函数声明*);
                }
            }
            
            // 第二遍：生成函数体
            循环 (整数 i = 0; i < 程序.声明.长度; i = i + 1) {
                变量 声明 = 程序.声明.获取(i);
                如果 (声明.类型 == 节点类型.函数声明) {
                    生成函数(声明 作为 函数声明*);
                }
            }
            
            返回 当前模块;
        }
        
    私有:
        函数 收集函数声明(函数声明* 声明) -> 空类型 {
            变量 函数 = 创建IR函数(声明.名称);
            函数.返回类型 = 声明.返回类型;
            
            // 添加参数
            循环 (整数 i = 0; i < 声明.参数.长度; i = i + 1) {
                变量 参数 = 声明.参数.获取(i);
                函数.参数.添加(参数);
            }
            
            当前模块.函数.添加(函数);
        }
        
        函数 生成函数(函数声明* 声明) -> 空类型 {
            当前函数 = 查找函数(声明.名称);
            寄存器计数 = 0;
            
            // 创建入口基本块
            当前基本块 = 创建基本块(生成标签("入口"));
            当前函数.基本块.添加(当前基本块);
            
            // 添加函数入口指令
            发射函数入口(声明);
            
            // 生成函数体
            如果 (声明.函数体 != 无) {
                生成块语句(声明.函数体);
            }
            
            // 添加隐式返回（如果需要）
            如果 (!当前基本块.以返回结尾()) {
                发射返回(无);
            }
            
            当前函数.虚拟寄存器计数 = 寄存器计数;
            当前函数 = 无;
        }
        
        函数 生成块语句(块语句* 块) -> 空类型 {
            循环 (整数 i = 0; i < 块.语句.长度; i = i + 1) {
                生成语句(块.语句.获取(i));
            }
        }
        
        函数 生成语句(语句节点* 语句) -> 空类型 {
            选择 语句.类型 {
                情况 节点类型.变量声明:
                    生成变量声明(语句 作为 变量声明*);
                    中断;
                    
                情况 节点类型.表达式语句:
                    生成表达式(语句 作为 表达式语句*.表达式);
                    中断;
                    
                情况 节点类型.返回语句:
                    生成返回语句(语句 作为 返回语句*);
                    中断;
                    
                情况 节点类型.如果语句:
                    生成如果语句(语句 作为 如果语句*);
                    中断;
                    
                情况 节点类型.当语句:
                    生成当语句(语句 作为 当语句*);
                    中断;
                    
                情况 节点类型.循环语句:
                    生成循环语句(语句 作为 循环语句*);
                    中断;
                    
                情况 节点类型.选择语句:
                    生成选择语句(语句 作为 选择语句*);
                    中断;
            }
        }
        
        函数 生成变量声明(变量声明* 声明) -> 空类型 {
            // 分配栈空间
            变量 地址 = 发射分配(声明.类型, 声明.名称);
            
            // 如果有初始值，生成存储指令
            如果 (声明.初始值 != 无) {
                变量 初始值 = 生成表达式(声明.初始值);
                发射存储(地址, 初始值);
            }
        }
        
        函数 生成表达式(表达式节点* 表达式) -> IR值 {
            选择 表达式.类型 {
                情况 节点类型.字面量表达式:
                    返回 生成字面量(表达式 作为 字面量表达式*);
                    
                情况 节点类型.标识符表达式:
                    返回 生成标识符(表达式 作为 标识符表达式*);
                    
                情况 节点类型.二元表达式:
                    返回 生成二元表达式(表达式 作为 二元表达式*);
                    
                情况 节点类型.一元表达式:
                    返回 生成一元表达式(表达式 作为 一元表达式*);
                    
                情况 节点类型.函数调用表达式:
                    返回 生成函数调用(表达式 作为 函数调用表达式*);
                    
                情况 节点类型.成员访问表达式:
                    返回 生成成员访问(表达式 作为 成员访问表达式*);
                    
                情况 节点类型.赋值表达式:
                    返回 生成赋值(表达式 作为 赋值表达式*);
                    
                默认:
                    返回 创建空IR值();
            }
        }
        
        函数 生成字面量(字面量表达式* 表达式) -> IR值 {
            选择 表达式.字面量类型 {
                情况 字面量类型.整数:
                    返回 创建整数常量(表达式.值);
                    
                情况 字面量类型.小数:
                    返回 创建小数常量(表达式.值);
                    
                情况 字面量类型.字符串:
                    返回 创建字符串常量(表达式.值);
                    
                情况 字面量类型.布尔:
                    返回 创建整数常量(表达式.值 == "真" ? 1 : 0);
                    
                情况 字面量类型.空:
                    返回 创建整数常量(0);
                    
                默认:
                    返回 创建空IR值();
            }
        }
        
        函数 生成二元表达式(二元表达式* 表达式) -> IR值 {
            变量 左 = 生成表达式(表达式.左操作数);
            变量 右 = 生成表达式(表达式.右操作数);
            变量 结果 = 分配虚拟寄存器();
            
            选择 表达式.运算符 {
                情况 二元运算符.加:
                    发射(IR操作码.加, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.减:
                    发射(IR操作码.减, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.乘:
                    发射(IR操作码.乘, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.除:
                    发射(IR操作码.除, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.取模:
                    发射(IR操作码.取模, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.等于:
                    发射(IR操作码.相等, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.不等于:
                    发射(IR操作码.不等, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.小于:
                    发射(IR操作码.小于, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.小于等于:
                    发射(IR操作码.小于等于, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.大于:
                    发射(IR操作码.大于, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.大于等于:
                    发射(IR操作码.大于等于, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.逻辑与:
                    发射(IR操作码.逻辑与, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.逻辑或:
                    发射(IR操作码.逻辑或, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.按位与:
                    发射(IR操作码.按位与, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.按位或:
                    发射(IR操作码.按位或, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.按位异或:
                    发射(IR操作码.按位异或, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.左移:
                    发射(IR操作码.左移, 结果, 左, 右);
                    中断;
                    
                情况 二元运算符.右移:
                    发射(IR操作码.右移, 结果, 左, 右);
                    中断;
            }
            
            返回 结果;
        }
        
        函数 生成函数调用(函数调用表达式* 表达式) -> IR值 {
            // 生成参数
            变量 参数值列表 = 创建IR值列表();
            循环 (整数 i = 0; i < 表达式.参数.长度; i = i + 1) {
                变量 参数值 = 生成表达式(表达式.参数.获取(i));
                参数值列表.添加(参数值);
            }
            
            // 获取函数引用
            变量 函数引用 = 生成表达式(表达式.被调函数);
            
            // 发射调用指令
            变量 结果 = 分配虚拟寄存器();
            发射调用(结果, 函数引用, 参数值列表);
            
            返回 结果;
        }
        
        函数 生成如果语句(如果语句* 语句) -> 空类型 {
            变量 条件 = 生成表达式(语句.条件);
            
            // 创建基本块
            变量 真块 = 创建基本块(生成标签("如果真"));
            变量 假块 = 创建基本块(生成标签("如果假"));
            变量 结束块 = 创建基本块(生成标签("如果结束"));
            
            // 条件跳转
            发射条件跳转(条件, 真块, 假块);
            
            // 真分支
            当前函数.基本块.添加(真块);
            当前基本块 = 真块;
            生成块语句(语句.真分支);
            发射跳转(结束块);
            
            // 假分支
            当前函数.基本块.添加(假块);
            当前基本块 = 假块;
            如果 (语句.假分支 != 无) {
                如果 (语句.假分支.类型 == 节点类型.块语句) {
                    生成块语句(语句.假分支 作为 块语句*);
                } 否则 {
                    生成语句(语句.假分支);  // 否则 如果
                }
            }
            发射跳转(结束块);
            
            // 结束块
            当前函数.基本块.添加(结束块);
            当前基本块 = 结束块;
        }
        
        函数 生成当语句(当语句* 语句) -> 空类型 {
            // 创建基本块
            变量 条件块 = 创建基本块(生成标签("当条件"));
            变量 循环体块 = 创建基本块(生成标签("当循环体"));
            变量 结束块 = 创建基本块(生成标签("当结束"));
            
            // 跳转到条件块
            发射跳转(条件块);
            
            // 条件块
            当前函数.基本块.添加(条件块);
            当前基本块 = 条件块;
            变量 条件 = 生成表达式(语句.条件);
            发射条件跳转(条件, 循环体块, 结束块);
            
            // 循环体块
            当前函数.基本块.添加(循环体块);
            当前基本块 = 循环体块;
            生成块语句(语句.循环体);
            发射跳转(条件块);
            
            // 结束块
            当前函数.基本块.添加(结束块);
            当前基本块 = 结束块;
        }
        
        函数 生成循环语句(循环语句* 语句) -> 空类型 {
            // 创建基本块
            变量 初始化块 = 创建基本块(生成标签("循环初始化"));
            变量 条件块 = 创建基本块(生成标签("循环条件"));
            变量 循环体块 = 创建基本块(生成标签("循环体"));
            变量 更新块 = 创建基本块(生成标签("循环更新"));
            变量 结束块 = 创建基本块(生成标签("循环结束"));
            
            // 初始化
            如果 (语句.初始化 != 无) {
                当前函数.基本块.添加(初始化块);
                当前基本块 = 初始化块;
                生成变量声明(语句.初始化);
            }
            发射跳转(条件块);
            
            // 条件块
            当前函数.基本块.添加(条件块);
            当前基本块 = 条件块;
            如果 (语句.条件 != 无) {
                变量 条件 = 生成表达式(语句.条件);
                发射条件跳转(条件, 循环体块, 结束块);
            } 否则 {
                发射跳转(循环体块);
            }
            
            // 循环体块
            当前函数.基本块.添加(循环体块);
            当前基本块 = 循环体块;
            生成块语句(语句.循环体);
            发射跳转(更新块);
            
            // 更新块
            当前函数.基本块.添加(更新块);
            当前基本块 = 更新块;
            如果 (语句.更新 != 无) {
                生成表达式(语句.更新);
            }
            发射跳转(条件块);
            
            // 结束块
            当前函数.基本块.添加(结束块);
            当前基本块 = 结束块;
        }
        
        函数 生成返回语句(返回语句* 语句) -> 空类型 {
            变量 返回值 = 无;
            如果 (语句.返回值 != 无) {
                返回值 = 生成表达式(语句.返回值);
            }
            发射返回(返回值);
        }
        
        // 辅助函数
        函数 分配虚拟寄存器() -> IR值 {

        函数 生成标签(字符串 前缀) -> 字符串 {
```

### 3.3 优化Pass设计

```cn
// src_cn/IR生成器/优化Pass.cn
模块 IR生成器 {
public:
    // 优化Pass基类
    接口 优化Pass {
        函数 执行(IR模块* 模块) -> 布尔;
        函数 获取名称() -> 字符串;
    }
    
    // 常量折叠Pass
    类 常量折叠Pass : 优化Pass {
    公开:
        函数 执行(IR模块* 模块) -> 布尔 {
            变量 有变化 = 假;
            
            循环 (整数 i = 0; i < 模块.函数.长度; i = i + 1) {
                变量 函数 = 模块.函数.获取(i);
                有变化 = 有变化 || 折叠函数(函数);
            }
            
            返回 有变化;
        }
        
        函数 获取名称() -> 字符串 {
            返回 "常量折叠";
        }
        
    私有:
        函数 折叠函数(IR函数* 函数) -> 布尔 {
            变量 有变化 = 假;
            
            循环 (整数 i = 0; i < 函数.基本块.长度; i = i + 1) {
                变量 块 = 函数.基本块.获取(i);
                变量 指令 = 块.第一条指令;
                
                当 (指令 != 无) {
                    如果 (可以折叠(指令)) {
                        折叠指令(指令);
                        有变化 = 真;
                    }
                    指令 = 指令.下一条;
                }
            }
            
            返回 有变化;
        }
        
        函数 可以折叠(IR指令* 指令) -> 布尔 {
            // 检查是否是算术运算且操作数都是常量
            如果 (!是算术运算(指令.操作码)) {
                返回 假;
            }
            
            返回 是常量(指令.源操作数1) && 是常量(指令.源操作数2);
        }
        
        函数 折叠指令(IR指令* 指令) -> 空类型 {
            变量 左 = 指令.源操作数1.整数值;
            变量 右 = 指令.源操作数2.整数值;
            变量 结果 = 0;
            
            选择 指令.操作码 {
                情况 IR操作码.加:
                    结果 = 左 + 右;
                    中断;
                情况 IR操作码.减:
                    结果 = 左 - 右;
                    中断;
                情况 IR操作码.乘:
                    结果 = 左 * 右;
                    中断;
                情况 IR操作码.除:
                    结果 = 左 / 右;
                    中断;
                情况 IR操作码.取模:
                    结果 = 左 % 右;
                    中断;
                默认:
                    返回;
            }
            
            // 将指令替换为常量
            指令.操作码 = IR操作码.空操作;
            指令.目标.类型 = IR值类型.常量整数;
            指令.目标.整数值 = 结果;
        }
    }
    
    // 死代码消除Pass
    类 死代码消除Pass : 优化Pass {
    公开:
        函数 执行(IR模块* 模块) -> 布尔 {
            变量 有变化 = 假;
            
            循环 (整数 i = 0; i < 模块.函数.长度; i = i + 1) {
                变量 函数 = 模块.函数.获取(i);
                有变化 = 有变化 || 消除死代码(函数);
            }
            
            返回 有变化;
        }
        
        函数 获取名称() -> 字符串 {
            返回 "死代码消除";
        }
        
    私有:
        函数 消除死代码(IR函数* 函数) -> 布尔 {
            变量 有变化 = 假;
            变量 使用计数 = 创建使用计数表(函数);
            
            // 计算每个虚拟寄存器的使用次数
            计算使用次数(函数, 使用计数);
            
            // 移除未使用的指令
            循环 (整数 i = 0; i < 函数.基本块.长度; i = i + 1) {
                变量 块 = 函数.基本块.获取(i);
                变量 指令 = 块.第一条指令;
                
                当 (指令 != 无) {
                    如果 (可以移除(指令, 使用计数)) {
                        移除指令(块, 指令);
                        有变化 = 真;
                    }
                    指令 = 指令.下一条;
                }
            }
            
            返回 有变化;
        }
        
        函数 可以移除(IR指令* 指令, 使用计数表 表) -> 布尔 {
            // 有副作用的指令不能移除
            如果 (有副作用(指令)) {
                返回 假;
            }
            
            // 结果未被使用的指令可以移除
            如果 (指令.目标.类型 == IR值类型.虚拟寄存器) {
                返回 表.获取(指令.目标.编号) == 0;
            }
            
            返回 假;
        }
    }
    
    // Pass管理器
    类 Pass管理器 {
    私有:
        优化Pass列表 优化Pass序列;
        
    公开:
        函数 添加Pass(优化Pass* Pass) -> 空类型 {
            优化Pass序列.添加(Pass);
        }
        
        函数 运行(IR模块* 模块) -> 空类型 {
            变量 迭代次数 = 0;
            变量 最大迭代 = 10;
            
            当 (迭代次数 < 最大迭代) {
                变量 有变化 = 假;
                
                循环 (整数 i = 0; i < 优化Pass序列.长度; i = i + 1) {
                    变量 Pass = 优化Pass序列.获取(i);
                    如果 (Pass.执行(模块)) {
                        有变化 = 真;
                    }
                }
                
                如果 (!有变化) {
                    中断;
                }
                
                迭代次数 = 迭代次数 + 1;
            }
        }
    }
}
```
