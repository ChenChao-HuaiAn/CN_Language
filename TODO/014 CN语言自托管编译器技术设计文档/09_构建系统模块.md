## 3. 详细任务描述

### 3.1 根CMake配置

```cmake
# CMakeLists.txt - 根目录配置
cmake_minimum_required(VERSION 3.16)
project(CN_Language VERSION 1.0.0 LANGUAGES C)

# 构建选项
option(BUILD_BOOTSTRAP "构建引导编译器" ON)
option(BUILD_SELF_HOSTED "构建自托管编译器" OFF)
option(BUILD_TESTS "构建测试" ON)
option(ENABLE_LTO "启用链接时优化" OFF)

# 设置C标准
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# 编译选项
if(MSVC)
    add_compile_options(/W4 /WX)
else()
    add_compile_options(-Wall -Wextra -Werror -pedantic)
endif()

# 第一阶段：构建引导编译器
if(BUILD_BOOTSTRAP)
    message(STATUS "=== 第一阶段: 构建引导编译器 ===")
    add_subdirectory(src)
    
    # 安装引导编译器
    install(TARGETS cnc RUNTIME DESTINATION bin)
    
    # 安装运行时库
    install(DIRECTORY include/cnlang/ DESTINATION include/cnlang)
endif()

# 第二阶段：构建自托管编译器
if(BUILD_SELF_HOSTED)
    message(STATUS "=== 第二阶段: 构建自托管编译器 ===")
    
    # 查找引导编译器
    find_program(CNC_BOOTSTRAP 
        NAMES cnc cnc.exe
        PATHS ${CMAKE_SOURCE_DIR}/build/bin
        ${CMAKE_INSTALL_PREFIX}/bin
        REQUIRED
    )
    
    message(STATUS "引导编译器: ${CNC_BOOTSTRAP}")
    
    # 定义CN源码目录
    set(CN_SOURCE_DIR ${CMAKE_SOURCE_DIR}/src_cn)
    
    # 收集所有CN源文件
    file(GLOB_RECURSE CN_SOURCES "${CN_SOURCE_DIR}/*.cn")
    
    # 自定义命令：使用引导编译器编译CN源码
    add_custom_command(
        OUTPUT ${CMAKE_BINARY_DIR}/cn.c
        COMMAND ${CNC_BOOTSTRAP}
                ${CN_SOURCE_DIR}
                -o ${CMAKE_BINARY_DIR}/cn.c
                --emit-c
        DEPENDS ${CN_SOURCES}
        COMMENT "使用引导编译器编译CN源码"
    )
    
    # 编译生成的C代码
    add_executable(cn ${CMAKE_BINARY_DIR}/cn.c)
    target_link_libraries(cn PRIVATE cn_runtime)
    
    # 自举验证目标
    add_custom_target(bootstrap_verify
        COMMAND ${CMAKE_COMMAND} -E echo "=== 自举验证 ==="
        COMMAND ${CNC_BOOTSTRAP} ${CN_SOURCE_DIR} -o ${CMAKE_BINARY_DIR}/v1.c
        COMMAND ${CMAKE_BINARY_DIR}/cn ${CN_SOURCE_DIR} -o ${CMAKE_BINARY_DIR}/v2.c
        COMMAND ${CMAKE_COMMAND} -E compare_files ${CMAKE_BINARY_DIR}/v1.c ${CMAKE_BINARY_DIR}/v2.c
        COMMENT "执行自举验证"
    )
endif()

# 测试
if(BUILD_TESTS)
    enable_testing()
    add_subdirectory(tests)
endif()
```

# src/CMakeLists.txt - 源码配置

# 运行时库
add_library(cn_runtime STATIC
    runtime/core/runtime.c
    runtime/core/stdlib.c
    runtime/memory/memory.c
    runtime/io/io.c
    runtime/collections/collections.c
    runtime/cli/cli.c

### 3.4 构建脚本

```bash
#!/bin/bash
# build.sh - 构建脚本

set -e

BUILD_TYPE=${1:-Release}
BUILD_DIR="build"

echo "=== CN语言编译器构建 ==="
echo "构建类型: $BUILD_TYPE"
echo "构建目录: $BUILD_DIR"

# 创建构建目录
mkdir -p $BUILD_DIR
cd $BUILD_DIR

# 第一阶段：构建引导编译器
echo ""
echo "=== 第一阶段: 构建引导编译器 ==="
cmake .. \
    -DCMAKE_BUILD_TYPE=$BUILD_TYPE \
    -DBUILD_BOOTSTRAP=ON \
    -DBUILD_SELF_HOSTED=OFF \
    -DBUILD_TESTS=ON

cmake --build . --config $BUILD_TYPE

# 运行测试
echo ""
echo "=== 运行测试 ==="
ctest --output-on-failure

# 安装
echo ""
echo "=== 安装引导编译器 ==="
cmake --install . --prefix ./install

echo ""
echo "✅ 第一阶段构建完成"
echo "引导编译器: $BUILD_DIR/src/cnc.exe"
```

### 3.5 自举构建脚本

```bash
#!/bin/bash
# bootstrap.sh - 自举构建脚本

set -e

BUILD_DIR="build"
SRC_CN="src_cn"

echo "=== CN语言自托管编译器构建 ==="

# 检查引导编译器
if [ ! -f "$BUILD_DIR/src/cnc.exe" ]; then
    echo "错误: 未找到引导编译器，请先运行 build.sh"
    exit 1
fi

# 第二阶段：构建自托管编译器
echo ""
echo "=== 第二阶段: 构建自托管编译器 ==="
cd $BUILD_DIR

cmake .. \
    -DCMAKE_BUILD_TYPE=Release \
    -DBUILD_BOOTSTRAP=OFF \
    -DBUILD_SELF_HOSTED=ON

cmake --build . --config Release

# 自举验证
echo ""
echo "=== 自举验证 ==="
cmake --build . --target bootstrap_verify

echo ""
echo "✅ 自托管编译器构建完成"
echo "CN编译器: $BUILD_DIR/cn.exe"
```
