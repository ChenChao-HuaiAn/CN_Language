## 3. 详细任务描述

### 3.1 根目录CMake配置

```cmake
# CMakeLists.txt - 根目录配置
cmake_minimum_required(VERSION 3.16)
project(CN_Language VERSION 1.0.0 LANGUAGES C)

# 构建选项
option(BUILD_BOOTSTRAP "构建引导编译器（第一阶段）" ON)
option(BUILD_SELF_HOSTED "构建自托管编译器（第二阶段）" OFF)
option(BUILD_TESTS "构建测试" ON)
option(ENABLE_LTO "启用链接时优化" OFF)

# C标准
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# 编译选项
if(MSVC)
    add_compile_options(/W4 /utf-8)
else()
    add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# 输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

# ========================================
# 第一阶段：构建引导编译器
# ========================================
if(BUILD_BOOTSTRAP)
    message(STATUS "=== 第一阶段: 构建引导编译器 ===")
    
    # 添加源码目录
    add_subdirectory(src)
    
    # 添加运行时库
    add_subdirectory(src/runtime)
    
    # 安装目标
    install(TARGETS cnc RUNTIME DESTINATION bin)
    install(DIRECTORY include/cnlang DESTINATION include)
    
    # 测试
    if(BUILD_TESTS)
        enable_testing()
        add_subdirectory(tests)
    endif()
endif()

# ========================================
# 第二阶段：构建自托管编译器
# ========================================
if(BUILD_SELF_HOSTED)
    message(STATUS "=== 第二阶段: 构建自托管编译器 ===")
    
    # 查找引导编译器
    find_program(CNC_BOOTSTRAP 
        NAMES cnc cnc.exe
        PATHS 
            ${CMAKE_BINARY_DIR}/bin
            ${CMAKE_SOURCE_DIR}/build/bin
        REQUIRED
    )
    
    message(STATUS "引导编译器: ${CNC_BOOTSTRAP}")
    
    # CN源码目录
    set(CN_SOURCE_DIR ${CMAKE_SOURCE_DIR}/src_cn)
    
    # 收集CN源文件
    file(GLOB_RECURSE CN_SOURCES "${CN_SOURCE_DIR}/*.cn")
    
    if(NOT CN_SOURCES)
        message(FATAL_ERROR "未找到CN源文件: ${CN_SOURCE_DIR}")
    endif()
    
    message(STATUS "找到 ${CN_SOURCES} 个CN源文件")
    
    # 生成的C代码目录
    set(GENERATED_DIR ${CMAKE_BINARY_DIR}/generated)
    file(MAKE_DIRECTORY ${GENERATED_DIR})
    
    # 使用引导编译器编译CN源码
    add_custom_command(
        OUTPUT ${GENERATED_DIR}/cn_compiler.c
        COMMAND ${CNC_BOOTSTRAP}
                ${CN_SOURCE_DIR}
                -o ${GENERATED_DIR}/cn_compiler.c
                --emit-c
                --no-link
        DEPENDS ${CN_SOURCES}
        COMMENT "使用引导编译器编译CN源码"
        VERBATIM
    )
    
    # 编译生成的C代码
    add_executable(cn ${GENERATED_DIR}/cn_compiler.c)
    
    target_include_directories(cn PRIVATE
        ${CMAKE_SOURCE_DIR}/include
    )
    
    target_link_libraries(cn PRIVATE cn_runtime)
    
    # 链接时优化
    if(ENABLE_LTO)
        set_target_properties(cn PROPERTIES
            INTERPROCEDURAL_OPTIMIZATION TRUE
        )
    endif()
    
    # 安装
    install(TARGETS cn RUNTIME DESTINATION bin)
endif()

# ========================================
# 自举验证目标
# ========================================
if(BUILD_SELF_HOSTED)
    add_custom_target(bootstrap_verify
        COMMAND ${CMAKE_COMMAND} -E echo "=== 自举验证 ==="
        COMMAND ${CMAKE_COMMAND} -E echo "[1/3] 使用C编译器编译..."
        COMMAND ${CNC_BOOTSTRAP} ${CN_SOURCE_DIR} -o ${GENERATED_DIR}/v1.c --emit-c
        COMMAND ${CMAKE_COMMAND} -E echo "[2/3] 使用CN编译器v1编译..."
        COMMAND ${CMAKE_BINARY_DIR}/bin/cn ${CN_SOURCE_DIR} -o ${GENERATED_DIR}/v2.c --emit-c
        COMMAND ${CMAKE_COMMAND} -E echo "[3/3] 使用CN编译器v2编译..."
        COMMAND ${CMAKE_BINARY_DIR}/bin/cn ${CN_SOURCE_DIR} -o ${GENERATED_DIR}/v3.c --emit-c
        COMMAND ${CMAKE_COMMAND} -E echo "[验证] 比较编译结果..."
        COMMAND ${CMAKE_COMMAND} -E compare_files ${GENERATED_DIR}/v2.c ${GENERATED_DIR}/v3.c
        COMMAND ${CMAKE_COMMAND} -E echo "✅ 自举验证通过！"
        DEPENDS cn
        COMMENT "执行自举验证"
    )
endif()

# ========================================
# 打印配置摘要
# ========================================
message(STATUS "")
message(STATUS "=== 配置摘要 ===")
message(STATUS "构建引导编译器: ${BUILD_BOOTSTRAP}")
message(STATUS "构建自托管编译器: ${BUILD_SELF_HOSTED}")
message(STATUS "构建测试: ${BUILD_TESTS}")
message(STATUS "链接时优化: ${ENABLE_LTO}")
message(STATUS "")
```

### 3.4 构建脚本

```bash
#!/bin/bash
# build.sh - 构建脚本

set -e

echo "=== CN语言自托管编译器构建 ==="

# 清理
if [ "$1" == "clean" ]; then
    echo "[清理] 删除构建目录..."
    rm -rf build
    exit 0
fi

# 创建构建目录
mkdir -p build
cd build

# ========================================
# 第一阶段：构建引导编译器
# ========================================
echo ""
echo "=== 第一阶段: 构建引导编译器 ==="

cmake .. \
    -DBUILD_BOOTSTRAP=ON \
    -DBUILD_SELF_HOSTED=OFF \
    -DBUILD_TESTS=ON

cmake --build . --config Release

echo "✅ 引导编译器构建完成: build/bin/cnc.exe"

# ========================================
# 第二阶段：构建自托管编译器
# ========================================
if [ "$1" == "self-hosted" ] || [ "$1" == "all" ]; then
    echo ""
    echo "=== 第二阶段: 构建自托管编译器 ==="
    
    cmake .. \
        -DBUILD_BOOTSTRAP=OFF \
        -DBUILD_SELF_HOSTED=ON
    
    cmake --build . --config Release
    
    echo "✅ 自托管编译器构建完成: build/bin/cn.exe"
fi

# ========================================
# 自举验证
# ========================================
if [ "$1" == "verify" ] || [ "$1" == "all" ]; then
    echo ""
    echo "=== 自举验证 ==="
    
    cmake --build . --target bootstrap_verify
    
    echo "✅ 自举验证通过！"
fi

# ========================================
# 运行测试
# ========================================
if [ "$1" == "test" ] || [ "$1" == "all" ]; then
    echo ""
    echo "=== 运行测试 ==="
    
    ctest --output-on-failure
    
    echo "✅ 测试完成"
fi

echo ""
echo "=== 构建完成 ==="
echo "引导编译器: build/bin/cnc.exe"
if [ -f "bin/cn.exe" ]; then
    echo "自托管编译器: build/bin/cn.exe"
fi
```

### 3.5 Windows构建脚本

```powershell
# build.ps1 - Windows构建脚本

param(
    [string]$Target = "bootstrap"
)

Write-Host "=== CN语言自托管编译器构建 ===" -ForegroundColor Cyan

# 清理
if ($Target -eq "clean") {
    Write-Host "[清理] 删除构建目录..." -ForegroundColor Yellow
    Remove-Item -Recurse -Force build -ErrorAction SilentlyContinue
    exit 0
}

# 创建构建目录
New-Item -ItemType Directory -Force -Path build | Out-Null
Set-Location build

# ========================================
# 第一阶段：构建引导编译器
# ========================================
Write-Host ""
Write-Host "=== 第一阶段: 构建引导编译器 ===" -ForegroundColor Cyan

cmake .. `
    -DBUILD_BOOTSTRAP=ON `
    -DBUILD_SELF_HOSTED=OFF `
    -DBUILD_TESTS=ON `
    -G "Visual Studio 17 2022" `
    -A x64

cmake --build . --config Release

Write-Host "✅ 引导编译器构建完成: build\bin\Release\cnc.exe" -ForegroundColor Green

# ========================================
# 第二阶段：构建自托管编译器
# ========================================
if ($Target -eq "self-hosted" -or $Target -eq "all") {
    Write-Host ""
    Write-Host "=== 第二阶段: 构建自托管编译器 ===" -ForegroundColor Cyan
    
    cmake .. `
        -DBUILD_BOOTSTRAP=OFF `
        -DBUILD_SELF_HOSTED=ON
    
    cmake --build . --config Release
    
    Write-Host "✅ 自托管编译器构建完成: build\bin\Release\cn.exe" -ForegroundColor Green
}

# ========================================
# 自举验证
# ========================================
if ($Target -eq "verify" -or $Target -eq "all") {
    Write-Host ""
    Write-Host "=== 自举验证 ===" -ForegroundColor Cyan
    
    cmake --build . --target bootstrap_verify --config Release
    
    Write-Host "✅ 自举验证通过！" -ForegroundColor Green
}

# ========================================
# 运行测试
# ========================================
if ($Target -eq "test" -or $Target -eq "all") {
    Write-Host ""
    Write-Host "=== 运行测试 ===" -ForegroundColor Cyan
    
    ctest -C Release --output-on-failure
    
    Write-Host "✅ 测试完成" -ForegroundColor Green
}

Write-Host ""
Write-Host "=== 构建完成 ===" -ForegroundColor Cyan
Write-Host "引导编译器: build\bin\Release\cnc.exe"
if (Test-Path "bin\Release\cn.exe") {
    Write-Host "自托管编译器: build\bin\Release\cn.exe"
}
```
