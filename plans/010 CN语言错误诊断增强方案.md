# CN语言错误诊断增强方案

**文档版本**: v1.0  
**创建日期**: 2026-03-28  
**作者**: AI架构师

## 一、概述

### 1.1 功能目标

本方案旨在增强CN语言编译器的错误诊断能力，提供更友好、更专业的用户体验。核心功能包括：

| 功能 | 描述 | 优先级 |
|------|------|--------|
| 中英文双语支持 | 错误消息同时提供中文和英文版本 | 高 ⭐⭐⭐ |
| 参数化消息 | 支持在消息中嵌入符号名、类型名等动态信息 | 高 ⭐⭐⭐ |
| 智能恢复 | 遇到错误后能够智能恢复，继续分析后续代码 | 中 ⭐⭐ |
| 修复建议 | 提供错误修复建议 | 中 ⭐⭐ |

### 1.2 价值分析

1. **国际化支持**：双语消息使编译器同时服务于中文用户和国际用户
2. **提升调试效率**：参数化消息提供更精确的错误定位信息
3. **减少编译轮次**：智能恢复允许一次编译发现多个错误
4. **降低学习曲线**：修复建议帮助新手快速解决问题

### 1.3 当前系统分析

**现有结构** ([`include/cnlang/support/diagnostics.h`](include/cnlang/support/diagnostics.h:57))：

```c
typedef struct CnDiagnostic {
    CnDiagSeverity severity;    // 严重级别
    CnDiagCode code;            // 错误码
    const char *filename;       // 文件名
    int line;                   // 行号
    int column;                 // 列号
    const char *message;        // 仅支持单一语言
} CnDiagnostic;
```

**现有问题**：
1. 消息为静态字符串，无法嵌入动态参数
2. 仅支持中文，无国际化支持
3. 无错误恢复机制
4. 无修复建议功能

#### 2.1.2 消息表索引机制

```c
// 获取消息模板
static const CnDiagMessageTemplate* cn_diag_get_template(CnDiagCode code) {
    if (code >= 0 && code < sizeof(g_diag_messages) / sizeof(g_diag_messages[0])) {
        return &g_diag_messages[code];
    }
    return &g_diag_messages[CN_DIAG_CODE_UNKNOWN];
```

### 2.2 参数化消息模板设计

#### 2.2.1 参数占位符格式

采用 `{N}` 格式的位置参数，支持最多 4 个参数：

| 占位符 | 说明 |
|--------|------|
| `{0}` | 第一个参数 |
| `{1}` | 第二个参数 |
| `{2}` | 第三个参数 |
| `{3}` | 第四个参数 |

#### 2.2.2 参数值结构

```c
// 参数值类型
typedef enum CnDiagArgType {
    CN_DIAG_ARG_TYPE_NONE = 0,      // 无参数
    CN_DIAG_ARG_TYPE_STRING = 1,    // 字符串参数
    CN_DIAG_ARG_TYPE_INT = 2,       // 整数参数
    CN_DIAG_ARG_TYPE_CHAR = 3       // 字符参数
} CnDiagArgType;

// 参数值联合体
typedef struct CnDiagArg {
    CnDiagArgType type;
    union {
        const char *str_val;        // 字符串值
        int int_val;                // 整数值
        char char_val;              // 字符值
    };
} CnDiagArg;

// 参数列表（最多4个参数）
typedef struct CnDiagArgs {
    CnDiagArg args[4];
    int count;
} CnDiagArgs;
```

#### 2.2.3 消息格式化函数

```c
// 格式化消息（输出到缓冲区）
// 返回实际写入的字符数（不含终止符）
int cn_diag_format_message(
    const CnDiagMessageTemplate *tmpl,
    CnDiagLanguage lang,
    const CnDiagArgs *args,
    char *buffer,
    size_t buffer_size
);
```

**格式化算法**：
1. 根据语言选择消息模板
2. 遍历模板字符串，遇到 `{N}` 时替换为对应参数值
3. 参数值根据类型转换为字符串
4. 输出到目标缓冲区

### 2.3 智能恢复策略设计

#### 2.3.1 错误恢复级别

```c
// 错误恢复策略
typedef enum CnDiagRecoveryStrategy {
    CN_DIAG_RECOVERY_NONE = 0,          // 不恢复，立即终止
    CN_DIAG_RECOVERY_SKIP_TOKEN = 1,    // 跳过当前token，继续
    CN_DIAG_RECOVERY_SKIP_TO_SYNC = 2,  // 跳到同步点（分号、右大括号等）
    CN_DIAG_RECOVERY_INSERT_TOKEN = 3,  // 插入缺失的token
    CN_DIAG_RECOVERY_DELETE_TOKEN = 4   // 删除多余token
} CnDiagRecoveryStrategy;
```

#### 2.3.2 同步点定义

```c
// 同步点token类型
static const CnTokenKind g_sync_tokens[] = {
    CN_TOKEN_SEMICOLON,         // ;
    CN_TOKEN_RBRACE,            // }
    CN_TOKEN_RPAREN,            // )
    CN_TOKEN_RBRACKET,          // ]
    CN_TOKEN_EOF                // 文件结束
};

// 检查是否为同步点
bool cn_diag_is_sync_point(CnTokenKind kind) {
    for (size_t i = 0; i < sizeof(g_sync_tokens)/sizeof(g_sync_tokens[0]); i++) {
        if (kind == g_sync_tokens[i]) {
            return true;
        }
    }
    return false;
}
```

#### 2.3.3 错误恢复上下文

```c
// 错误恢复上下文
typedef struct CnDiagRecoveryContext {
    int error_count;                    // 当前错误计数
    int max_errors;                     // 最大错误数限制（默认20）
    int consecutive_errors;             // 连续错误计数
    int max_consecutive_errors;         // 最大连续错误数（默认3）
    bool in_recovery;                   // 是否处于恢复模式
    CnTokenKind expected_token;         // 期望的token（用于插入）
} CnDiagRecoveryContext;

// 初始化恢复上下文
void cn_diag_recovery_init(CnDiagRecoveryContext *ctx) {
    ctx->error_count = 0;
    ctx->max_errors = 20;
    ctx->consecutive_errors = 0;
    ctx->max_consecutive_errors = 3;
    ctx->in_recovery = false;
    ctx->expected_token = CN_TOKEN_EOF;
}

// 检查是否应该继续编译
bool cn_diag_should_continue(CnDiagRecoveryContext *ctx) {
    if (ctx->error_count >= ctx->max_errors) {
        return false;  // 错误数超限
    }
    if (ctx->consecutive_errors >= ctx->max_consecutive_errors) {
        return false;  // 连续错误过多
    }
    return true;
}
```

#### 2.3.4 恢复策略选择

```c
// 根据错误码选择恢复策略
CnDiagRecoveryStrategy cn_diag_select_recovery_strategy(
    CnDiagCode code,
    CnTokenKind current_token
) {
    switch (code) {
        // 词法错误：跳过当前token
        case CN_DIAG_CODE_LEX_INVALID_CHAR:
        case CN_DIAG_CODE_LEX_INVALID_HEX:
        case CN_DIAG_CODE_LEX_INVALID_BINARY:
        case CN_DIAG_CODE_LEX_INVALID_OCTAL:
            return CN_DIAG_RECOVERY_SKIP_TOKEN;
        
        // 语法错误：跳到同步点
        case CN_DIAG_CODE_PARSE_EXPECTED_TOKEN:
        case CN_DIAG_CODE_PARSE_INVALID_FUNCTION_NAME:
        case CN_DIAG_CODE_PARSE_INVALID_PARAM:
        case CN_DIAG_CODE_PARSE_INVALID_VAR_DECL:
            return CN_DIAG_RECOVERY_SKIP_TO_SYNC;
        
        // 语义错误：继续分析（语义错误不影响解析）
        case CN_DIAG_CODE_SEM_DUPLICATE_SYMBOL:
        case CN_DIAG_CODE_SEM_UNDEFINED_IDENTIFIER:
        case CN_DIAG_CODE_SEM_TYPE_MISMATCH:
            return CN_DIAG_RECOVERY_NONE;
        
        default:
            return CN_DIAG_RECOVERY_SKIP_TOKEN;
    }
}
```

### 2.4 修复建议机制设计

#### 2.4.1 建议类型

```c
// 修复建议类型
typedef enum CnDiagFixKind {
    CN_DIAG_FIX_INSERT = 0,     // 插入文本
    CN_DIAG_FIX_DELETE = 1,     // 删除文本
    CN_DIAG_FIX_REPLACE = 2,    // 替换文本
    CN_DIAG_FIX_RENAME = 3      // 重命名
} CnDiagFixKind;

// 修复建议
typedef struct CnDiagFix {
    CnDiagFixKind kind;
    int start_line;             // 起始行
    int start_column;           // 起始列
    int end_line;               // 结束行
    int end_column;             // 结束列
    const char *new_text;       // 新文本（插入/替换用）
    const char *description;    // 修复描述
} CnDiagFix;
```

#### 2.4.2 建议生成函数

```c
// 为特定错误生成修复建议
int cn_diag_generate_fixes(
    CnDiagCode code,
    const char *filename,
    int line,
    int column,
    const CnDiagArgs *args,
    CnDiagFix *fixes,
    int max_fixes
);
```

**示例建议生成**：

```c
// 未定义标识符 -> 建议相似标识符
if (code == CN_DIAG_CODE_SEM_UNDEFINED_IDENTIFIER) {
    const char *name = args->args[0].str_val;
    // 查找相似标识符
    char *suggestions[3];
    int count = find_similar_symbols(name, suggestions, 3);
    for (int i = 0; i < count && i < max_fixes; i++) {
        fixes[i].kind = CN_DIAG_FIX_RENAME;
        fixes[i].start_line = line;
        fixes[i].start_column = column;
        fixes[i].end_line = line;
        fixes[i].end_column = column + strlen(name);
        fixes[i].new_text = suggestions[i];
        fixes[i].description = "使用相似标识符";
    }
    return count;
}
```

## 九、风险与缓解

| 风险 | 影响 | 缓解措施 |
|------|------|----------|
| 向后兼容性破坏 | 现有代码无法编译 | 提供兼容包装函数 |

| 错误恢复误判 | 隐藏真实错误 | 设置连续错误上限 |
