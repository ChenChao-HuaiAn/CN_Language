# CN语言接口模板参数支持方案

> **文档说明**：本文档设计CN语言接口的模板参数支持，用于替代"自身类型"关键字，实现类型安全的接口方法签名。
>
> **创建时间**：2026-04-02
> **相关文档**：[`plans/001 CN Language语法规范设计文档.md`](plans/001 CN Language语法规范设计文档.md)、[`plans/013 CN语言泛型编程支持方案.md`](plans/013 CN语言泛型编程支持方案.md)

---

## 一、背景与目标

### 1.1 问题背景

接口方法需要引用"实现此接口的类的类型"，例如：

```cn
接口 可比较 {
    函数 比较(自身类型 其他) -> 整数;  // "自身类型" 未在语法规范中定义
};
```

### 1.2 解决方案

使用已有的模板机制，为接口添加模板参数支持：

```cn
模板<T>
接口 可比较 {
    函数 比较(T 其他) -> 整数;  // T 是实现此接口的类的类型
};

类 数据项 : 实现 可比较<数据项> {
    函数 比较(数据项 其他) -> 整数 {
        // ...
    }
};
```

### 1.3 设计目标

| 目标 | 说明 |
|------|------|
| 类型安全 | 编译期检查接口实现类型一致性 |
| 复用现有机制 | 利用已有的模板支持，无需新增关键字 |
| 语法简洁 | 接口定义和使用语法清晰易懂 |
| 向后兼容 | 不影响现有的非泛型接口 |

## 二、语法设计

### 2.1 接口模板语法

**语法规则**：
```
接口声明 ::= ("模板" "<" 模板参数列表 ">" )? "接口" 标识符 (":" 接口列表)? "{" 接口成员* "}"
接口成员 ::= 方法签名
方法签名 ::= "函数" 标识符 "(" 参数列表? ")" ("->" 类型)? ";"
```

**示例**：
```cn
// 无模板参数的接口（现有语法）
接口 可序列化 {
    函数 序列化() -> 字符串;
    函数 反序列化(字符串 数据);
}

// 带模板参数的接口（新语法）
模板<T>
接口 可比较 {
    函数 比较(T 其他) -> 整数;
}

// 多模板参数的接口
模板<K, V>
接口 映射接口 {
    函数 获取(K 键) -> V;
    函数 设置(K 键, V 值);
}
```

### 2.2 类实现模板接口

**语法规则**：
```
类实现 ::= "类" 标识符 (":" 基类)? ("实现" 接口实例化列表)? "{" 类成员* "}"
接口实例化 ::= 接口名 "<" 类型列表 ">"
```

**示例**：
```cn
// 实现模板接口
类 数据项 : 实现 可比较<数据项> {
    公开:
    整数 编号;
    字符串 名称;
    
    函数 比较(数据项 其他) -> 整数 {
        如果 (自身.编号 < 其他.编号) {
            返回 -1;
        } 否则 如果 (自身.编号 > 其他.编号) {
            返回 1;
        }
        返回 0;
    }
}

// 实现多个模板接口
类 配置项 : 实现 可序列化, 可比较<配置项> {
    // ...
}
```

### 2.3 接口继承

**示例**：
```cn
// 模板接口继承普通接口
模板<T>
接口 有序可比较 : 可比较<T> {
    函数 前一个() -> T;
    函数 后一个() -> T;
}
```

```c
/**
 * @brief 解析接口声明
 * 
 * 语法：模板<T>? 接口 接口名 { 方法* }
 */
static CnAstStmt *parse_interface_decl(CnParser *parser) {
    // 1. 检查是否有模板参数
    CnAstTemplateParams *template_params = NULL;
    if (parser->current.kind == CN_TOKEN_KEYWORD_TEMPLATE) {
        template_params = parse_template_params(parser);
        if (!template_params) {
            return NULL;  // 模板参数解析失败
        }
    }
    
    // 2. 期望 '接口' 关键字
    if (!parser_expect(parser, CN_TOKEN_KEYWORD_INTERFACE)) {
        if (template_params) cn_ast_template_params_free(template_params);
        return NULL;
    }
    
    // 3. 解析接口名
    // ... 现有逻辑 ...
    
    // 4. 创建接口声明节点
    CnAstInterfaceDecl *decl = cn_ast_interface_decl_new();
    decl->template_params = template_params;  // 保存模板参数
    // ... 其他字段 ...
    
    return decl;
}
```

### 4.2 解析接口实例化

**文件**：[`src/frontend/parser/parser.c`](src/frontend/parser/parser.c)

**修改位置**：`parse_implements_list()` 函数

```c
/**
 * @brief 解析接口实例化列表
 * 
 * 语法：实现 接口名<类型1, 类型2>, ...
 */
static CnAstInterfaceInstantiation **parse_implements_list(
    CnParser *parser, 
    size_t *out_count
) {
    // 1. 期望 '实现' 关键字
    if (!parser_expect(parser, CN_TOKEN_KEYWORD_IMPLEMENTS)) {
        *out_count = 0;
        return NULL;
    }
    
    // 2. 解析接口列表
    CnAstInterfaceInstantiation **list = NULL;
    size_t count = 0;
    
    do {
        // 解析接口名
        const char *name = parser->current.value;
        size_t name_len = parser->current.length;
        parser_advance(parser);
        
        // 检查是否有模板实参
        CnType **type_args = NULL;
        size_t type_arg_count = 0;
        
        if (parser->current.kind == CN_TOKEN_LESS) {
            // 解析模板实参列表
            type_args = parse_type_arg_list(parser, &type_arg_count);
        }
        
        // 创建接口实例化节点
        CnAstInterfaceInstantiation *inst = cn_ast_interface_instantiation_new(
            name, name_len, type_args, type_arg_count
        );
        
        // 添加到列表
        // ...
        
    } while (parser_match(parser, CN_TOKEN_COMMA));
    
    *out_count = count;
    return list;
}
```

## 五、语义分析

### 5.1 接口模板参数绑定

**文件**：[`src/semantics/checker/semantic_passes.c`](src/semantics/checker/semantic_passes.c)

```c
/**
 * @brief 处理接口模板参数作用域
 */
void bind_interface_template_params(
    CnScope *scope,
    CnAstInterfaceDecl *interface_decl
) {
    if (!interface_decl->template_params) return;
    
    // 将模板参数绑定到作用域
    for (size_t i = 0; i < interface_decl->template_params->param_count; i++) {
        CnAstTemplateParam *param = &interface_decl->template_params->params[i];
        
        // 创建类型参数符号
        CnSymbol *sym = cn_symbol_new_type_param(
            param->name, 
            param->name_length,
            CN_SYM_TYPE_PARAMETER
        );
        
        // 添加到作用域
        cn_scope_add_symbol(scope, sym);
    }
}
```

### 5.2 接口实现类型检查

```c
/**
 * @brief 检查类是否正确实现了接口
 */
bool check_interface_implementation(
    CnAstClassDecl *class_decl,
    CnAstInterfaceInstantiation *interface_inst,
    CnDiagnostics *diagnostics
) {
    // 1. 查找接口定义
    CnAstInterfaceDecl *interface_def = find_interface(interface_inst->interface_name);
    if (!interface_def) {
        cn_diag_report_error(diagnostics, ...);
        return false;
    }
    
    // 2. 如果接口有模板参数，检查类型实参数量
    if (interface_def->template_params) {
        if (interface_inst->type_arg_count != interface_def->template_params->param_count) {
            cn_diag_report_error(diagnostics, ...);
            return false;
        }
    }
    
    // 3. 检查每个接口方法是否实现
    for (size_t i = 0; i < interface_def->method_count; i++) {
        CnAstMethodSignature *method = interface_def->methods[i];
        
        // 在类中查找对应方法
        CnClassMember *impl = find_class_method(class_decl, method->name);
        if (!impl) {
            cn_diag_report_error(diagnostics, ...);
            return false;
        }
        
        // 检查方法签名是否匹配（考虑模板参数替换）
        if (!method_signatures_match(method, impl, interface_inst->type_args)) {
            cn_diag_report_error(diagnostics, ...);
            return false;
        }
    }
    
    return true;
}
```

## 六、代码生成

### 6.1 接口vtable生成

**文件**：[`src/backend/cgen/class_cgen.c`](src/backend/cgen/class_cgen.c)

```c
/**
 * @brief 生成接口vtable定义
 */
void cn_cgen_interface_vtable(
    FILE *out,
    CnAstInterfaceDecl *interface_decl,
    CnType **type_args  // 模板实参（可为NULL）
) {
    // 生成vtable结构体名
    // 例如：可比较<数据项> -> 可比较_数据项_vtable
    char vtable_name[256];
    generate_interface_vtable_name(vtable_name, interface_decl, type_args);
    
    fprintf(out, "typedef struct %s {\n", vtable_name);
    
    // 生成方法指针
    for (size_t i = 0; i < interface_decl->method_count; i++) {
        CnAstMethodSignature *method = interface_decl->methods[i];
        
        // 替换模板参数为实际类型
        CnType *return_type = substitute_type_params(method->return_type, type_args);
        CnType **param_types = substitute_param_types(method->params, type_args);
        
        // 生成方法指针
        fprintf(out, "    %s (*%.*s)(void* self",
                get_c_type_string(return_type),
                (int)method->name_length, method->name);
        
        for (size_t j = 0; j < method->param_count; j++) {
            fprintf(out, ", %s", get_c_type_string(param_types[j]));
        }
        
        fprintf(out, ");\n");
    }
    
    fprintf(out, "} %s;\n\n", vtable_name);
}
```

### 6.2 类vtable初始化

```c
/**
 * @brief 生成类vtable初始化
 */
void cn_cgen_class_vtable_init(
    FILE *out,
    CnAstClassDecl *class_decl,
    CnAstInterfaceInstantiation *interface_inst
) {
    // 生成接口vtable成员初始化
    fprintf(out, ".%s_iface = {\n", interface_inst->interface_name);
    
    // 查找接口定义
    CnAstInterfaceDecl *interface_def = find_interface(interface_inst->interface_name);
    
    // 生成每个方法的绑定
    for (size_t i = 0; i < interface_def->method_count; i++) {
        CnAstMethodSignature *method = interface_def->methods[i];
        
        // 查找类中的实现方法
        CnClassMember *impl = find_class_method(class_decl, method->name);
        
        fprintf(out, "        .%.*s = %.*s_%.*s,\n",
                (int)method->name_length, method->name,
                (int)class_decl->name_length, class_decl->name,
                (int)method->name_length, method->name);
    }
    
    fprintf(out, "    },\n");
}
```

## 九、验收标准

### 9.1 语法验收

```cn
// 以下代码应能正确解析
模板<T>
接口 可比较 {
    函数 比较(T 其他) -> 整数;
}

类 数据项 : 实现 可比较<数据项> {
    函数 比较(数据项 其他) -> 整数 {
        返回 0;
    }
}
```

### 9.2 类型检查验收

```cn
// 以下代码应报类型错误
类 错误实现 : 实现 可比较<数据项> {
    函数 比较(字符串 其他) -> 整数 {  // 错误：参数类型不匹配
        返回 0;
    }
}
```

### 9.3 代码生成验收

生成的C代码应包含正确的vtable定义：

```c
// 接口vtable
typedef struct 可比较_数据项_vtable {
    long long (*比较)(void* self, struct 数据项 其他);
} 可比较_数据项_vtable;

// 类vtable
typedef struct 数据项_vtable {
    可比较_数据项_vtable 可比较_iface;
} 数据项_vtable;

// vtable初始化
static 数据项_vtable _数据项_vtable = {
    .可比较_iface = {
        .比较 = 数据项_比较,
    },
};
```

## 十、与"自身类型"关键字的对比

| 特性 | 模板接口 | 自身类型关键字 |
|------|---------|---------------|
| 语法规范 | 已有模板机制，无需新增关键字 | 需要新增关键字 |
| 类型安全 | 编译期检查 | 编译期检查 |
| 实现复杂度 | 复用现有模板机制 | 需要新增语义处理 |
| 使用语法 | `实现 可比较<数据项>` | `实现 可比较` |
| 灵活性 | 支持多模板参数 | 仅支持自身类型 |
