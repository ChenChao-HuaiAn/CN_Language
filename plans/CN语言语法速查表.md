# CN语言语法速查表

> 基于 [`docs/specifications/CN_Language 语言规范草案（核心子集）.md`](docs/specifications/CN_Language 语言规范草案（核心子集）.md) 和 [`plans/001 CN Language语法规范设计文档.md`](plans/001 CN Language语法规范设计文档.md) 整理

---

## 1. 外部函数声明

CN语言目前**没有专门的"外部函数声明"语法**。调用C运行时库函数有两种方式：

### 方式一：直接使用运行时API（推荐）

```cn
// 直接调用运行时库提供的中文接口
函数 主程序() -> 整数 {
    整数 参数数量 = 获取参数个数();
    字符串 参数值 = 获取参数(0);
    打印("参数: " + 参数值);
    返回 0;
}
```

### 方式二：通过运行时头文件声明

在C代码层面，运行时函数在 [`include/cnlang/runtime/`](include/cnlang/runtime/) 目录下声明，CN代码直接调用即可。

**注意**：CN语言不支持 `extern` 关键字声明外部函数，所有外部接口通过运行时库封装提供。

## 2. 类型定义

CN语言目前**不支持 `typedef` 或 `类型定义` 语法**。

### 替代方案：使用结构体或枚举

```cn
// 使用结构体定义复合类型
结构体 点 {
    整数 x;
    整数 y;
}

// 使用枚举定义状态类型
枚举 状态 {
    成功 = 0,
    失败 = 1,
    待定 = 2
}

// 使用时
函数 主程序() {
    点 位置 = {x: 10, y: 20};
    变量 当前状态 = 状态.成功;  // 枚举访问需要类型名前缀
    返回 0;
}
```

## 3. 函数声明和定义

### 基本语法

```cn
// 无返回值函数
函数 函数名(参数列表) {
    // 函数体
}

// 带返回类型函数（推荐）
函数 函数名(参数列表) -> 返回类型 {
    // 函数体
}
```

### 完整示例

```cn
// 无参数无返回值
函数 打印问候() {
    打印("你好！");
}

// 有参数有返回值
函数 加法(整数 a, 整数 b) -> 整数 {
    返回 a + b;
}

// 多种返回类型
函数 求面积(小数 半径) -> 小数 {
    返回 3.14159 * 半径 * 半径;
}

函数 是偶数(整数 n) -> 布尔 {
    返回 (n % 2) == 0;
}

函数 获取名称() -> 字符串 {
    返回 "CN语言";
}

// 空类型返回
函数 处理数据() -> 空类型 {
    打印("处理中...");
}
```

### 参数语法

```cn
// 基本类型参数
函数 示例(整数 a, 小数 b, 字符串 c, 布尔 d) { }

// 指针参数
函数 处理指针(整数* 数据, 小数* 结果) { }

// 数组参数（C风格）
函数 求和(整数 arr[], 整数 长度) -> 整数 {
    变量 总和 = 0;
    循环 (整数 i = 0; i < 长度; i++) {
        总和 = 总和 + arr[i];
    }
    返回 总和;
}

// 函数指针参数
函数 排序(整数* 数据, 整数(*比较)(整数, 整数) 比较器) { }
```

## 4. 指针类型

### 指针声明

```cn
// 单级指针
整数* 整数指针;
小数* 小数指针;
字符串* 字符串指针;

// 多级指针
整数** 二级指针;
整数*** 三级指针;
```

### 指针操作

```cn
函数 指针示例() {
    整数 值 = 42;
    整数* 指针 = &值;      // 取地址
    
    变量 解引用值 = *指针;  // 解引用
    
    打印(解引用值);        // 输出 42
}
```

### 指针作为返回值

```cn
函数 获取指针(整数* 数据) -> 整数* {
    返回 数据;
}
```

### 结构体指针访问

```cn
结构体 点 {
    整数 x;
    整数 y;
}

函数 访问成员() {
    点 位置 = {x: 10, y: 20};
    点* 位置指针 = &位置;
    
    // 值访问
    打印(位置.x);
    
    // 指针访问（箭头操作符）
    打印(位置指针->x);
}
```

## 5. 模块/包结构

### 内联模块定义

```cn
模块 数学工具 {
    // 默认私有
    整数 内部计数 = 0;
    
公开:
    整数 版本 = 1;
    
    函数 加(整数 a, 整数 b) -> 整数 {
        返回 a + b;
    }
    
私有:
    函数 内部辅助() -> 整数 {
        返回 0;
    }
    
公开:
    函数 乘(整数 a, 整数 b) -> 整数 {
        返回 a * b;
    }
}
```

### 文件模块

一个 `.cn` 文件自动成为一个模块，文件名即模块名：

```cn
// 文件: 数学工具.cn

公开:
    整数 圆周率近似 = 3;
    
    函数 加(整数 a, 整数 b) -> 整数 {
        返回 a + b;
    }

私有:
    函数 内部实现() { }
```

### 包结构

```
项目/
├── 主程序.cn              # 入口文件
├── 工具/                   # 「工具」包
│   ├── __包__.cn          # 包初始化文件
│   ├── 数学.cn            # 「工具.数学」模块
│   └── 字符串.cn          # 「工具.字符串」模块
└── 配置/                   # 「配置」包
    ├── __包__.cn
    └── 默认.cn
```

## 6. 导入语句

### 全量导入

```cn
导入 数学工具;

函数 主程序() {
    变量 结果 = 加(1, 2);  // 直接使用，无需前缀
    返回 0;
}
```

### 选择性导入

```cn
导入 数学工具 { 加, 乘 };

函数 主程序() {
    变量 a = 加(1, 2);
    变量 b = 乘(3, 4);
    返回 0;
}
```

### 模块别名

```cn
导入 数学工具(数学);

函数 主程序() {
    变量 结果 = 数学.加(1, 2);  // 使用别名访问
    返回 0;
}
```

### 从...导入

```cn
// 从模块导入指定成员
从 数学工具 导入 { 加, 乘 };

// 从模块导入所有公开成员
从 数学工具 导入 *;

// 从包内模块导入（斜杠路径）
从 工具/数学 导入 { 加 };

// 从包内模块导入（点分路径）
从 工具.数学 导入 { 加 };

// 相对导入
从 ./兄弟模块 导入 { 成员 };
从 ../父级模块 导入 { 成员 };
```

## 7. 变量声明

```cn
// 类型推断
变量 x = 10;              // 推断为整数
变量 文本 = "你好";       // 推断为字符串
变量 值 = 3.14;           // 推断为小数

// 显式类型
整数 a = 1;
小数 b = 2.5;
字符串 c = "文本";
布尔 d = 真;

// 常量
常量 PI = 3.14159;        // 类型推断
常量 整数 MAX = 100;      // 显式类型

// 数组
整数 arr[10];             // 固定大小数组
整数 matrix[3][4];        // 多维数组
变量 列表[] = {1, 2, 3};  // 初始化列表

// 静态局部变量
函数 计数器() -> 整数 {
    静态 整数 计数 = 0;   // 只初始化一次
    计数 = 计数 + 1;
    返回 计数;
}
```

## 8. 控制流语句

```cn
// 条件语句
如果 (x > 0) {
    打印("正数");
} 否则 如果 (x < 0) {
    打印("负数");
} 否则 {
    打印("零");
}

// 当循环
当 (i < 10) {
    i = i + 1;
}

// 循环
循环 (整数 i = 0; i < 10; i++) {
    打印(i);
}

// 选择语句
选择 (x) {
    情况 1:
        打印("一");
        中断;
    情况 2:
        打印("二");
        中断;
    默认:
        打印("其他");
        中断;
}
```

## 9. 结构体和枚举

### 结构体

```cn
结构体 点 {
    整数 x;
    整数 y;
}

// 初始化
点 p1 = {10, 20};           // 位置初始化
点 p2 = {x: 10, y: 20};     // 冒号初始化
点 p3 = {x = 10, y = 20};   // 等号初始化
```

### 枚举

```cn
枚举 颜色 {
    红,        // 默认值 0
    绿,        // 默认值 1
    蓝 = 10,   // 显式赋值
    黄         // 值为 11
}

// 访问（必须通过类型名）
变量 c = 颜色.红;
```

## 10. 函数指针

```cn
// 函数指针变量
整数(*回调)(整数, 整数);

// 函数指针作为结构体字段
结构体 事件处理器 {
    整数(*点击)(整数, 整数);
    整数(*按键)(整数);
};

// 函数指针作为函数参数
函数 遍历(整数* 数组, 整数(*)(整数) 处理函数) { }
函数 排序(整数* 数据, 整数(*比较)(整数, 整数) 比较器) { }
```

## 11. 入口函数

```cn
函数 主程序() -> 整数 {
    打印("你好，世界！");
    返回 0;
}
```

编译器将 `主程序` 映射为 C 的 `main` 函数。

## 12. 关键字速查

| 类别 | 关键字 |
|------|--------|
| **控制流** | `如果`、`否则`、`当`、`循环`、`返回`、`中断`、`继续`、`选择`、`情况`、`默认` |
| **类型** | `整数`、`小数`、`字符串`、`布尔`、`空类型`、`结构体`、`枚举` |
| **声明** | `函数`、`变量`、`模块`、`导入`、`从`、`公开`、`私有`、`静态` |
| **常量** | `真`、`假`、`无` |
| **异常** | `尝试`、`捕获`、`抛出`、`最终` |
| **OOP** | `类`、`接口`、`保护`、`虚拟`、`重写`、`抽象`、`实现`、`自身`、`基类` |
