# 符号表

## 概述

符号表用于存储程序中定义的符号（变量、函数、类型等）及其属性。符号表按照作用域组织，支持嵌套作用域和名称查找。

[`src/semantics/symbols/symbol_table.c`](../../src/semantics/symbols/symbol_table.c:1)

## 核心数据结构

### 符号 (CnSemSymbol)

```c
typedef struct CnSemSymbol {
    const char *name;              // 符号名称
    size_t name_length;            // 名称长度
    CnSemSymbolKind kind;          // 符号类型
    CnSemScope *decl_scope;        // 声明作用域
    CnType *type;                  // 类型信息
    int is_public;                 // 是否公开（模块成员）
    int is_const;                  // 是否为常量
    union {
        struct {
            CnSemScope *module_scope;  // 模块作用域
        } module;
        struct {
            CnAstFunctionDecl *decl;   // 函数声明
        } function;
        struct {
            size_t offset;             // 内存偏移
        } variable;
    } as;
} CnSemSymbol;
```

### 符号类型 (CnSemSymbolKind)

```c
typedef enum {
    CN_SEM_SYMBOL_VARIABLE,      // 变量
    CN_SEM_SYMBOL_FUNCTION,      // 函数
    CN_SEM_SYMBOL_TYPE,          // 类型
    CN_SEM_SYMBOL_MODULE,        // 模块
    CN_SEM_SYMBOL_STRUCT,        // 结构体
    CN_SEM_SYMBOL_ENUM,          // 枚举
    CN_SEM_SYMBOL_ENUM_MEMBER,   // 枚举成员
    CN_SEM_SYMBOL_STRUCT_FIELD,  // 结构体字段
} CnSemSymbolKind;
```

### 作用域 (CnSemScope)

```c
typedef struct CnSemScope {
    CnSemScopeKind kind;         // 作用域类型
    CnSemScope *parent;          // 父作用域
    CnSemSymbolNode *symbols;    // 符号链表
    const char *name;            // 作用域名称
    size_t name_length;          // 名称长度
    CnFileModuleSemInfo *file_module_info;  // 文件模块信息
} CnSemScope;
```

### 作用域类型 (CnSemScopeKind)

```c
typedef enum {
    CN_SEM_SCOPE_GLOBAL,         // 全局作用域
    CN_SEM_SCOPE_FILE_MODULE,    // 文件模块作用域
    CN_SEM_SCOPE_FUNCTION,       // 函数作用域
    CN_SEM_SCOPE_BLOCK,          // 代码块作用域
    CN_SEM_SCOPE_STRUCT,         // 结构体作用域
    CN_SEM_SCOPE_ENUM,           // 枚举作用域
} CnSemScopeKind;
```


**示例：**
```c
// 创建全局作用域
CnSemScope *global = cn_sem_scope_new(CN_SEM_SCOPE_GLOBAL, NULL);

// 创建函数作用域（父作用域为全局）
CnSemScope *func_scope = cn_sem_scope_new(CN_SEM_SCOPE_FUNCTION, global);

// 创建块作用域
CnSemScope *block_scope = cn_sem_scope_new(CN_SEM_SCOPE_BLOCK, func_scope);
```

## 符号操作

### 插入符号

```c
CnSemSymbol *cn_sem_scope_insert_symbol(CnSemScope *scope,
                                         const char *name,
                                         size_t name_length,
                                         CnSemSymbolKind kind);
```

**返回值：**
- 成功：符号指针
- 失败：NULL（符号已存在）

**示例：**
```c
// 插入变量符号
CnSemSymbol *var_sym = cn_sem_scope_insert_symbol(scope, "x", 1, CN_SEM_SYMBOL_VARIABLE);

// 插入函数符号
CnSemSymbol *func_sym = cn_sem_scope_insert_symbol(scope, "加法", 6, CN_SEM_SYMBOL_FUNCTION);
```

### 查找符号（当前作用域）

```c
CnSemSymbol *cn_sem_scope_lookup_shallow(CnSemScope *scope,
                                          const char *name,
                                          size_t name_length);
```

**说明：** 只在当前作用域查找，不搜索父作用域

### 查找符号（包含父作用域）

```c
CnSemSymbol *cn_sem_scope_lookup(CnSemScope *scope,
                                  const char *name,
                                  size_t name_length);
```

**说明：** 从当前作用域开始，逐级向上查找

**示例：**
```c
// 查找符号
CnSemSymbol *sym = cn_sem_scope_lookup(scope, "x", 1);

if (sym) {
    printf("找到符号：%.*s\n", (int)sym->name_length, sym->name);
    printf("符号类型：%d\n", sym->kind);
} else {
    printf("符号未定义\n");
}
```

## 作用域链

### 作用域链查找

```
块作用域 → 函数作用域 → 模块作用域 → 全局作用域
```

**示例：**
```cn
整数 全局变量 = 0;  // 全局作用域

函数 外层 () {
    整数 外层变量 = 1;  // 函数作用域
    
    如果 (真) {
        整数 内层变量 = 2;  // 块作用域
        
        // 可以访问：内层变量、外层变量、全局变量
        打印 (内层变量);
        打印 (外层变量);
        打印 (全局变量);
    }
    
    // 可以访问：外层变量、全局变量
    // 不能访问：内层变量（已超出作用域）
}
```

### 符号遮蔽

```cn
整数 x = 10;  // 全局 x

函数 测试 () {
    整数 x = 20;  // 遮蔽全局 x
    
    如果 (真) {
        整数 x = 30;  // 遮蔽外层 x
        打印 (x);  // 输出 30
    }
    
    打印 (x);  // 输出 20
}

打印 (x);  // 输出 10
```

### 可见性


**示例：**



```cn
// 常量变量
常量 整数 最大值 = 100;
```

## 符号表构建流程

### 1. 全局作用域构建

```c
// 伪代码
CnSemScope *global_scope = cn_sem_scope_new(CN_SEM_SCOPE_GLOBAL, NULL);

// 插入内置类型符号
insert_builtin_types(global_scope);

// 处理顶级声明
for (each top_level_decl in program) {
    if (decl is function) {
        CnSemSymbol *sym = cn_sem_scope_insert_symbol(global_scope, 
                                                       decl->name,
                                                       CN_SEM_SYMBOL_FUNCTION);
        sym->as.function.decl = decl;
    } else if (decl is struct) {
        CnSemSymbol *sym = cn_sem_scope_insert_symbol(global_scope,
                                                       decl->name,
                                                       CN_SEM_SYMBOL_TYPE);
        sym->type = cn_type_new_struct(...);
    }
}
```

### 2. 函数作用域构建

```c
// 为每个函数创建作用域
for (each function in program) {
    CnSemScope *func_scope = cn_sem_scope_new(CN_SEM_SCOPE_FUNCTION, global_scope);
    
    // 插入参数
    for (each param in function->params) {
        cn_sem_scope_insert_symbol(func_scope, param->name, CN_SEM_SYMBOL_VARIABLE);
    }
    
    // 处理函数体
    build_block_scope(func_scope, function->body);
}
```

### 3. 块作用域构建

```c
void build_block_scope(CnSemScope *parent, CnAstBlock *block) {
    CnSemScope *block_scope = cn_sem_scope_new(CN_SEM_SCOPE_BLOCK, parent);
    
    for (each stmt in block->stmts) {
        if (stmt is var_decl) {
            cn_sem_scope_insert_symbol(block_scope, stmt->name, CN_SEM_SYMBOL_VARIABLE);
        } else if (stmt is block) {
            build_block_scope(block_scope, stmt->as.block);
        }
    }
    
    cn_sem_scope_free(block_scope);
}
```

## 错误处理

### 重复定义

```c
CnSemSymbol *sym = cn_sem_scope_insert_symbol(scope, "x", 1, CN_SEM_SYMBOL_VARIABLE);
if (sym == NULL) {
    // 符号已存在，报告错误
    report_error("变量 'x' 重复定义");
}
```

### 未定义符号

```c
CnSemSymbol *sym = cn_sem_scope_lookup(scope, "undefined", 9);
if (sym == NULL) {
    // 符号未定义，报告错误
    report_error("未定义的标识符 'undefined'");
}
```

## 使用示例

### 完整的符号表示例

```c
#include "cnlang/frontend/semantics.h"
#include <stdio.h>

int main() {
    // 创建全局作用域
    CnSemScope *global = cn_sem_scope_new(CN_SEM_SCOPE_GLOBAL, NULL);
    
    // 插入全局变量
    CnSemSymbol *global_var = cn_sem_scope_insert_symbol(global, "全局变量", 8, CN_SEM_SYMBOL_VARIABLE);
    
    // 创建函数作用域
    CnSemScope *func_scope = cn_sem_scope_new(CN_SEM_SCOPE_FUNCTION, global);
    cn_sem_scope_set_name(func_scope, "主程序", 7);
    
    // 插入参数
    CnSemSymbol *param = cn_sem_scope_insert_symbol(func_scope, "参数", 4, CN_SEM_SYMBOL_VARIABLE);
    
    // 插入局部变量
    CnSemSymbol *local = cn_sem_scope_insert_symbol(func_scope, "局部变量", 8, CN_SEM_SYMBOL_VARIABLE);
    
    // 查找符号
    CnSemSymbol *found;
    
    // 在当前作用域查找
    found = cn_sem_scope_lookup_shallow(func_scope, "局部变量", 8);
    if (found) {
        printf("找到局部变量\n");
    }
    
    // 在父作用域查找
    found = cn_sem_scope_lookup(func_scope, "全局变量", 8);
    if (found) {
        printf("找到全局变量\n");
    }
    
    // 清理
    cn_sem_scope_free(func_scope);
    cn_sem_scope_free(global);
    
    return 0;
}
```

## 性能优化

1. **哈希表**：使用哈希表加速符号查找
2. **名字池**：使用字符串池减少内存占用
3. **惰性创建**：只在需要时创建嵌套作用域
