# IR API

## 概述

IR API 提供中间表示的创建、操作和优化功能。

## 模块操作

### CnIrModule 结构

```c
typedef struct CnIrModule {
    CnIrFunction *first_func;    // 函数链表头
    CnIrFunction *last_func;     // 函数链表尾
    CnIrGlobalVar *first_global; // 全局变量链表头
    CnIrGlobalVar *last_global;  // 全局变量链表尾
    CnTargetTriple target;       // 目标三元组
    CnCompileMode compile_mode;  // 编译模式
} CnIrModule;
```

### 函数

```c
// 创建模块
CnIrModule *cn_ir_module_new(void);

// 销毁模块
void cn_ir_module_free(CnIrModule *module);

// 添加函数
void cn_ir_module_add_function(CnIrModule *module, CnIrFunction *func);

// 添加全局变量
void cn_ir_module_add_global(CnIrModule *module, CnIrGlobalVar *global);

// 打印模块
void cn_ir_module_print(CnIrModule *module);
```

## 函数操作

### CnIrFunction 结构

```c
typedef struct CnIrFunction {
    char *name;                  // 函数名
    CnType *return_type;         // 返回类型
    CnIrOperand *params;         // 参数
    size_t param_count;          // 参数数量
    CnIrOperand *locals;         // 局部变量
    size_t local_count;          // 局部变量数量
    CnIrBasicBlock *first_block; // 基本块链表头
    CnIrBasicBlock *last_block;  // 基本块链表尾
    int next_reg_id;             // 下一个寄存器 ID
    int is_interrupt_handler;    // 是否中断处理函数
    int interrupt_vector;        // 中断向量
} CnIrFunction;
```

### 函数

```c
// 创建函数
CnIrFunction *cn_ir_function_new(const char *name, CnType *return_type);

// 添加参数
void cn_ir_function_add_param(CnIrFunction *func, CnIrOperand param);

// 添加局部变量
void cn_ir_function_add_local(CnIrFunction *func, CnIrOperand local);

// 添加基本块
void cn_ir_function_add_block(CnIrFunction *func, CnIrBasicBlock *block);

// 创建新寄存器
int cn_ir_function_new_reg(CnIrFunction *func);
```

## 基本块操作

### CnIrBasicBlock 结构

```c
typedef struct CnIrBasicBlock {
    char *name;                  // 块名
    CnIrInst *first_inst;        // 指令链表头
    CnIrInst *last_inst;         // 指令链表尾
    CnIrBasicBlockList *preds;   // 前驱
    CnIrBasicBlockList *succs;   // 后继
    CnIrBasicBlock *next;        // 下一个块
    CnIrBasicBlock *prev;        // 上一个块
} CnIrBasicBlock;
```

### 函数

```c
// 创建基本块
CnIrBasicBlock *cn_ir_basic_block_new(const char *name_hint);

// 添加指令
void cn_ir_basic_block_add_inst(CnIrBasicBlock *block, CnIrInst *inst);

// 连接基本块
void cn_ir_basic_block_connect(CnIrBasicBlock *from, CnIrBasicBlock *to);
```

## 指令操作

### CnIrInst 结构

```c
typedef struct CnIrInst {
    CnIrInstKind kind;           // 指令类型
    CnIrOperand dest;            // 目标
    CnIrOperand src1;            // 源 1
    CnIrOperand src2;            // 源 2
    CnIrOperand *extra_args;     // 额外参数
    size_t extra_args_count;     // 额外参数数量
    CnIrInst *next;              // 下一条指令
    CnIrInst *prev;              // 上一条指令
} CnIrInst;
```

### 指令类型

```c
typedef enum {
    // 算术

    CN_IR_INST_MUL,

    CN_IR_INST_NOT,

    CN_IR_INST_NE,
    CN_IR_INST_LT,
    CN_IR_INST_LE,
    CN_IR_INST_GT,
    CN_IR_INST_GE,
    
    // 控制流
    CN_IR_INST_JMP,
    CN_IR_INST_JZ,
    CN_IR_INST_JNZ,
    CN_IR_INST_RET,
```

## 操作数

### CnIrOperand 结构

```c
typedef struct CnIrOperand {
    enum {
        CN_IR_OP_NONE,
        CN_IR_OP_REG,
        CN_IR_OP_IMM_INT,
        CN_IR_OP_IMM_FLOAT,
        CN_IR_OP_IMM_STRING,
        CN_IR_OP_GLOBAL,
        CN_IR_OP_LOCAL,
        CN_IR_OP_PARAM,
        CN_IR_OP_ADDR,
    } kind;
    
    union {
        int reg_id;
        long long imm_int;
        double imm_float;
        const char *imm_string;
        const char *global_name;
        int local_index;
        int param_index;
        struct {
            CnIrOperand base;
            CnIrOperand offset;
        } addr;
    } as;
    
    CnType *type;
} CnIrOperand;
```

### 函数

```c
// 无操作数
CnIrOperand cn_ir_op_none(void);

// 寄存器操作数
CnIrOperand cn_ir_op_reg(int reg_id, CnType *type);

// 整数立即数
CnIrOperand cn_ir_op_imm_int(long long val, CnType *type);

// 浮点立即数
CnIrOperand cn_ir_op_imm_float(double val, CnType *type);

// 全局变量
CnIrOperand cn_ir_op_global(const char *name, CnType *type);

// 局部变量
CnIrOperand cn_ir_op_local(int index, CnType *type);

// 参数
CnIrOperand cn_ir_op_param(int index, CnType *type);
```
