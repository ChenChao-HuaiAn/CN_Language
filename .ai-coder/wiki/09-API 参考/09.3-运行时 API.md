# 运行时 API

## 概述

运行时 API 提供 CN_Language 程序运行所需的基础功能。

## 核心运行时

### 初始化

```c
// 运行时初始化
void cn_rt_init(void);

// 运行时退出
void cn_rt_exit(void);
```

### 打印函数

```c
// 打印整数
void cn_rt_print_int(long long val);

// 打印浮点数
void cn_rt_print_float(double val);

// 打印布尔
void cn_rt_print_bool(int val);

// 打印字符串
void cn_rt_print_string(const char *str);

// 打印换行
void cn_rt_print_newline(void);

// 格式化打印
void cn_rt_printf(const char *format, ...);
```

### 字符串操作

```c
// 字符串连接
char* cn_rt_string_concat(const char *a, const char *b);

// 字符串长度
size_t cn_rt_string_length(const char *str);

// 字符串复制
char* cn_rt_string_dup(const char *str);

// 整数转字符串
char* cn_rt_int_to_string(long long val);

// 浮点数转字符串
char* cn_rt_float_to_string(double val);
```

## 内存管理

### 基础分配

```c
// 分配
void* cn_rt_malloc(size_t size);

// 分配并清零
void* cn_rt_calloc(size_t count, size_t size);

// 重新分配
void* cn_rt_realloc(void* ptr, size_t new_size);

// 释放
void cn_rt_free(void* ptr);
```

### Arena 分配器

```c
// 创建 Arena
CnArena* cn_arena_create(size_t size);

// 从 Arena 分配
void* cn_arena_alloc(CnArena *arena, size_t size);

// 重置 Arena
void cn_arena_reset(CnArena *arena);

// 销毁 Arena
void cn_arena_destroy(CnArena *arena);

// 获取统计
void cn_arena_get_stats(CnArena *arena, CnArenaStats *stats);
```

### 内存操作

```c
// 内存复制
void* cn_rt_memcpy(void *dest, const void *src, size_t n);

// 内存设置
void* cn_rt_memset(void *s, int c, size_t n);

// 内存移动
void* cn_rt_memmove(void *dest, const void *src, size_t n);

// 内存比较
int cn_rt_memcmp(const void *s1, const void *s2, size_t n);
```

### 内存安全

```c
// 设置安全检查
void cn_rt_memory_set_safety_check(int enabled);

// 检查对齐
int cn_rt_memory_check_alignment(uintptr_t addr, size_t alignment);

// 检查可读
int cn_rt_memory_check_readable(uintptr_t addr, size_t size);

// 检查可写
int cn_rt_memory_check_writable(uintptr_t addr, size_t size);

// 安全读取
uintptr_t cn_rt_memory_read_safe(uintptr_t addr, size_t size);

// 安全写入
void cn_rt_memory_write_safe(uintptr_t addr, size_t size, uintptr_t value);
```

### 内存映射

```c
// 内存映射
void* cn_rt_mmap(void *addr, size_t length, int prot, int flags, int fd, off_t offset);

// 解除映射
int cn_rt_munmap(void *addr, size_t length);
```

## 输入输出

### 控制台 I/O

```c
// 读取整数
int cn_rt_scanf_int(int *value);

// 读取浮点数
int cn_rt_scanf_float(double *value);

// 读取字符串
int cn_rt_scanf_string(char *buffer, size_t size);

// 读取字符
int cn_rt_getchar(void);
```

### 文件 I/O

```c
// 打开文件
CnFile* cn_rt_file_open(const char *path, const char *mode);

// 关闭文件
void cn_rt_file_close(CnFile *file);

// 读取
size_t cn_rt_file_read(void *buffer, size_t size, size_t count, CnFile *file);

// 写入
size_t cn_rt_file_write(const void *buffer, size_t size, size_t count, CnFile *file);

// 读取行
char* cn_rt_file_read_line(CnFile *file);

// 定位
int cn_rt_file_seek(CnFile *file, long offset, int origin);

// 获取位置
long cn_rt_file_tell(CnFile *file);

// 倒回
void cn_rt_file_rewind(CnFile *file);

// 检查结束
int cn_rt_file_eof(CnFile *file);

// 检查错误
int cn_rt_file_error(CnFile *file);

// 清除错误
void cn_rt_file_clear_error(CnFile *file);
```

### 便捷函数

```c
// 读取整个文件
char* cn_rt_file_read_all(const char *path);

// 写入整个文件
int cn_rt_file_write_all(const char *path, const char *content);

// 检查文件存在
int cn_rt_file_exists(const char *path);
```

### 标准流

```c
CnFile* cn_rt_stdin(void);
CnFile* cn_rt_stdout(void);
CnFile* cn_rt_stderr(void);
```

### 随机数

## 并发支持

### 互斥锁

```c
// 创建
CnMutex* cn_mutex_create(void);

// 销毁
void cn_mutex_destroy(CnMutex *mutex);

// 加锁
void cn_mutex_lock(CnMutex *mutex);

// 尝试加锁
int cn_mutex_trylock(CnMutex *mutex);

// 解锁
void cn_mutex_unlock(CnMutex *mutex);
```

### 读写锁

```c
// 创建
CnRWLock* cn_rwlock_create(void);

// 销毁
void cn_rwlock_destroy(CnRWLock *lock);

// 读锁
void cn_rwlock_rdlock(CnRWLock *lock);

// 写锁
void cn_rwlock_wrlock(CnRWLock *lock);

// 解锁
void cn_rwlock_unlock(CnRWLock *lock);
```

### 条件变量

```c
// 创建
CnCondvar* cn_condvar_create(void);

// 销毁
void cn_condvar_destroy(CnCondvar *cond);

// 等待
void cn_condvar_wait(CnCondvar *cond, CnMutex *mutex);

// 限时等待
int cn_condvar_wait_timeout(CnCondvar *cond, CnMutex *mutex, int timeout_ms);

// 通知
void cn_condvar_signal(CnCondvar *cond);
void cn_condvar_broadcast(CnCondvar *cond);
```

### 原子操作

```c
// 加载
int cn_atomic_load(int *ptr);

// 存储
void cn_atomic_store(int *ptr, int value);

// 交换
int cn_atomic_exchange(int *ptr, int value);

// 比较交换
int cn_atomic_compare_exchange(int *ptr, int expected, int desired);

// 增加
int cn_atomic_increment(int *ptr);

// 减少
int cn_atomic_decrement(int *ptr);

// 加法
int cn_atomic_add(int *ptr, int value);

// 减法
int cn_atomic_sub(int *ptr, int value);
```

## Freestanding 支持

### 内置函数

```c
// 内存操作
void* cn_rt_freestanding_memcpy(void *dest, const void *src, size_t n);
void* cn_rt_freestanding_memset(void *s, int c, size_t n);

// 字符串操作
size_t cn_rt_freestanding_strlen(const char *s);

```

### 内核 I/O

```c
// 注册回调
void cn_rt_register_kernel_io(void (*puts)(const char*),
                               void (*print_int)(long long),
                               void (*putchar)(char));

// 端口 I/O
void cn_rt_outb(uint16_t port, uint8_t value);

```
