# 工具链 API

## 概述

工具链 API 提供编译器、格式化工具、静态检查等工具的编程接口。

## 编译器 API

### 编译选项

```c
typedef struct CnCompileOptions {
    const char *output_file;     // 输出文件
    int optimize_level;          // 优化级别 (0-3)
    int debug_info;              // 调试信息
    int emit_c;                  // 生成 C 代码
    int freestanding;            // Freestanding 模式
    const char *target;          // 目标三元组
    const char **include_paths;  // 包含路径
    int include_path_count;
    const char **defines;        // 宏定义
    int define_count;
} CnCompileOptions;
```

### 编译函数

```c
// 编译文件
int cn_compile_file(const char *input, CnCompileOptions *options);

// 编译内存中的源代码
int cn_compile_string(const char *source, 
                      const char *filename,
                      CnCompileOptions *options);

// 创建编译上下文
CnCompileContext* cn_compile_context_create(void);

// 销毁编译上下文
void cn_compile_context_destroy(CnCompileContext *ctx);

// 设置选项
void cn_compile_set_options(CnCompileContext *ctx, CnCompileOptions *options);

// 执行编译
int cn_compile_execute(CnCompileContext *ctx);

// 获取错误
const char* cn_compile_get_error(CnCompileContext *ctx);
```

## 格式化 API

### 格式选项

```c
typedef struct CnFormatOptions {
    int tab_width;               // 制表符宽度
    int indent_width;            // 缩进宽度
    int use_tabs;                // 使用制表符
    int max_line_length;         // 最大行长度
    int insert_final_newline;    // 插入末尾换行
} CnFormatOptions;
```

### 格式化函数

```c
// 格式化文件
int cn_format_file(const char *filename, CnFormatOptions *options);

// 格式化字符串
char* cn_format_string(const char *source, CnFormatOptions *options);

// 检查格式
int cn_format_check(const char *filename, CnFormatOptions *options);

// 获取格式差异
char* cn_format_diff(const char *original, 
                     const char *formatted,
                     const char *filename);
```

## 静态检查 API

### 检查选项

```c
typedef struct CnCheckOptions {
    int enable_all;              // 启用所有检查
    const char **enabled_rules;  // 启用的规则
    int enabled_rule_count;
    const char **disabled_rules; // 禁用的规则
    int disabled_rule_count;
    int fail_on_error;           // 错误时失败
    int fail_on_warning;         // 警告时失败
} CnCheckOptions;
```

### 检查函数

```c
// 检查文件
CnCheckResult* cn_check_file(const char *filename, CnCheckOptions *options);

// 检查字符串
CnCheckResult* cn_check_string(const char *source, 
                                const char *filename,
                                CnCheckOptions *options);

// 获取诊断
CnDiagnostic* cn_check_get_diagnostic(CnCheckResult *result, int index);

// 获取诊断数量
int cn_check_get_diagnostic_count(CnCheckResult *result);

// 释放结果
void cn_check_result_free(CnCheckResult *result);

// 打印结果
void cn_check_print_result(CnCheckResult *result);
```

### 诊断结构

```c
typedef struct CnDiagnostic {
    enum {
        CN_DIAG_ERROR,
        CN_DIAG_WARNING,
        CN_DIAG_INFO,
    } severity;
    
    const char *message;         // 错误消息
    const char *filename;        // 文件名
    int line;                    // 行号
    int column;                  // 列号
    const char *rule;            // 规则名称
    const char *source_line;     // 源代码行
} CnDiagnostic;
```

## 语言服务器 API

### LSP 配置

```c
typedef struct CnLspConfig {
    int use_stdio;               // 使用标准输入输出
    int use_tcp;                 // 使用 TCP
    int tcp_port;                // TCP 端口
    const char *workspace;       // 工作区路径
} CnLspConfig;
```

### LSP 函数

```c
// 创建服务器
CnLspServer* cn_lsp_server_create(CnLspConfig *config);

// 销毁服务器
void cn_lsp_server_destroy(CnLspServer *server);

// 运行服务器
int cn_lsp_server_run(CnLspServer *server);

// 处理消息
CnLspMessage* cn_lsp_handle_message(CnLspServer *server, 
                                     CnLspMessage *message);

// 发送通知
void cn_lsp_send_notification(CnLspServer *server, 
                              const char *method,
                              cJSON *params);
```

### 文档管理

```c
// 打开文档
void cn_lsp_document_open(CnLspDocumentManager *mgr,
```

## REPL API

### REPL 配置

```c
typedef struct CnReplConfig {
    int quiet;                   // 安静模式
    const char *history_file;    // 历史文件
    const char *init_file;       // 初始化文件
    const char *prompt;          // 提示符
} CnReplConfig;
```

### REPL 函数

```c
// 创建 REPL
CnRepl* cn_repl_create(CnReplConfig *config);

// 销毁 REPL
void cn_repl_destroy(CnRepl *repl);

// 运行 REPL
int cn_repl_run(CnRepl *repl);

// 执行命令
CnReplResult* cn_repl_execute(CnRepl *repl, const char *input);

// 获取历史
const char** cn_repl_get_history(CnRepl *repl, int *count);

// 添加历史
void cn_repl_add_history(CnRepl *repl, const char *input);

// 保存历史
int cn_repl_save_history(CnRepl *repl);

// 加载历史
int cn_repl_load_history(CnRepl *repl);
```

## 性能分析 API

### 分析配置

```c
typedef struct CnPerfConfig {
    int enable_timing;           // 启用计时
    int enable_memory;           // 启用内存分析
    int enable_call_graph;       // 启用调用图
    int hotspot_count;           // 热点数量
} CnPerfConfig;
```

### 分析函数

```c
// 创建分析器
CnPerf* cn_perf_create(CnPerfConfig *config);

// 销毁分析器
void cn_perf_destroy(CnPerf *perf);

// 开始分析
void cn_perf_start(CnPerf *perf);

// 停止分析
void cn_perf_stop(CnPerf *perf);

// 获取结果
CnPerfResult* cn_perf_get_result(CnPerf *perf);

// 打印结果
void cn_perf_print_result(CnPerfResult *result);

// 导出 JSON
char* cn_perf_export_json(CnPerfResult *result);

// 导出调用图
char* cn_perf_export_callgraph(CnPerfResult *result);
```
