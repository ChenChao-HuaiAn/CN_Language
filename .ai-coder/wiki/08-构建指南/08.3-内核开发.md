# 内核开发

## 概述

CN_Language 支持使用 Freestanding 模式开发操作系统内核，无需依赖标准库。

## Freestanding 模式

### 启用 Freestanding

```cn
#定义 自由站立

模块 内核;

函数 内核入口 () {
    // 内核代码
}
```

### 编译选项

```bash
# 编译为 Freestanding 目标
cnc 内核.cn --freestanding -o 内核.bin

# 指定目标平台
cnc 内核.cn --freestanding --target i686-elf -o 内核.bin
```

## 内核模板

### 最小内核

```cn
#定义 自由站立

模块 内核;

// VGA 文本缓冲区
字符* VGA 缓冲区 = (字符*) 0xB8000;

函数 内核入口 () {
    // 清屏
    循环 (整数 i = 0; i < 80 * 25; i = i + 1) {
        VGA 缓冲区 [i * 2] = ' ';
        VGA 缓冲区 [i * 2 + 1] = 0x07;  // 白色字符，黑色背景
    }
    
    // 打印欢迎信息
    字符串 消息 = "你好，内核！";
    循环 (整数 i = 0; 消息 [i] != 0; i = i + 1) {
        VGA 缓冲区 [i * 2] = 消息 [i];
        VGA 缓冲区 [i * 2 + 1] = 0x07;
    }
    
    // 无限循环
    当 (真) {
        // 停机指令
        内联汇编 ("hlt");
    }
}
```

### 带中断处理的内核

```cn
#定义 自由站立

模块 内核;

// 中断描述符表
结构体 中断门 {
    整数 偏移_低：16;
    整数 选择子：16;
    整数 零：8;
    整数 类型：8;
    整数 偏移_高：16;
};

中断门 中断表 [256];

// 中断处理程序
中断处理程序 定时器中断 (整数 中断号) {
    // 处理定时器中断
    打印 ("定时器中断：%d\n", 中断号);
}

函数 初始化中断 () {
    // 设置中断描述符表
    // ...
}

函数 内核入口 () {
    初始化中断 ();
    
    // 启用中断
    内联汇编 ("sti");
    
    // 主循环
    当 (真) {
        内联汇编 ("hlt");
    }
}
```

## 硬件访问

### 端口 I/O

```cn
// 输出到端口
函数 输出端口 (整数 端口，整数 值) {
    内联汇编 ("outb %0, %1" :: "a"(值), "d"(端口));
}

// 从端口输入
函数 输入端口 (整数 端口) {
    整数 值;
    内联汇编 ("inb %1, %0" : "=a"(值) : "d"(端口));
    返回 值;
}

// 使用示例
函数 串口初始化 () {
    输出端口 (0x3F8 + 1, 0x00);  // 禁用中断
    输出端口 (0x3F8 + 3, 0x80);  // 启用除数锁存器
    输出端口 (0x3F8 + 0, 0x01);  // 设置波特率
    输出端口 (0x3F8 + 1, 0x00);
    输出端口 (0x3F8 + 3, 0x03);  // 8 位，无奇偶校验
}
```

## 内存管理

### 物理内存分配

```cn
#定义 物理内存大小 (64 * 1024 * 1024)  // 64MB

整数 位图 [(物理内存大小 / 4096) / 32];

函数 初始化物理内存 () {
    // 清零位图
    内存。设置 (位图，0, sizeof(位图));
    
    // 标记内核占用的页面
    // ...
}

函数 分配物理页 () {
    // 查找空闲页面
    循环 (整数 i = 0; i < sizeof(位图) / 4; i = i + 1) {
        如果 (位图 [i] != 0xFFFFFFFF) {
            // 找到空闲位
            循环 (整数 j = 0; j < 32; j = j + 1) {
                如果 ((位图 [i] & (1 << j)) == 0) {
                    位图 [i] |= (1 << j);
                    返回 (i * 32 + j) * 4096;
                }
            }
        }
    }
    返回 0;  // 内存不足
}
```

## 引导加载

### Multiboot 头

```c
// boot/multiboot_header.S
.section .multiboot
.align 4
.long 0x1BADB002
.long 0x00000000
.long -(0x1BADB002 + 0x00000000)
```

### 链接脚本

```ld
/* boot/linker.ld */
ENTRY(内核入口)

SECTIONS
{
    . = 1M;
    
    .boot :
    {
        *(.multiboot)
    }
    
    .text :
    {
        *(.text)
    }
    
    .rodata :
    {
        *(.rodata)
    }
    
    .data :
    {
        *(.data)
    }
    
    .bss :
    {
        *(.bss)
    }
}
```

## 使用 QEMU 测试

### 编译内核

```bash
# 编译为二进制
cnc 内核.cn --freestanding -o 内核.bin

# 或使用链接脚本
gcc -nostdlib -T boot/linker.ld -o 内核.elf boot/boot.o 内核.o
```

### 运行 QEMU

```bash
# 基本运行
qemu-system-x86_64 -kernel 内核.elf

# 带串口输出
qemu-system-x86_64 -kernel 内核.elf -serial stdio

# 带调试
qemu-system-x86_64 -kernel 内核.elf -s -S

# 在另一个终端连接 GDB
gdb 内核.elf
(gdb) target remote localhost:1234
```

### 创建 ISO

```bash
# 创建引导 ISO
grub-mkrescue -o 内核.iso iso/

# 运行
qemu-system-x86_64 -cdrom 内核.iso
```
