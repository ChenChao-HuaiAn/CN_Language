# VSCode 扩展

## 概述

CN_Language 的 VSCode 扩展提供语法高亮、代码补全、调试等 IDE 功能。

## 文件位置

[`tools/vscode/cnlang-lsp/`](../../tools/vscode/cnlang-lsp/)

## 目录结构

```
tools/vscode/cnlang-lsp/
├── package.json              # 扩展配置
├── extension.js              # 扩展入口
├── language-configuration.json # 语言配置
├── cnlang.tmLanguage.json    # 语法高亮
└── README.md                 # 扩展说明
```

## 安装扩展

### 从源码安装

```bash
# 进入扩展目录
cd tools/vscode/cnlang-lsp

# 安装依赖
npm install

# 编译扩展
npm run compile

# 在 VSCode 中运行
# 按 F5 启动扩展开发宿主
```

### 打包安装

```bash
# 安装 vsce
npm install -g vsce

# 打包
vsce package

# 安装 .vsix 文件
# 在 VSCode 中：扩展 -> ... -> 从 VSIX 安装
```

## 扩展功能

### extension.js

```javascript
const vscode = require('vscode');

function activate(context) {
    console.log('CN Language 扩展已激活');
    
    // 注册编译命令
    let compileCommand = vscode.commands.registerCommand('cnlang.compile', function () {
        let editor = vscode.window.activeTextEditor;
        if (!editor) return;
        
        let document = editor.document;
        let filePath = document.fileName;
        
        // 执行编译
        let terminal = vscode.window.createTerminal('CN Compile');
        terminal.sendText(`cnc "${filePath}" -o "${filePath}.exe"`);
        terminal.show();
    });
    
    // 注册格式化命令
    let formatCommand = vscode.commands.registerCommand('cnlang.format', function () {
        let editor = vscode.window.activeTextEditor;
        if (!editor) return;
        
        let document = editor.document;
        let filePath = document.fileName;
        
        // 执行格式化
        let terminal = vscode.window.createTerminal('CN Format');
        terminal.sendText(`cnfmt "${filePath}"`);
        terminal.show();
    });
    
    context.subscriptions.push(compileCommand);
    context.subscriptions.push(formatCommand);
    
    // 注册语言服务器
    startLanguageServer(context);
}

function startLanguageServer(context) {
    // LSP 客户端配置
    let serverOptions = {
        command: 'cnlsp',
        args: ['--stdio']
    };
    
    let clientOptions = {
        documentSelector: [{ scheme: 'file', language: 'cnlang' }]
    };
    
    // 启动 LSP 客户端
    // 使用 vscode-languageclient 包
}

function deactivate() {
    console.log('CN Language 扩展已停用');
}

module.exports = {
    activate,
    deactivate
}
```

## 测试扩展

### 测试文件

```javascript
// test/suite/extension.test.js
const assert = require('assert');
const vscode = require('vscode');

suite('CN Language 扩展测试', () => {
    test('扩展应该激活', async () => {
        const ext = vscode.extensions.getExtension('your-extension.cnlang-lsp');
        assert.ok(ext);
        await ext.activate();
        assert.ok(ext.isActive);
    });
    
    test('编译命令应该注册', async () => {
        const commands = await vscode.commands.getCommands();
        assert.ok(commands.includes('cnlang.compile'));
    });
});
```
