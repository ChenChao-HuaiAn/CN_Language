# IR 核心结构

## 概述

CN_Language 的 IR 采用三地址码形式，每个指令最多有三个操作数（一个目标，两个源）。IR 设计简洁，便于优化和代码生成。

## 文件位置

[`src/ir/core/ir.c`](../../src/ir/core/ir.c:1)

## 核心数据结构

### IR 模块 (CnIrModule)

```c
typedef struct CnIrModule {
    CnIrFunction *first_func;    // 函数链表头
    CnIrFunction *last_func;     // 函数链表尾
    CnIrGlobalVar *first_global; // 全局变量链表头
    CnIrGlobalVar *last_global;  // 全局变量链表尾
    CnTargetTriple target;       // 目标平台三元组
    CnCompileMode compile_mode;  // 编译模式（hosted/freestanding）
} CnIrModule;
```

**说明：**
- 模块是 IR 的顶层容器
- 包含所有函数和全局变量
- 目标三元组指定生成代码的目标平台
- 编译模式决定是否可以依赖标准库

### IR 函数 (CnIrFunction)

```c
typedef struct CnIrFunction {
    char *name;                  // 函数名
    CnType *return_type;         // 返回类型
    CnIrOperand *params;         // 参数列表
    size_t param_count;          // 参数数量
    CnIrOperand *locals;         // 局部变量列表
    size_t local_count;          // 局部变量数量
    CnIrBasicBlock *first_block; // 基本块链表头
    CnIrBasicBlock *last_block;  // 基本块链表尾
    int next_reg_id;             // 下一个寄存器 ID
    int is_interrupt_handler;    // 是否为中断处理函数
    int interrupt_vector;        // 中断向量号
    CnIrFunction *next;          // 下一个函数
} CnIrFunction;
```

### 基本块 (CnIrBasicBlock)

```c
typedef struct CnIrBasicBlock {
    char *name;                  // 块名称（用于调试）
    CnIrInst *first_inst;        // 指令链表头
    CnIrInst *last_inst;         // 指令链表尾
    CnIrBasicBlockList *preds;   // 前驱块列表
    CnIrBasicBlockList *succs;   // 后继块列表
    CnIrBasicBlock *next;        // 下一个块
    CnIrBasicBlock *prev;        // 上一个块
} CnIrBasicBlock;
```

**说明：**
- 基本块是 IR 的基本控制流单位
- 每个块有一个入口（第一条指令）和一个出口（最后一条指令）
- 控制流通过前驱/后继关系表示

### IR 指令 (CnIrInst)

```c
typedef struct CnIrInst {
    CnIrInstKind kind;           // 指令类型
    CnIrOperand dest;            // 目标操作数
    CnIrOperand src1;            // 源操作数 1
    CnIrOperand src2;            // 源操作数 2
    CnIrOperand *extra_args;     // 额外参数（用于调用等）
    size_t extra_args_count;     // 额外参数数量
    CnIrInst *next;              // 下一条指令
    CnIrInst *prev;              // 上一条指令
} CnIrInst;
```

### IR 操作数 (CnIrOperand)

```c
typedef struct CnIrOperand {
    enum {
        CN_IR_OP_NONE,           // 无操作数
        CN_IR_OP_REG,            // 寄存器
        CN_IR_OP_IMM_INT,        // 整数立即数
        CN_IR_OP_IMM_FLOAT,      // 浮点立即数
        CN_IR_OP_IMM_STRING,     // 字符串立即数
        CN_IR_OP_GLOBAL,         // 全局变量
        CN_IR_OP_LOCAL,          // 局部变量
        CN_IR_OP_PARAM,          // 参数
        CN_IR_OP_ADDR,           // 地址（用于内存访问）
    } kind;
    
    union {
        int reg_id;              // 寄存器 ID
        long long imm_int;       // 整数立即数
        double imm_float;        // 浮点立即数
        const char *imm_string;  // 字符串立即数
        const char *global_name; // 全局变量名
        int local_index;         // 局部变量索引
        int param_index;         // 参数索引
        struct {
            CnIrOperand base;    // 基址
            CnIrOperand offset;  // 偏移
        } addr;
    } as;
    
    CnType *type;                // 操作数类型
} CnIrOperand;
```

## IR 示例

### 源代码

```cn
函数 加法 (整数 a, 整数 b) {
    整数 结果 = a + b;
    返回 结果;
}

函数 主程序 () {
    整数 x = 10;
    整数 y = 20;
    整数 和 = 加法 (x, y);
    打印 (和);
    返回 0;
}
```

### 生成的 IR

```ir
; 函数：加法
define 整数 @加法 (整数 %a, 整数 %b) {
entry:
    %结果 = ALLOCA 4
    %tmp1 = ADD %a, %b
    STORE %tmp1, %结果
    %tmp2 = LOAD %结果
    RET %tmp2
}

; 函数：主程序
define 整数 @主程序 () {
entry:
    %x = ALLOCA 4
    %y = ALLOCA 4
    %和 = ALLOCA 4
    STORE 10, %x
    STORE 20, %y
    %tmp1 = LOAD %x
    %tmp2 = LOAD %y
    %tmp3 = CALL @加法，[%tmp1, %tmp2]
    STORE %tmp3, %和
    %tmp4 = LOAD %和
    CALL @打印，[%tmp4]
    RET 0
}
```
