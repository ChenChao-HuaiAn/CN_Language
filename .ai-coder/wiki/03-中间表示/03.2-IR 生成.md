# IR 生成

## 概述

IR 生成是将 AST 转换为中间表示的过程。CN_Language 的 IR 生成器遍历 AST，为每个节点生成相应的 IR 指令。

## 文件位置

[`src/ir/gen/irgen.c`](../../src/ir/gen/irgen.c:1)

## IR 生成器结构

```c
typedef struct CnIrGen {
    CnIrModule *module;          // IR 模块
    CnIrFunction *current_func;  // 当前函数
    CnIrBasicBlock *current_block; // 当前基本块
    CnHashMap *symbol_table;     // 符号表
    CnDiagnostics *diagnostics;  // 诊断信息
    int next_reg_id;             // 下一个寄存器 ID
} CnIrGen;
```

## 生成流程

### 1. 初始化

```c
void cn_ir_gen_init(CnIrGen *gen, CnIrModule *module)
{
    gen->module = module;
    gen->current_func = NULL;
    gen->current_block = NULL;
    gen->symbol_table = create_hash_map();
    gen->next_reg_id = 0;
}
```

### 2. 生成程序

```c
CnIrModule *cn_ir_gen_program(CnIrGen *gen, CnAstProgram *program)
{
    // 生成全局变量
    for (size_t i = 0; i < program->global_var_count; i++) {
        gen_global_var(gen, program->global_vars[i]);
    }
    
    // 生成函数
    for (size_t i = 0; i < program->function_count; i++) {
        gen_function(gen, program->functions[i]);
    }
    
    return gen->module;
}
```

### 3. 生成函数

```c
void gen_function(CnIrGen *gen, CnAstFunctionDecl *func_decl)
{
    // 创建函数
    gen->current_func = cn_ir_function_new(func_decl->name, func_decl->return_type);
    
    // 添加参数
    for (size_t i = 0; i < func_decl->param_count; i++) {
        CnIrOperand param = cn_ir_op_param(i, func_decl->params[i].type);
        cn_ir_function_add_param(gen->current_func, param);
        // 将参数加入符号表
        hash_map_put(gen->symbol_table, func_decl->params[i].name, param);
    }
    
    // 创建入口块
    CnIrBasicBlock *entry = cn_ir_basic_block_new("entry");
    cn_ir_function_add_block(gen->current_func, entry);
    gen->current_block = entry;
    
    // 生成函数体
    gen_block(gen, func_decl->body);
    
    // 添加模块
    add_function_to_module(gen->module, gen->current_func);
}
```

### 4. 生成语句

#### 变量声明

```c
void gen_var_decl(CnIrGen *gen, CnAstVarDecl *decl)
{
    // 分配栈空间
    int size = get_type_size(decl->type);
    CnIrOperand local = cn_ir_op_local(gen->current_func->local_count, decl->type);
    cn_ir_function_add_local(gen->current_func, local);
    
    // 生成初始化表达式（如果有）
    if (decl->initializer) {
        CnIrOperand value = gen_expr(gen, decl->initializer);
        CnIrInst *store = cn_ir_inst_new(CN_IR_INST_STORE, cn_ir_op_none(), value, local);
        cn_ir_basic_block_add_inst(gen->current_block, store);
    }
    
    // 加入符号表
    hash_map_put(gen->symbol_table, decl->name, local);
}
```

#### 返回语句

```c
void gen_return_stmt(CnIrGen *gen, CnAstReturnStmt *ret)
{
    if (ret->expr) {
        CnIrOperand value = gen_expr(gen, ret->expr);
        CnIrInst *inst = cn_ir_inst_new(CN_IR_INST_RET, cn_ir_op_none(), value, cn_ir_op_none());
        cn_ir_basic_block_add_inst(gen->current_block, inst);
    } else {
        CnIrInst *inst = cn_ir_inst_new(CN_IR_INST_RET_VOID, cn_ir_op_none(), cn_ir_op_none(), cn_ir_op_none());
        cn_ir_basic_block_add_inst(gen->current_block, inst);
    }
}
```

#### 条件语句

```c
void gen_if_stmt(CnIrGen *gen, CnAstIfStmt *if_stmt)
{
    // 生成条件表达式
    CnIrOperand cond = gen_expr(gen, if_stmt->condition);
    
    // 创建块
    CnIrBasicBlock *then_block = cn_ir_basic_block_new("then");
    CnIrBasicBlock *else_block = if_stmt->else_block ? cn_ir_basic_block_new("else") : NULL;
    CnIrBasicBlock *merge_block = cn_ir_basic_block_new("merge");
    
    // 生成条件跳转
    CnIrInst *jz;
    if (else_block) {
        jz = cn_ir_inst_new(CN_IR_INST_JZ, cn_ir_op_none(), cond, cn_ir_op_none());
    } else {
        jz = cn_ir_inst_new(CN_IR_INST_JZ, cn_ir_op_none(), cond, cn_ir_op_none());
    }
    cn_ir_basic_block_add_inst(gen->current_block, jz);
    
    // 连接控制流
    cn_ir_basic_block_connect(gen->current_block, then_block);
    if (else_block) {
        cn_ir_basic_block_connect(gen->current_block, else_block);
    } else {
        cn_ir_basic_block_connect(gen->current_block, merge_block);
    }
    
    // 生成 then 块
    gen->current_block = then_block;
    gen_block(gen, if_stmt->then_block);
    cn_ir_basic_block_add_inst(gen->current_block, 
                               cn_ir_inst_new(CN_IR_INST_JMP, cn_ir_op_none(), cn_ir_op_none(), cn_ir_op_none()));
    
    // 生成 else 块（如果有）
    if (else_block) {
        gen->current_block = else_block;
        gen_block(gen, if_stmt->else_block);
        cn_ir_basic_block_add_inst(gen->current_block,
                                   cn_ir_inst_new(CN_IR_INST_JMP, cn_ir_op_none(), cn_ir_op_none(), cn_ir_op_none()));
    }
    
    // 连接到合并块
    gen->current_block = merge_block;
}
```

#### 循环语句

```c
void gen_while_stmt(CnIrGen *gen, CnAstWhileStmt *while_stmt)
{
    CnIrBasicBlock *cond_block = cn_ir_basic_block_new("while.cond");
    CnIrBasicBlock *body_block = cn_ir_basic_block_new("while.body");
    CnIrBasicBlock *exit_block = cn_ir_basic_block_new("while.exit");
    
    // 跳转到条件检查
    cn_ir_basic_block_add_inst(gen->current_block,
                               cn_ir_inst_new(CN_IR_INST_JMP, cn_ir_op_none(), cn_ir_op_none(), cn_ir_op_none()));
    cn_ir_basic_block_connect(gen->current_block, cond_block);
    
    // 生成条件
    gen->current_block = cond_block;
    CnIrOperand cond = gen_expr(gen, while_stmt->condition);
    cn_ir_basic_block_add_inst(cond_block,
                               cn_ir_inst_new(CN_IR_INST_JNZ, cn_ir_op_none(), cond, cn_ir_op_none()));
    cn_ir_basic_block_connect(cond_block, body_block);
    cn_ir_basic_block_connect(cond_block, exit_block);
    
    // 生成循环体
    gen->current_block = body_block;
    gen_block(gen, while_stmt->body);
    cn_ir_basic_block_add_inst(gen->current_block,
                               cn_ir_inst_new(CN_IR_INST_JMP, cn_ir_op_none(), cn_ir_op_none(), cn_ir_op_none()));
    cn_ir_basic_block_connect(gen->current_block, cond_block);
    
    // 继续生成
    gen->current_block = exit_block;
}
```

### 5. 生成表达式

#### 二元运算

```c
CnIrOperand gen_binary(CnIrGen *gen, CnAstBinaryExpr *binary)
{
    CnIrOperand left = gen_expr(gen, binary->left);
    CnIrOperand right = gen_expr(gen, binary->right);
    
    CnIrInstKind op;
    switch (binary->op) {
        case CN_AST_BINARY_ADD: op = CN_IR_INST_ADD; break;
        case CN_AST_BINARY_SUB: op = CN_IR_INST_SUB; break;
        case CN_AST_BINARY_MUL: op = CN_IR_INST_MUL; break;
        case CN_AST_BINARY_DIV: op = CN_IR_INST_DIV; break;
        case CN_AST_BINARY_MOD: op = CN_IR_INST_MOD; break;
        // ... 其他运算符
        default: return cn_ir_op_none();
    }
    
    int reg_id = gen->next_reg_id++;
    CnIrOperand dest = cn_ir_op_reg(reg_id, binary->type);
    
    CnIrInst *inst = cn_ir_inst_new(op, dest, left, right);
    cn_ir_basic_block_add_inst(gen->current_block, inst);
    
    return dest;
}
```

#### 函数调用

```c
CnIrOperand gen_call(CnIrGen *gen, CnAstCallExpr *call)
{
    // 生成参数
    CnIrOperand *args = malloc(sizeof(CnIrOperand) * call->argument_count);
    for (size_t i = 0; i < call->argument_count; i++) {
        args[i] = gen_expr(gen, call->arguments[i]);
    }
    
    // 生成调用
    CnIrOperand callee = gen_expr(gen, call->callee);
    
    int reg_id = gen->next_reg_id++;
    CnIrOperand dest = cn_ir_op_reg(reg_id, call->type);
    
    CnIrInst *inst = cn_ir_inst_new(CN_IR_INST_CALL, dest, callee, cn_ir_op_none());
    inst->extra_args = args;
    inst->extra_args_count = call->argument_count;
    
    cn_ir_basic_block_add_inst(gen->current_block, inst);
    
    return dest;
}
```

## 符号表管理

### 作用域

```c
typedef struct CnIrScope {
    CnHashMap *symbols;
    struct CnIrScope *parent;
} CnIrScope;

void cn_ir_gen_enter_scope(CnIrGen *gen)
{
    CnIrScope *scope = malloc(sizeof(CnIrScope));
    scope->symbols = create_hash_map();
    scope->parent = gen->current_scope;
    gen->current_scope = scope;
}

void cn_ir_gen_exit_scope(CnIrGen *gen)
{
    CnIrScope *scope = gen->current_scope;
    gen->current_scope = scope->parent;
    hash_map_free(scope->symbols);
    free(scope);
}
```

## 错误处理

```c
static void report_irgen_error(CnIrGen *gen, CnDiagCode code, const char *message,
                               size_t line, size_t column)
{
    if (!gen || !gen->diagnostics) {
        return;
    }
    
    cn_support_diagnostics_report(gen->diagnostics,
                                  CN_DIAG_SEVERITY_ERROR,
                                  code,
                                  gen->current_filename,
                                  line,
                                  column,
                                  message);
}
```

## 使用示例

```c
#include "cnlang/ir/ir.h"
#include "cnlang/ir/gen/irgen.h"
#include "cnlang/frontend/parser.h"
#include <stdio.h>

int main() {
    // 解析源代码
    CnLexer lexer;
    CnParser *parser;
    CnAstProgram *program;
    
    // ... 解析代码 ...
    
    // 生成 IR
    CnIrModule *module = cn_ir_module_new();
    CnIrGen gen;
    cn_ir_gen_init(&gen, module);
    
    CnIrModule *ir = cn_ir_gen_program(&gen, program);
    
    // 打印 IR
    print_ir_module(ir);
    
    // 清理
    cn_ir_module_free(ir);
    
    return 0;
}
```
