# 抽象语法树 (AST)

## 概述

抽象语法树（Abstract Syntax Tree，AST）是源代码的树状表示，每个节点代表源代码中的一个结构。AST 去除了源代码中的无关细节（如空白字符、注释），保留了程序的逻辑结构。

## 核心数据结构

### 表达式节点 (CnAstExpr)

```c
typedef struct CnAstExpr {
    enum {
        CN_AST_EXPR_BINARY,           // 二元运算
        CN_AST_EXPR_LOGICAL,          // 逻辑运算
        CN_AST_EXPR_UNARY,            // 一元运算
        CN_AST_EXPR_ASSIGN,           // 赋值
        CN_AST_EXPR_CALL,             // 函数调用
        CN_AST_EXPR_IDENTIFIER,       // 标识符
        CN_AST_EXPR_INTEGER_LITERAL,  // 整数
        CN_AST_EXPR_FLOAT_LITERAL,    // 浮点数
        CN_AST_EXPR_STRING_LITERAL,   // 字符串
        CN_AST_EXPR_BOOL_LITERAL,     // 布尔值
        CN_AST_EXPR_ARRAY_LITERAL,    // 数组字面量
        CN_AST_EXPR_INDEX,            // 索引访问
        CN_AST_EXPR_MEMBER_ACCESS,    // 成员访问
        CN_AST_EXPR_STRUCT_LITERAL,   // 结构体字面量
        CN_AST_EXPR_MEMORY_READ,      // 内存读取
        CN_AST_EXPR_MEMORY_WRITE,     // 内存写入
        CN_AST_EXPR_MEMORY_COPY,      // 内存复制
        CN_AST_EXPR_MEMORY_SET,       // 内存设置
        CN_AST_EXPR_MEMORY_MAP,       // 内存映射
        CN_AST_EXPR_MEMORY_UNMAP,     // 解除内存映射
        CN_AST_EXPR_INLINE_ASM,       // 内联汇编
    } kind;
    
    union {
        struct {
            CnAstBinaryOp op;
            CnAstExpr *left;
            CnAstExpr *right;
        } binary;
        
        struct {
            CnAstLogicalOp op;
            CnAstExpr *left;
            CnAstExpr *right;
        } logical;
        
        struct {
            CnAstUnaryOp op;
            CnAstExpr *operand;
        } unary;
        
        struct {
            CnAstExpr *target;
            CnAstExpr *value;
        } assign;
        
        struct {
            CnAstExpr *callee;
            CnAstExpr **arguments;
            size_t argument_count;
        } call;
        
        // ... 其他表达式类型
    } as;
} CnAstExpr;
```

### 语句节点 (CnAstStmt)

```c
typedef struct CnAstStmt {
    enum {
        CN_AST_STMT_BLOCK,            // 代码块
        CN_AST_STMT_VAR_DECL,         // 变量声明
        CN_AST_STMT_EXPR,             // 表达式语句
        CN_AST_STMT_RETURN,           // 返回语句
        CN_AST_STMT_IF,               // 条件语句
        CN_AST_STMT_WHILE,            // while 循环
        CN_AST_STMT_FOR,              // for 循环
        CN_AST_STMT_SWITCH,           // switch 语句
        CN_AST_STMT_BREAK,            // break 语句
        CN_AST_STMT_CONTINUE,         // continue 语句
        CN_AST_STMT_STRUCT_DECL,      // 结构体声明
        CN_AST_STMT_ENUM_DECL,        // 枚举声明
        CN_AST_STMT_MODULE_DECL,      // 模块声明
        CN_AST_STMT_IMPORT,           // 导入语句
    } kind;
    
    union {
        struct {
            CnAstStmt **stmts;
            size_t stmt_count;
        } block;
        
        struct {
            const char *name;
            size_t name_length;
            CnType *type;
            CnAstExpr *initializer;
            CnVisibility visibility;
        } var_decl;
        
        struct {
            CnAstExpr *expr;
        } expr;
        
        struct {
            CnAstExpr *expr;
        } return_stmt;
        
        // ... 其他语句类型
    } as;
} CnAstStmt;
```

### 函数声明节点

typedef struct CnAstFunctionDecl {

### 示例：打印 AST

```c
void print_ast(CnAstProgram *program, int indent)
{
    printf("%*sProgram:\n", indent, "");
    
    // 打印函数
    for (size_t i = 0; i < program->function_count; i++) {
        print_function(program->functions[i], indent + 2);
    }
    
    // 打印结构体
    for (size_t i = 0; i < program->struct_count; i++) {
        print_struct(program->structs[i], indent + 2);
    }
}

void print_function(CnAstFunctionDecl *func, int indent)
{
    printf("%*sFunction: %.*s\n", indent, "", 
           (int)func->name_length, func->name);
    
    // 打印参数
    for (size_t i = 0; i < func->param_count; i++) {
        printf("%*s  Param: %.*s\n", indent + 2, "",
               (int)func->params[i].name_length, 
               func->params[i].name);
    }
    
    // 打印函数体
    print_block(func->body, indent + 2);
}
```

## 内存管理

### 释放表达式

```c
void cn_frontend_ast_expr_free(CnAstExpr *expr)
{
    if (!expr) {
        return;
    }

    switch (expr->kind) {
    case CN_AST_EXPR_BINARY:
        cn_frontend_ast_expr_free(expr->as.binary.left);
        cn_frontend_ast_expr_free(expr->as.binary.right);
        break;
    case CN_AST_EXPR_CALL:
        cn_frontend_ast_expr_free(expr->as.call.callee);
        cn_frontend_ast_expr_array_free(expr->as.call.arguments, 
                                        expr->as.call.argument_count);
        free(expr->as.call.arguments);
        break;
    // ... 其他情况
    }

    free(expr);
}
```

### 释放语句

```c
void cn_frontend_ast_stmt_free(CnAstStmt *stmt)
{
    if (!stmt) {
        return;
    }

    switch (stmt->kind) {
    case CN_AST_STMT_BLOCK:
        cn_frontend_ast_block_free(stmt->as.block);
        break;
    case CN_AST_STMT_VAR_DECL:
        cn_frontend_ast_expr_free(stmt->as.var_decl.initializer);
        break;
    case CN_AST_STMT_IF:
        cn_frontend_ast_expr_free(stmt->as.if_stmt.condition);
        cn_frontend_ast_block_free(stmt->as.if_stmt.then_block);
        cn_frontend_ast_block_free(stmt->as.if_stmt.else_block);
        break;
    // ... 其他情况
    }

    free(stmt);
}
```

### 释放程序

```c
void cn_frontend_ast_program_free(CnAstProgram *program)
{
    if (!program) {
        return;
    }

    // 释放函数
    for (size_t i = 0; i < program->function_count; i++) {
        cn_frontend_ast_function_free(program->functions[i]);
    }
    free(program->functions);

    // 释放结构体
    for (size_t i = 0; i < program->struct_count; i++) {
        cn_frontend_ast_stmt_free(program->structs[i]);
    }
    free(program->structs);

    // 释放枚举
    for (size_t i = 0; i < program->enum_count; i++) {
        cn_frontend_ast_stmt_free(program->enums[i]);
    }
    free(program->enums);

    free(program);
}
```

## AST 示例

### 源代码

```cn
函数 加法 (整数 a, 整数 b) {
    返回 a + b;
}

函数 主程序 () {
    整数 结果 = 加法 (5, 3);
    打印 (结果);
    返回 0;
}
```

### 对应的 AST

```
Program
├── Function: 加法
│   ├── params: [a: 整数，b: 整数]
│   ├── return_type: (推断)
│   └── body:
│       └── Block
│           └── Return
│               └── Binary(+)
│                   ├── Identifier(a)
│                   └── Identifier(b)
└── Function: 主程序
    ├── params: []
    ├── return_type: (推断)
    └── body:
        └── Block
            ├── VarDecl: 结果
            │   └── Call: 加法
            │       ├── IntegerLiteral(5)
            │       └── IntegerLiteral(3)
            ├── Expr: Call: 打印
            │   └── Identifier(结果)
            └── Return
                └── IntegerLiteral(0)
```

## 类型信息

AST 节点可以携带类型信息，用于后续的语义分析和代码生成：

```c
typedef struct CnType {
    enum {
        CN_TYPE_INTEGER,
        CN_TYPE_FLOAT,
        CN_TYPE_BOOL,
        CN_TYPE_STRING,
        CN_TYPE_VOID,
        CN_TYPE_ARRAY,
        CN_TYPE_POINTER,
        CN_TYPE_FUNCTION,
        CN_TYPE_STRUCT,
        CN_TYPE_ENUM,
    } kind;
    
    union {
        struct {
            CnType *element_type;
            size_t size;
        } array;
        
        struct {
            CnType *pointee;
        } pointer;
        
        struct {
            CnType *return_type;
            CnType **param_types;
            size_t param_count;
            int is_variadic;
        } function;
        
        struct {
            const char *name;
        } named;
    } as;
} CnType;
```
