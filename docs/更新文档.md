### 🔧 API设计

```c
// 基础参数获取
int cn_rt_cli_argc(void);                    // 获取参数个数
const char* cn_rt_cli_argv(int index);       // 获取指定参数
const char* cn_rt_cli_program_name(void);    // 获取程序名称

// 选项查找
const char* cn_rt_cli_find_option(const char* option_name);  // 查找选项值
int cn_rt_cli_has_option(const char* option_name);           // 检查选项存在

// 高级功能
const char* cn_rt_cli_find_option_ex(const char* short_opt, const char* long_opt);
int cn_rt_cli_has_option_ex(const char* short_opt, const char* long_opt);
const char* cn_rt_cli_get_positional_arg(int index);  // 获取位置参数
int cn_rt_cli_get_positional_arg_count(void);         // 获取位置参数个数
```
