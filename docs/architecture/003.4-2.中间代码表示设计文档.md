# CN_Language 中间代码表示设计文档

## 概述

中间代码表示（Intermediate Representation，IR）是编译器前端和后端之间的桥梁，它将抽象语法树（AST）转换为更适合优化和代码生成的中间形式。本设计文档定义了CN_Language的中间代码表示系统。

## 设计目标

1. **平台无关性**：IR应该独立于目标平台
2. **优化友好**：支持各种编译器优化
3. **易于生成**：从AST容易转换到IR
4. **易于转换**：从IR容易转换到目标代码
5. **可扩展性**：支持新的IR类型和优化

## 架构设计

### 分层IR设计

CN_Language采用分层IR设计，支持多种IR形式：

1. **高级IR（HIR）**：接近AST，保留高级语言特性
2. **中级IR（MIR）**：三地址码形式，适合优化
3. **低级IR（LIR）**：接近目标机器，包含寄存器分配等信息

### 核心接口

```c
// IR接口定义
typedef struct Stru_IrInterface_t {
    // 基本信息
    const char* (*get_name)(void);
    const char* (*get_version)(void);
    
    // IR构建
    void* (*create_from_ast)(Stru_AstNode_t* ast);
    void* (*create_empty)(void);
    
    // IR操作
    bool (*add_instruction)(void* ir, Stru_IrInstruction_t* instr);
    bool (*remove_instruction)(void* ir, size_t index);
    Stru_IrInstruction_t* (*get_instruction)(void* ir, size_t index);
    size_t (*get_instruction_count)(void* ir);
    
    // IR转换
    void* (*convert_to_lower)(void* ir);
    void* (*convert_to_higher)(void* ir);
    
    // IR优化
    bool (*apply_optimization)(void* ir, Eum_OptimizationType type);
    
    // 序列化/反序列化
    const char* (*serialize)(void* ir);
    void* (*deserialize)(const char* data);
    
    // 资源管理
    void (*destroy)(void* ir);
} Stru_IrInterface_t;
```

## IR类型定义

### 1. 三地址码IR（Three-Address Code IR）

三地址码是最常用的中间表示形式，每条指令最多包含三个操作数。

```c
// 三地址码指令类型
typedef enum {
    Eum_TAC_ASSIGN,      // x = y
    Eum_TAC_BINARY_OP,   // x = y op z
    Eum_TAC_UNARY_OP,    // x = op y
    Eum_TAC_COPY,        // x = y
    Eum_TAC_GOTO,        // goto L
    Eum_TAC_IF,          // if x goto L
    Eum_TAC_PARAM,       // param x
    Eum_TAC_CALL,        // call p, n
    Eum_TAC_RETURN,      // return x
    Eum_TAC_LABEL,       // L:
    Eum_TAC_FUNCTION,    // function f
    Eum_TAC_END_FUNCTION // end function
} Eum_TacInstructionType;

// 三地址码指令结构
typedef struct {
    Eum_TacInstructionType type;
    char* result;        // 结果操作数
    char* arg1;          // 第一个操作数
    char* arg2;          // 第二个操作数
    char* op;            // 操作符
    int label;           // 标签（用于跳转）
    size_t line;         // 源代码行号
    size_t column;       // 源代码列号
} Stru_TacInstruction_t;
```

### 2. 静态单赋值形式（SSA Form）

SSA形式是优化友好的IR，每个变量只被赋值一次。

```c
// SSA指令类型
typedef enum {
    Eum_SSA_PHI,         // x = φ(x1, x2, ..., xn)
    Eum_SSA_COPY,        // x = y
    Eum_SSA_BINARY,      // x = y op z
    Eum_SSA_UNARY,       // x = op y
    Eum_SSA_LOAD,        // x = *y
    Eum_SSA_STORE,       // *x = y
    Eum_SSA_CALL,        // x = call f(args)
    Eum_SSA_RETURN,      // return x
    Eum_SSA_BRANCH,      // br cond, label1, label2
    Eum_SSA_JUMP,        // jump label
    Eum_SSA_LABEL        // label:
} Eum_SsaInstructionType;

// SSA指令结构
typedef struct {
    Eum_SsaInstructionType type;
    char* result;        // 结果变量（SSA变量）
    char** operands;     // 操作数数组
    size_t operand_count;// 操作数数量
    char* op;            // 操作符
    int* labels;         // 标签数组（用于分支）
    size_t label_count;  // 标签数量
    size_t line;         // 源代码行号
    size_t column;       // 源代码列号
} Stru_SsaInstruction_t;
```

### 3. 控制流图（Control Flow Graph）

控制流图表示程序的控制流结构。

```c
// 基本块结构
typedef struct Stru_BasicBlock_t {
    char* name;                     // 基本块名称
    Stru_IrInstruction_t** instructions; // 指令数组
    size_t instruction_count;       // 指令数量
    
    // 前驱和后继基本块
    struct Stru_BasicBlock_t** predecessors;
    size_t predecessor_count;
    struct Stru_BasicBlock_t** successors;
    size_t successor_count;
    
    // 支配关系
    struct Stru_BasicBlock_t** dominators;
    size_t dominator_count;
    struct Stru_BasicBlock_t* immediate_dominator;
    
    // 数据流信息
    void* in_set;   // 到达定义入集
    void* out_set;  // 到达定义出集
} Stru_BasicBlock_t;

// 控制流图结构
typedef struct {
    char* name;                     // 函数名
    Stru_BasicBlock_t** blocks;     // 基本块数组
    size_t block_count;             // 基本块数量
    Stru_BasicBlock_t* entry_block; // 入口基本块
    Stru_BasicBlock_t* exit_block;  // 出口基本块
} Stru_ControlFlowGraph_t;
```

## IR操作接口

### IR构建器接口

```c
typedef struct Stru_IrBuilderInterface_t {
    // 创建IR
    void* (*create_ir)(Eum_IrType type);
    
    // 添加指令
    bool (*add_instruction)(void* ir, Stru_IrInstruction_t* instr);
    
    // 创建指令
    Stru_IrInstruction_t* (*create_instruction)(Eum_IrInstructionType type);
    
    // 设置指令操作数
    bool (*set_instruction_operands)(Stru_IrInstruction_t* instr, 
                                     char* result, char* arg1, char* arg2, char* op);
    
    // 构建控制流图
    Stru_ControlFlowGraph_t* (*build_cfg)(void* ir);
    
    // 转换为SSA形式
    void* (*convert_to_ssa)(void* ir);
    
    // 销毁IR
    void (*destroy_ir)(void* ir);
} Stru_IrBuilderInterface_t;
```

### IR优化器接口

```c
typedef struct Stru_IrOptimizerInterface_t {
    // 常量折叠
    bool (*constant_folding)(void* ir);
    
    // 死代码消除
    bool (*dead_code_elimination)(void* ir);
    
    // 公共子表达式消除
    bool (*common_subexpression_elimination)(void* ir);
    
    // 强度削减
    bool (*strength_reduction)(void* ir);
    
    // 循环优化
    bool (*loop_invariant_code_motion)(void* ir);
    bool (*loop_unrolling)(void* ir, int factor);
    
    // 函数内联
    bool (*function_inlining)(void* ir, const char* function_name);
    
    // 应用所有优化
    bool (*apply_all_optimizations)(void* ir, Eum_OptimizationLevel level);
} Stru_IrOptimizerInterface_t;
```

```c
typedef struct Stru_DataFlowAnalyzerInterface_t {
    // 到达定义分析
    Stru_ReachingDefinition_t** (*analyze_reaching_definitions)(void* ir);
    
    // 活跃变量分析
    char*** (*analyze_live_variables)(void* ir, size_t* count);
    
    // 可用表达式分析
    char*** (*analyze_available_expressions)(void* ir, size_t* count);
    
    // 非常量表达式分析
    char*** (*analyze_very_busy_expressions)(void* ir, size_t* count);
} Stru_DataFlowAnalyzerInterface_t;
```
